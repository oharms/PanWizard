# LOCKED DECISIONS (from 02-research.md — NON-NEGOTIABLE)

1. **Stability invariant:** `Array.prototype.sort` is stable since V8 7.0 / Node 12 / ECMAScript 2019. Project requires Node ≥ 22.17.0 → stability is GUARANTEED. **DO NOT** add a tiebreaker index field to rows. **DO NOT** wrap with merge-sort. SORT-04 is satisfied by writing a comparator that returns 0 on equal keys and trusting the runtime.
2. **Comparator construction:** Build the closure ONCE outside `sort()`. `Array.prototype.sort(buildComparator(items))` — not inline lambda. (Pitfalls.md performance trap.)
3. **Read row[key] LITERALLY** — comparator does NOT do dot-path resolution. Why: post-projection rows have aliases as flat keys (e.g. `out: 42`) and group columns from `finalizeBuckets` as flat keys (e.g. `'usage.output_tokens': 42` — a key with a literal dot in it). dot-walking would break for the alias case AND would break for the group-column case (because `usage.output_tokens` as a flat key is NOT the same as walking `row.usage.output_tokens`). LITERAL access is correct.
4. **Nullish policy (LOCKED):** `null` and `undefined` both sort BEFORE any non-nullish value (NULLS FIRST under ASC). Documented in code. Symmetric: `null` sorts AFTER non-null under DESC because the direction multiplier flips the comparator result.
5. **NaN policy (LOCKED):** `NaN` sorts BEFORE non-NaN numerics. `NaN` vs `NaN` returns 0. This keeps the comparator total-ordered (no `>` / `<` weirdness from NaN propagation).
6. **Numeric vs numeric:** `a - b` (returns 0 exactly when `a === b`).
7. **String vs string:** `<` / `>` (works for the basic-Latin alphabetical use cases; locale-aware sorting is out of scope per "small precise grammar" principle).
8. **Mixed types:** lex-compare `String(a)` vs `String(b)`. Documented as "rare in practice" — production queries typically have homogeneous columns.
9. **No external imports.** `src/sort.js` is fully self-contained. Zero-deps invariant.

```javascript
// =============================================================================
// Sort — comparator builder for ORDER BY
// =============================================================================
//
// buildComparator(orderByItems) returns a function suitable for
// Array.prototype.sort. The returned comparator reads row[key] LITERALLY —
// no dot-path resolution — because POST-PROJECTION rows have aliases AND
// dotted group-column names as flat keys.
//
// Stability: Array.prototype.sort is stable since V8 7.0 / Node 12 / ES2019.
// The project's Node ≥ 22.17.0 floor guarantees this. DO NOT add tiebreaker
// index fields — research.md Anti-Pattern 3.
//
// Nullish policy: null and undefined sort BEFORE any non-nullish value (NULLS
// FIRST under ASC). DESC flips the direction so they sort LAST.
//
// NaN policy: NaN sorts BEFORE any non-NaN numeric, keeps total ordering.
//
// Mixed-type policy: lex-compare String(a) vs String(b). Documented as rare.
// =============================================================================

/**
 * Build a stable, multi-key comparator from an OrderByItem array.
 *
 * @param {Array<{ key: string, dir: 'ASC'|'DESC' }>} orderByItems
 * @returns {(rowA: Record<string, any>, rowB: Record<string, any>) => number}
 */
export function buildComparator(orderByItems) {
  return (rowA, rowB) => {
    for (const { key, dir } of orderByItems) {
      const cmp = compareValues(rowA[key], rowB[key]);
      if (cmp !== 0) return dir === 'DESC' ? -cmp : cmp;
    }
    return 0; // equal under all keys → stable native sort retains input order
  };
}

/**
 * Compare two scalar values. Returns negative / zero / positive (sort contract).
 *
 * Policy:
 *   - null / undefined → smaller than any non-nullish value (NULLS FIRST under ASC).
 *   - NaN              → smaller than any non-NaN number.
 *   - number vs number → a - b.
 *   - string vs string → < / >.
 *   - mixed types      → String(a) vs String(b) (lex compare).
 *
 * @param {any} a
 * @param {any} b
 * @returns {number}
 */
export function compareValues(a, b) {
  if (a === b) return 0;
  // Nullish handling — null and undefined are equivalent for sort purposes
  // (both "missing"); both sort before any non-nullish value under ASC.
  const aNull = a === null || a === undefined;
  const bNull = b === null || b === undefined;
  if (aNull && bNull) return 0;
  if (aNull) return -1;
  if (bNull) return 1;

  // Numeric vs numeric
  if (typeof a === 'number' && typeof b === 'number') {
    const aNaN = Number.isNaN(a);
    const bNaN = Number.isNaN(b);
    if (aNaN && bNaN) return 0;
    if (aNaN) return -1;
    if (bNaN) return 1;
    return a - b;
  }
  // String vs string
  if (typeof a === 'string' && typeof b === 'string') {
    return a < b ? -1 : 1; // we already know a !== b
  }
  // Mixed-type fallback
  const sa = String(a);
  const sb = String(b);
  if (sa === sb) return 0;
  return sa < sb ? -1 : 1;
}
```

Implementation notes:
- The comparator is built ONCE per `Array.prototype.sort()` call. Do NOT inline `array.sort((a,b) => …)` at the call site.
- `compareValues` handles all primitive types JSON can produce: number, string, boolean, null. (boolean is uncommon in ORDER BY but the mixed-type fallback handles it via `String(true)` vs `String(false)`.)
- The comparator reads `row[key]` LITERALLY — `key` is a flat string. For a query `ORDER BY out` where `out` is the alias, the post-projection row has `{ out: 42 }` and `row.out` returns 42. For a query `ORDER BY usage.output_tokens` over a SELECT that projected `usage.output_tokens` directly (no alias), the post-projection row has `{ 'usage.output_tokens': 42 }` (a literal-dot key from `project.js`'s output) and `row['usage.output_tokens']` returns 42. Both work with literal access.

import { test } from 'node:test';
import assert from 'node:assert/strict';
import { buildComparator, compareValues } from '../src/sort.js';

// ============================================================================
// compareValues — primitive comparison policy
// ============================================================================

test('compareValues: numeric vs numeric uses a - b', () => {
  assert.ok(compareValues(1, 2) < 0);
  assert.ok(compareValues(2, 1) > 0);
  assert.equal(compareValues(5, 5), 0);
});

test('compareValues: null sorts before any non-null (NULLS FIRST)', () => {
  assert.ok(compareValues(null, 5) < 0);
  assert.ok(compareValues(5, null) > 0);
  assert.equal(compareValues(null, null), 0);
});

test('compareValues: undefined sorts equal to null (both nullish)', () => {
  assert.equal(compareValues(undefined, null), 0);
  assert.ok(compareValues(undefined, 0) < 0);
});

test('compareValues: NaN sorts before any non-NaN numeric', () => {
  assert.ok(compareValues(NaN, 5) < 0);
  assert.ok(compareValues(5, NaN) > 0);
  assert.equal(compareValues(NaN, NaN), 0);
});

test('compareValues: string vs string uses lex order', () => {
  assert.ok(compareValues('a', 'b') < 0);
  assert.ok(compareValues('b', 'a') > 0);
  assert.equal(compareValues('foo', 'foo'), 0);
});

test('compareValues: mixed types fall back to String() comparison', () => {
  // String(1) === '1' < '2' === String(2) — but mixed cases are rare; just lock the fallback works.
  const r = compareValues(1, '2');
  assert.ok(r < 0);
});

// ============================================================================
// buildComparator — single-key (SORT-01)
// ============================================================================

test('buildComparator: single-key ASC sorts ascending (default direction handling)', () => {
  const cmp = buildComparator([{ key: 'a', dir: 'ASC' }]);
  const arr = [{ a: 3 }, { a: 1 }, { a: 2 }];
  arr.sort(cmp);
  assert.deepEqual(arr.map((r) => r.a), [1, 2, 3]);
});

test('buildComparator: single-key DESC sorts descending', () => {
  const cmp = buildComparator([{ key: 'a', dir: 'DESC' }]);
  const arr = [{ a: 1 }, { a: 3 }, { a: 2 }];
  arr.sort(cmp);
  assert.deepEqual(arr.map((r) => r.a), [3, 2, 1]);
});

// ============================================================================
// buildComparator — multi-key (SORT-02)
// ============================================================================

test('buildComparator: multi-key ASC+ASC tiebreaks via second key', () => {
  const cmp = buildComparator([
    { key: 'a', dir: 'ASC' },
    { key: 'b', dir: 'ASC' },
  ]);
  const arr = [
    { a: 1, b: 'y' },
    { a: 1, b: 'x' },
    { a: 2, b: 'z' },
  ];
  arr.sort(cmp);
  assert.deepEqual(arr, [
    { a: 1, b: 'x' },
    { a: 1, b: 'y' },
    { a: 2, b: 'z' },
  ]);
});

test('buildComparator: multi-key ASC+DESC mixes directions independently', () => {
  const cmp = buildComparator([
    { key: 'a', dir: 'ASC' },
    { key: 'b', dir: 'DESC' },
  ]);
  const arr = [
    { a: 1, b: 'x' },
    { a: 1, b: 'y' },
    { a: 2, b: 'a' },
  ];
  arr.sort(cmp);
  // a=1 group sorted DESC by b: y before x. Then a=2 last.
  assert.deepEqual(arr, [
    { a: 1, b: 'y' },
    { a: 1, b: 'x' },
    { a: 2, b: 'a' },
  ]);
});

// ============================================================================
// SORT-04 — STABILITY GATE
// ============================================================================

test('SORT-04: equal-keyed rows retain input order (stable sort, NO tiebreaker added)', () => {
  // Phase 2 success criterion #1 anchor: ORDER BY out DESC LIMIT 5 with stable tie-breaking.
  const cmp = buildComparator([{ key: 'out', dir: 'DESC' }]);
  const arr = [
    { agent: 'a', out: 5 },
    { agent: 'b', out: 5 },
    { agent: 'c', out: 5 },
    { agent: 'd', out: 10 },
  ];
  arr.sort(cmp);
  // After sort: agent 'd' first (highest out=10), then a/b/c in INPUT ORDER for the out=5 group.
  assert.deepEqual(arr.map((r) => r.agent), ['d', 'a', 'b', 'c']);
});

test('SORT-04: stability holds for many equal-key rows (regression gate)', () => {
  const cmp = buildComparator([{ key: 'k', dir: 'ASC' }]);
  const arr = [];
  for (let i = 0; i < 50; i++) arr.push({ k: 1, idx: i });
  arr.sort(cmp);
  // All k values equal → input order preserved → idx 0..49 in order.
  for (let i = 0; i < 50; i++) {
    assert.equal(arr[i].idx, i);
  }
});

// ============================================================================
// SORT-03 — comparator reads LITERAL flat keys (post-projection contract)
// ============================================================================

test('SORT-03: comparator reads row[key] LITERALLY (does NOT dot-walk into nested objects)', () => {
  // Post-projection rows from finalizeBuckets store dotted group-column names as
  // FLAT keys. `usage.output_tokens` is a literal property name, NOT a path to walk.
  const cmp = buildComparator([{ key: 'usage.output_tokens', dir: 'DESC' }]);
  const arr = [
    { 'usage.output_tokens': 100 },
    { 'usage.output_tokens': 50 },
    { 'usage.output_tokens': 200 },
  ];
  arr.sort(cmp);
  assert.deepEqual(arr.map((r) => r['usage.output_tokens']), [200, 100, 50]);
});

test('SORT-03: ORDER BY on alias key works (alias is a flat key on post-projection row)', () => {
  const cmp = buildComparator([{ key: 'out', dir: 'DESC' }]);
  const arr = [
    { agent: 'a', out: 30 },
    { agent: 'b', out: 100 },
    { agent: 'c', out: 5 },
  ];
  arr.sort(cmp);
  assert.deepEqual(arr.map((r) => r.agent), ['b', 'a', 'c']);
});

// ============================================================================
// Edge cases
// ============================================================================

test('buildComparator: empty orderBy returns 0 always (sort.js contract — caller should not call sort)', () => {
  const cmp = buildComparator([]);
  assert.equal(cmp({ a: 1 }, { a: 2 }), 0);
});

test('buildComparator: nullish values in sorted column sort to the start under ASC', () => {
  const cmp = buildComparator([{ key: 'a', dir: 'ASC' }]);
  const arr = [{ a: 3 }, { a: null }, { a: 1 }, { a: undefined }, { a: 2 }];
  arr.sort(cmp);
  // null and undefined are equal-nullish; their relative order is preserved (stable).
  // Expected: [null, undefined, 1, 2, 3].
  assert.equal(arr[0].a, null);
  assert.equal(arr[1].a, undefined);
  assert.deepEqual(arr.slice(2).map((r) => r.a), [1, 2, 3]);
});

test('buildComparator: nullish values sort to the end under DESC', () => {
  const cmp = buildComparator([{ key: 'a', dir: 'DESC' }]);
  const arr = [{ a: 3 }, { a: null }, { a: 1 }, { a: 2 }];
  arr.sort(cmp);
  // DESC flips: 3, 2, 1, then nullish at the end.
  assert.deepEqual(arr.slice(0, 3).map((r) => r.a), [3, 2, 1]);
  assert.equal(arr[3].a, null);
});
