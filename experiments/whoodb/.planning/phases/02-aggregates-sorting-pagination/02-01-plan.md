# LOCKED DECISIONS (from 02-research.md — NON-NEGOTIABLE)

1. **Aggregate AST shape:** `{ type: 'Aggregate', func: 'COUNT'|'SUM'|'AVG'|'MIN'|'MAX', arg: 'STAR' | ColumnRef }`. Only `COUNT` may take `arg: 'STAR'`; `SUM(*)` etc. are parse errors.
2. **OrderByItem AST shape:** `{ type: 'OrderByItem', key: string, dir: 'ASC'|'DESC' }`. Default direction when keyword absent: `'ASC'`.
3. **LimitOffset AST shape:** `{ type: 'LimitOffset', count: number, offset: number }`. `offset` defaults to `0` when no `OFFSET` clause.
4. **AGGREGATE_FUNCS set in parser.js** = `new Set(['COUNT', 'SUM', 'AVG', 'MIN', 'MAX'])`. Aggregate function names are NOT lexer keywords; they remain `IDENT` tokens. Detection happens in `parseProjectionItem`: if peek is an IDENT whose uppercased value is in AGGREGATE_FUNCS AND the next token is `(`, route to `parseAggregate`.
5. **`SELECT count FROM f` (where `count` is a column name) MUST still parse as a regular ColumnRef** — the IDENT-followed-by-`(` check disambiguates.
6. **Implicit GROUP BY ()** — when projection contains any Aggregate AND ast.groupBy is null, the analyzer sets `plan.implicitGroupBy = true` and `plan.groupBy = []` (empty array, NOT null). The PARSE-09 rule must NOT reject all-aggregate projections that lack GROUP BY.
7. **ORDER BY post-projection scope (SORT-03):** the analyzer builds the set of projected names — for `Wildcard` projection the scope is "any column" (skip the check), for `ColumnRef` the projected name is `node.name`, for `Alias` the projected name is `node.alias`. ORDER BY keys not in this set throw `ParseError` with message containing `ORDER BY` and the unresolved column name. With `Wildcard` projection, the scope check is skipped (any name is acceptable post-projection because the row is the raw row).
8. **PAGE-01/PAGE-02 non-negative integer:** the parser MUST validate `Number.isInteger(value) && value >= 0` for both LIMIT and OFFSET arguments at parse time. Float NUMBER tokens (e.g. `1.5`) and negative numbers throw `ParseError`. Note that the lexer emits unsigned NUMBER tokens; a leading `-` would be a separate OPERATOR token — handle that case by detecting OPERATOR `-` directly before NUMBER and rejecting.

function parseAggregate(state) {
  const fnTok = consume(state); // IDENT — function name
  const fnName = fnTok.value.toUpperCase();
  expect(state, 'PUNCT', '(');
  const next = peek(state);
  let arg;
  if (next && next.type === 'WILDCARD') {

      throw new ParseError(
        `parse error at column ${next.col}: ${fnName}(*) is not allowed; use ${fnName}(<field>)`,
        next.col,
      );

    consume(state);
    arg = 'STAR';
  } else {
    const idTok = expect(state, 'IDENT');
    arg = columnRef(idTok.value);
  }
  expect(state, 'PUNCT', ')');
  return aggregate(fnName, arg);
}

(d) Modify `parseProjectionItem(state)` to detect aggregate calls. Replace the existing function body with:
```javascript
function parseProjectionItem(state) {
  const idTok = peek(state);
  if (!idTok || idTok.type !== 'IDENT') {
    // Fall through to expect() which throws the standard "expected IDENT, got ..." error.
    expect(state, 'IDENT');
  }
  // Aggregate? IDENT (uppercased) in AGGREGATE_FUNCS AND next token is '('.
  const lookahead = peek(state, 1);
  let projected;
  if (
    AGGREGATE_FUNCS.has(idTok.value.toUpperCase()) &&
    lookahead && lookahead.type === 'PUNCT' && lookahead.value === '('
  ) {
    projected = parseAggregate(state); // returns Aggregate node
  } else {
    consume(state);
    projected = columnRef(idTok.value);
  }

  // Optional AS alias (works for both ColumnRef and Aggregate)
  const next = peek(state);
  if (next && next.type === 'KEYWORD' && next.value === 'AS') {
    consume(state);
    const aliasTok = expect(state, 'IDENT');
    return alias(aliasTok.value, projected);
  }
  return projected;
}
```

Note: `idTok.value.toUpperCase()` is safe because aggregate-function name detection is case-insensitive (matches lexer's case-insensitive KEYWORD policy). But note the IDENT preserves its original case in `value` — so a lowercase `count(*)` MUST also be detected. The `.toUpperCase()` covers this.

function parseOrderBy(state) {
  const tok = peek(state);
  if (!(tok && tok.type === 'KEYWORD' && tok.value === 'ORDER')) {
    return null;
  }
  consume(state); // ORDER
  expect(state, 'KEYWORD', 'BY');
  const items = [];
  items.push(parseOrderByItem(state));
  while (peek(state)?.type === 'PUNCT' && peek(state).value === ',') {
    consume(state);
    items.push(parseOrderByItem(state));
  }
  return items;
}

function parseOrderByItem(state) {
  const keyTok = expect(state, 'IDENT');
  let dir = 'ASC';
  const next = peek(state);
  if (next && next.type === 'KEYWORD' && (next.value === 'ASC' || next.value === 'DESC')) {
    dir = next.value;
    consume(state);
  }
  return orderByItem(keyTok.value, dir);
}

function parseLimit(state) {
  const tok = peek(state);
  if (!(tok && tok.type === 'KEYWORD' && tok.value === 'LIMIT')) {
    return null;
  }
  consume(state); // LIMIT
  const count = parseNonNegativeInt(state, 'LIMIT');
  let offset = 0;
  const off = peek(state);
  if (off && off.type === 'KEYWORD' && off.value === 'OFFSET') {
    consume(state);
    offset = parseNonNegativeInt(state, 'OFFSET');
  }
  return limitOffset(count, offset);
}

function parseNonNegativeInt(state, ctxLabel) {
  // Reject leading minus sign explicitly (lexer emits '-' as OPERATOR, NUMBER as unsigned).
  const peeked = peek(state);
  if (peeked && peeked.type === 'OPERATOR' && peeked.value === '-') {
    throw new ParseError(
      `parse error at column ${peeked.col}: ${ctxLabel} requires a non-negative integer`,
      peeked.col,
    );
  }
  const numTok = expect(state, 'NUMBER');
  if (!Number.isInteger(numTok.value) || numTok.value < 0) {
    throw new ParseError(
      `parse error at column ${numTok.col}: ${ctxLabel} requires a non-negative integer (got ${numTok.value})`,
      numTok.col,
    );
  }
  return numTok.value;
}

```javascript
export function analyze(ast) {
  if (!ast || ast.type !== 'SelectStmt') {
    throw new ParseError('analyze: input is not a SelectStmt', 0);
  }

  const projection = ast.projection;
  let groupBy = ast.groupBy ?? null;
  let implicitGroupBy = false;

  // Detect aggregates in the projection (unwrap Alias to inspect inner expr).
  const hasAggregate = projection.some((item) => {
    const inner = item.type === 'Alias' ? item.expr : item;
    return isAggregate(inner);
  });

  // ---- PARSE-09 + implicit GROUP BY () handling ----
  if (groupBy === null && hasAggregate) {
    // Implicit GROUP BY () — single-bucket case. SELECT * is still illegal
    // when aggregates are mixed in (would be ambiguous).
    for (const item of projection) {
      if (item.type === 'Wildcard') {
        throw new ParseError('SELECT * is not allowed alongside aggregate functions', 0);
      }
      const inner = item.type === 'Alias' ? item.expr : item;
      if (inner.type === 'ColumnRef') {
        // Bare ColumnRef alongside aggregates and no GROUP BY → ambiguous.
        throw new ParseError(
          `column "${inner.name}" must appear in GROUP BY or be wrapped in an aggregate`,
          0,
        );
      }
    }
    groupBy = [];
    implicitGroupBy = true;
  } else if (groupBy !== null) {
    // Explicit GROUP BY — same rule as Phase 1, but now Aggregates pass the check.
    for (const item of projection) {
      if (item.type === 'Wildcard') {
        throw new ParseError('SELECT * is not allowed with GROUP BY', 0);
      }
    }
    const groupNames = new Set(groupBy.map((c) => c.name));
    for (const item of projection) {
      const inner = item.type === 'Alias' ? item.expr : item;
      if (inner.type === 'ColumnRef') {
        if (!groupNames.has(inner.name)) {
          throw new ParseError(
            `column "${inner.name}" must appear in GROUP BY or be wrapped in an aggregate`,
            0,
          );
        }
      } else if (isAggregate(inner)) {
        // OK
      }
    }
  }

  // ---- SORT-03: ORDER BY post-projection scope check ----
  const orderBy = ast.orderBy ?? null;
  if (orderBy !== null) {
    const isWildcard = projection.length === 1 && projection[0].type === 'Wildcard';
    if (!isWildcard) {
      const projectedNames = new Set();
      for (const item of projection) {
        if (item.type === 'Alias') projectedNames.add(item.alias);
        else if (item.type === 'ColumnRef') projectedNames.add(item.name);
        // Bare Aggregate without alias — its projected name is undefined here;
        // the executor uses a synthesized name. Phase 2 keeps it strict:
        // an Aggregate-without-alias has no addressable ORDER BY name.
      }
      for (const item of orderBy) {
        if (!projectedNames.has(item.key)) {
          throw new ParseError(
            `ORDER BY key "${item.key}" is not in the SELECT list (must be a projected column or alias)`,
            0,
          );
        }
      }
    }
    // Wildcard projection: allow any ORDER BY key — comparator will read from the raw row.
  }

  return {
    scan: { source: ast.from },
    filter: ast.where ?? null,
    projection,
    groupBy,
    implicitGroupBy,
    sort: orderBy,
    limit: ast.limit ?? null,
  };
}
```

Note the contract change: `plan.groupBy` is now either `null` (no aggregation), `[]` (implicit GROUP BY ()), or a non-empty `ColumnRef[]` (explicit GROUP BY). The `implicitGroupBy: true` flag distinguishes the empty-array case from a deliberate-but-bizarre `GROUP BY ()` syntax (which we do not support).
