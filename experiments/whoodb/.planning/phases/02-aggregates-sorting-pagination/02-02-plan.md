# LOCKED DECISIONS (from 02-research.md — NON-NEGOTIABLE)

1. **Group key formula:** `JSON.stringify(groupValues.map(v => [typeof v, v]))`. NEVER plain `JSON.stringify(groupValues)`. The type-tag distinguishes integer `1` (`["number",1]`) from string `"1"` (`["string","1"]`) from `null` (`["object",null]`) from `undefined` (`["undefined",null]`).
2. **Per-slot kinds:** COUNT_STAR, COUNT_FIELD, SUM, AVG, MIN, MAX. Bucket shape: `{ groupValues: any[], aggregates: Record<outputKey, slot> }`.
3. **AGGR-08 type-mismatch policy:** SUM/AVG/MIN/MAX of a non-numeric value (i.e. `typeof !== 'number' || Number.isNaN(v)`) emits ONE stderr warning per occurrence and skips that row's contribution. Null/undefined fields are silently skipped (no warning) — they are not "wrong type", they are "missing".
4. **AGGR-09 AVG empty:** finalize returns `null` when `slot.count === 0` (zero numeric inputs observed).
5. **MIN/MAX empty:** finalize returns `null` when no value observed.
6. **MIN/MAX numeric only (Phase 2 scope):** non-numeric values trigger the AGGR-08 warning + skip; string MIN/MAX is a v2 feature explicitly deferred per research.md "Don't Hand-Roll" table.
7. **Warning prefix:** `whoodb:` (matches the malformed-JSONL channel from Phase 1's evaluator).
8. **No external imports.** `src/accumulator.js` imports ONLY from `./project.js` (for `resolvePath`). Zero-deps invariant.

```javascript
// =============================================================================
// Accumulator — in-memory GROUP BY engine for whoodb
// =============================================================================
//
// The accumulator owns the type-tagged group key and the per-aggregate slot
// state machine. It is pure (no I/O) — the evaluator drives it row-by-row
// inside the buffered execution path.
//
// LOCKED group-key formula (research.md Pitfall 3 / AGGR-06):
//   JSON.stringify(groupValues.map(v => [typeof v, v]))
// This distinguishes:
//   {agent: 1}     → key '[["number",1]]'
//   {agent: "1"}   → key '[["string","1"]]'
//   {agent: null}  → key '[["object",null]]'
//   {agent: undef} → key '[["undefined",null]]'
//
// AGGR-08 type-mismatch policy:
//   SUM/AVG/MIN/MAX of a non-numeric value (typeof !== 'number' || NaN)
//   emits a `whoodb: <FUNC> non-numeric value at field "<path>" ...` warning
//   to stderr and skips the row's contribution. Null/undefined fields are
//   silently skipped (they are "missing", not "wrong type").
// =============================================================================

import { resolvePath } from './project.js';

/**
 * Compute the group-key tuple for a row.
 *
 * @param {Record<string, any>} row
 * @param {Array<{ type: 'ColumnRef', name: string, path: string[] }>} groupByCols
 * @returns {{ vals: any[], key: string }}
 */
export function groupKey(row, groupByCols) {
  if (groupByCols.length === 0) {
    // Implicit GROUP BY () — single bucket. Use a stable empty-tuple key.
    return { vals: [], key: '[]' };
  }
  const vals = groupByCols.map((c) => resolvePath(row, c.path));
  // Type-tag the values to make integer 1 vs string "1" distinct buckets.
  const tagged = vals.map((v) => [typeof v, v]);
  return { vals, key: JSON.stringify(tagged) };
}

/**
 * Build a fresh slot for an Aggregate node. The slot carries running state
 * sufficient to finalize the aggregate without revisiting input rows.
 *
 * @param {{ type: 'Aggregate', func: string, arg: any }} agg
 * @returns {object} slot
 */
export function newSlot(agg) {
  if (agg.func === 'COUNT' && agg.arg === 'STAR') {
    return { kind: 'COUNT_STAR', count: 0 };
  }
  // For non-STAR aggregates, agg.arg is a ColumnRef.
  const path = agg.arg.path;
  const fieldName = agg.arg.name;
  if (agg.func === 'COUNT') return { kind: 'COUNT_FIELD', count: 0, path, fieldName };
  if (agg.func === 'SUM')   return { kind: 'SUM',   sum: 0, path, fieldName };
  if (agg.func === 'AVG')   return { kind: 'AVG',   sum: 0, count: 0, path, fieldName };
  if (agg.func === 'MIN')   return { kind: 'MIN',   value: undefined, hasValue: false, path, fieldName };
  if (agg.func === 'MAX')   return { kind: 'MAX',   value: undefined, hasValue: false, path, fieldName };
  throw new Error(`accumulator: unsupported aggregate function ${agg.func}`);
}

/**
 * Format and emit a type-mismatch warning to stderr.
 * Exposed (not exported) so tests can spy on stderr instead of monkey-patching.
 */
function warnNonNumeric(funcName, fieldName, value) {
  process.stderr.write(
    `whoodb: ${funcName} non-numeric value at field "${fieldName}" (got ${typeof value}), skipping row\n`,
  );
}

/**
 * Update one slot with one row.
 *
 * @param {object} slot — mutated in place
 * @param {Record<string, any>} row
 */
export function updateSlot(slot, row) {
  if (slot.kind === 'COUNT_STAR') {
    slot.count++;
    return;
  }
  const v = resolvePath(row, slot.path);
  if (v === null || v === undefined) {
    // Null/undefined: missing, not wrong-type. Silently skipped per AGGR-02 / AGGR-08 distinction.
    return;
  }
  if (slot.kind === 'COUNT_FIELD') {
    slot.count++;
    return;
  }
  // SUM/AVG/MIN/MAX require numeric.
  if (typeof v !== 'number' || Number.isNaN(v)) {
    warnNonNumeric(slot.kind, slot.fieldName, v);
    return;
  }
  if (slot.kind === 'SUM') {
    slot.sum += v;
    return;
  }
  if (slot.kind === 'AVG') {
    slot.sum += v;
    slot.count++;
    return;
  }
  if (slot.kind === 'MIN') {
    if (!slot.hasValue || v < slot.value) {
      slot.value = v;
      slot.hasValue = true;
    }
    return;
  }
  if (slot.kind === 'MAX') {
    if (!slot.hasValue || v > slot.value) {
      slot.value = v;
      slot.hasValue = true;
    }
    return;
  }
  throw new Error(`accumulator: unknown slot kind ${slot.kind}`);
}

/**
 * Reduce a slot to its final scalar value.
 *
 * @param {object} slot
 * @returns {number|null}
 */
export function finalizeSlot(slot) {
  if (slot.kind === 'COUNT_STAR' || slot.kind === 'COUNT_FIELD') return slot.count;
  if (slot.kind === 'SUM') return slot.sum;
  if (slot.kind === 'AVG') return slot.count === 0 ? null : slot.sum / slot.count;
  if (slot.kind === 'MIN' || slot.kind === 'MAX') return slot.hasValue ? slot.value : null;
  throw new Error(`accumulator: unknown slot kind ${slot.kind}`);
}

/**
 * Drive accumulation across an iterable of rows. Returns a Map<keyString, bucket>.
 *
 * @param {Iterable<Record<string, any>>} rows — raw rows that have already passed WHERE
 * @param {Array<{ type: 'ColumnRef', name: string, path: string[] }>} groupByCols
 * @param {Record<string, { type: 'Aggregate', func: string, arg: any }>} aggregateDefs
 *        — map from output-column name (alias or synthesized) to Aggregate node
 * @returns {Map<string, { groupValues: any[], aggregates: Record<string, object> }>}
 */
export function accumulateGroups(rows, groupByCols, aggregateDefs) {
  const buckets = new Map();
  for (const row of rows) {
    const { vals, key } = groupKey(row, groupByCols);
    let bucket = buckets.get(key);
    if (!bucket) {
      bucket = {
        groupValues: vals,
        aggregates: {},
      };
      for (const [outKey, agg] of Object.entries(aggregateDefs)) {
        bucket.aggregates[outKey] = newSlot(agg);
      }
      buckets.set(key, bucket);
    }
    for (const slot of Object.values(bucket.aggregates)) {
      updateSlot(slot, row);
    }
  }
  return buckets;
}

/**
 * Reduce buckets to one synthesized row per group: { ...groupColumns, ...aggregates }.
 * The synthesized rows are intermediate — the evaluator passes them through `project()`
 * next, which is responsible for dropping any group columns the user did not select.
 *
 * @param {Map<string, { groupValues: any[], aggregates: Record<string, object> }>} buckets
 * @param {Array<{ type: 'ColumnRef', name: string, path: string[] }>} groupByCols
 * @returns {Array<Record<string, any>>}
 */
export function finalizeBuckets(buckets, groupByCols) {
  const result = [];
  for (const bucket of buckets.values()) {
    const synthRow = {};
    // Reconstruct group-column values into the synth row using the dotted name as the key.
    // (Works for both flat and dot-nested group keys; the synth row mirrors what the
    // raw-row projection would have produced for those columns.)
    groupByCols.forEach((col, i) => {
      synthRow[col.name] = bucket.groupValues[i];
    });
    for (const [outKey, slot] of Object.entries(bucket.aggregates)) {
      synthRow[outKey] = finalizeSlot(slot);
    }
    result.push(synthRow);
  }
  return result;
}
```

Key implementation notes (also enforce in code review):
- `groupKey` with `groupByCols.length === 0` returns `{ vals: [], key: '[]' }` — needed for the implicit-GROUP-BY-() case (single bucket).
- `updateSlot` silently skips null/undefined values for SUM/AVG/MIN/MAX (those are MISSING, not WRONG TYPE). Only `typeof !== 'number' || Number.isNaN` triggers the warning.
- `accumulateGroups` accepts an iterable so the evaluator can pass either an array (buffered) or a generator (if it ever wants to stream into the accumulator). Phase 2 uses the array path.
- The `synthRow` from `finalizeBuckets` carries the group-column dotted names AS-IS — this matches what `project()` expects when projecting `agent` (a ColumnRef) from the synth row whose key is `'agent'`. For dot-nested group keys (e.g., `GROUP BY usage.input_tokens`), `synthRow['usage.input_tokens'] = N` is set, and the projector will read that key directly via `resolvePath` — but `resolvePath` splits on `.` and would walk into `usage` if it existed. **Potential mismatch:** synth rows put dotted names as flat keys, but `project()`'s `resolvePath` splits on `.`. The evaluator (Plan 04) handles this — for grouped queries, projection of a group-column reads from the synth row by FLAT name, which Plan 04 wires up. The accumulator just promises: synth rows have the group-column DOTTED names as flat keys. Document this in the JSDoc on `finalizeBuckets`.

import assert from 'node:assert/strict';
import {
  groupKey, newSlot, updateSlot, finalizeSlot,
  accumulateGroups, finalizeBuckets,
} from '../src/accumulator.js';

// Helper: build a ColumnRef stand-in (matches the AST factory shape).
function col(name) {
  return { type: 'ColumnRef', name, path: name.split('.') };
}

// Helper: build an Aggregate stand-in.
function agg(func, arg) {
  return { type: 'Aggregate', func, arg };
}

// Helper: capture stderr writes synchronously during fn.
function captureStderr(fn) {
  const original = process.stderr.write.bind(process.stderr);
  const captured = [];
  process.stderr.write = (chunk) => {
    captured.push(typeof chunk === 'string' ? chunk : chunk.toString('utf8'));
    return true;
  };
  try {
    fn();
  } finally {
    process.stderr.write = original;
  }
  return captured.join('');
}

test('groupKey: integer 1 and string "1" produce DIFFERENT keys (AGGR-06)', () => {
  const k1 = groupKey({ a: 1 }, [col('a')]);

  assert.notEqual(k1.key, k2.key);

test('groupKey: null and missing field produce DIFFERENT keys', () => {

  assert.equal(k1.key, '[]');
