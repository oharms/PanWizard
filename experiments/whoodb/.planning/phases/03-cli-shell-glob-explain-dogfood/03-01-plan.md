Create `src/from-resolver.js` exporting a single async function `resolveFrom(source, cwd = process.cwd())`.

Implementation:

```js
import { glob } from 'node:fs/promises';
import { sep as PATH_SEP } from 'node:path';

const GLOB_CHARS = ['*', '?', '['];

function hasGlobChars(s) {
  return GLOB_CHARS.some((c) => s.includes(c));
}

function toForwardSlashes(p) {
  // Defensive on Windows: glob yields backslashes; normalize for cross-platform consistency.
  return PATH_SEP === '/' ? p : p.split(PATH_SEP).join('/');
}

/**
 * Resolve a FROM clause string into an array of normalized (forward-slash) file paths.
 *
 * @param {string} source - the FROM target (literal path or glob pattern)
 * @param {string} [cwd]  - working directory for glob resolution (defaults to process.cwd())
 * @returns {Promise<string[]>}
 * @throws if source is empty or if a glob matches zero files
 */
export async function resolveFrom(source, cwd = process.cwd()) {
  if (typeof source !== 'string' || source.length === 0) {
    throw new Error('resolveFrom: source must be a non-empty string');
  }

  // Defensive: strip surrounding single quotes if a caller passed them verbatim.
  // (Lexer normally strips them, but keep this for future-proofing — see research Pitfall 5.)
  const raw =
    source.length >= 2 && source.startsWith("'") && source.endsWith("'")
      ? source.slice(1, -1)
      : source;

  if (!hasGlobChars(raw)) {
    return [toForwardSlashes(raw)];
  }

  const matched = [];
  for await (const f of glob(raw, { cwd })) {
    matched.push(toForwardSlashes(f));
  }

  if (matched.length === 0) {
    throw new Error(`no files matched: ${raw}`);
  }

  // Sort for deterministic ordering across platforms (glob iteration order is not guaranteed).
  matched.sort();
  return matched;
}
```

Why these decisions:
- `cwd` parameter (default `process.cwd()`): glob expands relative patterns against cwd; making it injectable lets tests run with a fixed working directory.
- `matched.sort()`: ensures deterministic order across OS/filesystem — important for tests asserting array contents AND for reproducible GROUP BY across multi-file queries.
- Forward-slash normalization: keeps test assertions and `--explain` output consistent between Windows and POSIX.
- Zero-match throws (does NOT silently return []): loud failure beats silent zero-row output.

PRESERVE the existing PAGE-03 early-termination semantics: the `passed`, `emitted`, `offset`, `count` counters MUST persist ACROSS files (do NOT reset between files), and the `break` after `emitted >= count` MUST exit BOTH loops (use a labeled break or a sentinel flag).
4. In the buffered path, wrap the row-collection loop similarly: `for (const file of paths) { for await (const row of streamRows(file, plan, opts)) rows.push(row); }`.
5. Update the JSDoc `@param` to `@param {string | string[]} filePaths`.

CRITICAL — preserve existing behavior:
- Single-string callers (existing Phase 1+2 tests) must work unchanged. The `Array.isArray` branch handles this.
- The `streamRows` helper itself stays as-is (still accepts a single string per call).
- The aggregator (`accumulateGroups` / `finalizeBuckets`) is called ONCE after all files are collected, so GROUP BY merges naturally across files.
- PAGE-03 early termination: when `emitted >= count`, break out of BOTH the inner `for await` and the outer `for` loop. Use a labeled break:

```js
async function* runStreaming(plan, filePaths, opts) {
  const limit = plan.limit;
  const count = limit ? limit.count : Infinity;
  const offset = limit ? limit.offset : 0;
  let passed = 0;
  let emitted = 0;

  outer: for (const file of filePaths) {
    for await (const row of streamRows(file, plan, opts)) {
      passed++;
      if (passed <= offset) continue;
      yield project(plan.projection, row);
      emitted++;
      if (emitted >= count) break outer;
    }
  }
}
```

For the buffered path, no labeled break is needed — collect all rows, then process:

```js
const rows = [];
for (const file of filePaths) {
  for await (const row of streamRows(file, plan, opts)) {
    rows.push(row);
  }
}
```

```js
export async function* execute(plan, filePath, opts = {}) {
  const filePaths = Array.isArray(filePath) ? filePath : [filePath];
  const isBuffered = ...;
  if (!isBuffered) { yield* runStreaming(plan, filePaths, opts); return; }
  // ... rest uses filePaths instead of filePath
}
```
