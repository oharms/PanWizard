#!/usr/bin/env node
// scripts/augment-tokens-jsonl.js
//
// One-shot (idempotent) migration: read .planning/metrics/tokens.jsonl,
// add a nested `usage: { input_tokens, output_tokens, cache_read_tokens,
// cache_write_tokens }` field to every row that doesn't already have one,
// preserve all existing flat fields, write back atomically.
//
// Why:
//   .claude/hooks/pan-cost-logger.js (framework code) writes flat fields only.
//   The Phase 3 dogfood query uses SUM(usage.output_tokens) — nested.
//   This script bridges the two without modifying framework code.
//
// Usage:
//   node scripts/augment-tokens-jsonl.js            # default path
//   node scripts/augment-tokens-jsonl.js path/to.jsonl
//
// Idempotent: rows already containing a `usage` field are passed through
// unchanged.

import { readFileSync, writeFileSync } from 'node:fs';
import { resolve } from 'node:path';

const DEFAULT_PATH = '.planning/metrics/tokens.jsonl';
const USAGE_KEYS = ['input_tokens', 'output_tokens', 'cache_read_tokens', 'cache_write_tokens'];

function augmentRow(row) {
  if (row.usage && typeof row.usage === 'object') {
    return row; // already augmented — skip
  }
  const usage = {};
  for (const k of USAGE_KEYS) {
    if (k in row) usage[k] = row[k];
  }
  return { ...row, usage };
}

function main() {
  const target = resolve(process.argv[2] || DEFAULT_PATH);
  const raw = readFileSync(target, 'utf8');
  const lines = raw.split('\n');
  const out = [];
  let augmented = 0, skipped = 0;
  for (const line of lines) {
    const trimmed = line.trim();
    if (trimmed === '') { out.push(''); continue; }
    let row;
    try {
      row = JSON.parse(trimmed);
    } catch {
      // Preserve unparseable lines verbatim — don't lose data.
      out.push(line);
      continue;
    }
    const before = !!row.usage;
    const next = augmentRow(row);
    if (before) skipped++; else augmented++;
    out.push(JSON.stringify(next));
  }
  writeFileSync(target, out.join('\n'), 'utf8');
  process.stderr.write(`augment-tokens-jsonl: ${augmented} rows augmented, ${skipped} already had usage field, target=${target}\n`);
}

main();
```

// test/dogfood.test.js — TEST-03 + SC-4 validation
//
// Runs the locked dogfood query against the real PAN tokens log
// (.planning/metrics/tokens.jsonl) and asserts plausible aggregate output.
//
// Locked query (do NOT modify — verbatim from requirements.md TEST-03):
//   SELECT agent, COUNT(*) AS calls, SUM(usage.output_tokens) AS out
//   FROM '.planning/metrics/tokens.jsonl'
//   GROUP BY agent ORDER BY out DESC LIMIT 10

import { test } from 'node:test';
import assert from 'node:assert/strict';
import { existsSync } from 'node:fs';
import { main } from '../src/cli.js';

const DOGFOOD_SQL =
  "SELECT agent, COUNT(*) AS calls, SUM(usage.output_tokens) AS out " +
  "FROM '.planning/metrics/tokens.jsonl' " +
  "GROUP BY agent ORDER BY out DESC LIMIT 10";

const TOKENS_PATH = '.planning/metrics/tokens.jsonl';

function captureStdout(fn) {
  const original = process.stdout.write.bind(process.stdout);
  let buf = '';
  process.stdout.write = (chunk) => { buf += chunk; return true; };
  return Promise.resolve(fn()).finally(() => {
    process.stdout.write = original;
  }).then((code) => ({ out: buf, code }));
}

test('dogfood: SC-4 locked query returns 1-10 rows with non-zero `out`', async (t) => {
  if (!existsSync(TOKENS_PATH)) {
    t.skip(`${TOKENS_PATH} not found — skipping dogfood test`);
    return;
  }

  const { out, code } = await captureStdout(() => main(['query', DOGFOOD_SQL]));

  assert.equal(code, 0, 'dogfood query must exit 0');

  const lines = out.trim().split('\n').filter(Boolean);
  assert.ok(lines.length >= 1, 'expected at least 1 result row');
  assert.ok(lines.length <= 10, 'LIMIT 10 must cap the result set');

  const rows = lines.map((line) => JSON.parse(line));

  // Each row must have agent (string), calls (number ≥ 1), out (number).
  for (const r of rows) {
    assert.equal(typeof r.agent, 'string', 'each row has string agent');
    assert.equal(typeof r.calls, 'number', 'each row has numeric calls');
    assert.ok(r.calls >= 1, 'calls is at least 1');
    assert.equal(typeof r.out, 'number', 'each row has numeric out');
  }

  // SC-4 plausibility: at least ONE row's `out` is non-zero.
  // (If the augmentation worked, SUM(usage.output_tokens) cannot be 0 across all groups
  // unless every value in tokens.jsonl is 0, which the file inspection showed is not the case.)
  const totalOut = rows.reduce((s, r) => s + (r.out || 0), 0);
  assert.ok(totalOut > 0, 'expected at least one non-zero out value (proves usage.output_tokens is being read)');

  // ORDER BY out DESC: results must be sorted descending by `out`.
  for (let i = 1; i < rows.length; i++) {
    assert.ok(
      rows[i].out <= rows[i - 1].out,
      `row ${i} out=${rows[i].out} should be ≤ row ${i - 1} out=${rows[i - 1].out} (DESC order)`,
    );
  }
});

test('dogfood: --explain on locked query exits 0 without file I/O (SC-1 check applied to dogfood)', async () => {
  const { out, code } = await captureStdout(() =>
    main(['query', '--explain', DOGFOOD_SQL]),
  );
  assert.equal(code, 0);
  assert.match(out, /SCAN/);
  assert.match(out, /GROUP BY\s+agent/);
  assert.match(out, /AGGREGATE/);
  assert.match(out, /SORT\s+out DESC/);
  assert.match(out, /LIMIT\s+count=10/);
});
```

- `--explain` test: smoke-tests that the explain plan describes the dogfood query correctly without ever touching the file.

```js
// test/perf.test.js — PERF-01 + PERF-02
//
// Gated by WHOODB_RUN_PERF env var. Skipped by default because it generates
// large fixtures (~100MB) and runs for 5-20 seconds.
//
// Run with:
//   WHOODB_RUN_PERF=1 node --test test/perf.test.js
//   $env:WHOODB_RUN_PERF=1; node --test test/perf.test.js   (PowerShell)
//   set WHOODB_RUN_PERF=1 && node --test test/perf.test.js   (cmd)

import { test } from 'node:test';
import assert from 'node:assert/strict';
import { writeFileSync, statSync, unlinkSync, existsSync } from 'node:fs';
import { tmpdir } from 'node:os';
import { join } from 'node:path';
import { performance } from 'node:perf_hooks';
import { main } from '../src/cli.js';

const RUN = !!process.env.WHOODB_RUN_PERF;

// Helper: write rows to a temp JSONL file. Returns absolute path.
function writeJsonlFixture(name, rowCount, rowFactory) {
  const path = join(tmpdir(), name);
  // Stream-write in chunks to avoid string-concat blowup on huge files.
  const lines = [];
  let bytes = 0;
  for (let i = 0; i < rowCount; i++) {
    lines.push(JSON.stringify(rowFactory(i)));
    if (lines.length >= 10000) {
      const chunk = lines.join('\n') + '\n';
      writeFileSync(path, chunk, { flag: i === lines.length - 1 ? 'w' : (bytes === 0 ? 'w' : 'a') });
      bytes += chunk.length;
      lines.length = 0;
    }
  }
  if (lines.length > 0) {
    const chunk = lines.join('\n') + '\n';
    writeFileSync(path, chunk, { flag: bytes === 0 ? 'w' : 'a' });
    bytes += chunk.length;
  }
  return path;
}

function captureStdoutSink() {
  // Redirect to a sink — we don't want to print millions of rows to the terminal.
  const original = process.stdout.write.bind(process.stdout);
  let count = 0;
  process.stdout.write = () => { count++; return true; };
  return {
    restore: () => { process.stdout.write = original; return count; },
  };
}

test('PERF-01: WHERE-only query streams a ~100MB JSONL file in under 15 seconds', { skip: !RUN }, async () => {
  // Generate ~100MB fixture: each row ≈ 200 bytes → ~500K rows for 100MB.
  // We aim for ≥100MB to satisfy "in the ballpark of 100MB" budget interpretation.
  const fixturePath = join(tmpdir(), 'whoodb-perf-100mb.jsonl');
  const ROW_COUNT = 500000;

  const rowFactory = (i) => ({
    ts: '2026-05-02T12:00:00.000Z',
    agent: ['planner', 'researcher', 'executor', 'checker'][i % 4],
    level: i % 100 === 0 ? 'error' : 'info',
    msg: 'lorem ipsum dolor sit amet consectetur adipiscing elit, sed do eiusmod tempor incididunt ut labore',
    n: i,
  });

  // Generate fixture if it doesn't exist (cache across runs to avoid 5s of generation each time).
  if (!existsSync(fixturePath) || statSync(fixturePath).size < 100 * 1024 * 1024) {
    writeJsonlFixture('whoodb-perf-100mb.jsonl', ROW_COUNT, rowFactory);
  }
  const fileSize = statSync(fixturePath).size;
  assert.ok(fileSize >= 100 * 1024 * 1024, `fixture must be ≥100MB (got ${(fileSize / 1024 / 1024).toFixed(1)}MB)`);

  // Run WHERE-only streaming query.
  const sink = captureStdoutSink();
  const t0 = performance.now();
  const code = await main([
    'query',
    `SELECT agent, n FROM '${fixturePath}' WHERE level = 'error'`,
  ]);
  const elapsed = performance.now() - t0;
  const written = sink.restore();

  assert.equal(code, 0);
  assert.ok(written > 0, 'expected at least one matching row');

  console.log(`PERF-01: ${(elapsed / 1000).toFixed(2)}s for ~${(fileSize / 1024 / 1024).toFixed(1)}MB / ${written} matching rows`);
  assert.ok(elapsed < 15000, `PERF-01 budget: ${(elapsed / 1000).toFixed(2)}s < 15s`);

  // Don't unlink — let the cache survive across runs.
});

test('PERF-02: GROUP BY query over 100K rows completes in under 5 seconds', { skip: !RUN }, async () => {
  const fixturePath = join(tmpdir(), 'whoodb-perf-100k.jsonl');
  const ROW_COUNT = 100000;

  const rowFactory = (i) => ({
    agent: ['planner', 'researcher', 'executor', 'checker', 'roadmapper'][i % 5],
    output_tokens: 1000 + (i % 50000),
    usage: { output_tokens: 1000 + (i % 50000) },
  });

  if (!existsSync(fixturePath)) {
    writeJsonlFixture('whoodb-perf-100k.jsonl', ROW_COUNT, rowFactory);
  }

  const sink = captureStdoutSink();
  const t0 = performance.now();
  const code = await main([
    'query',
    `SELECT agent, COUNT(*) AS calls, SUM(usage.output_tokens) AS out FROM '${fixturePath}' GROUP BY agent ORDER BY out DESC LIMIT 10`,
  ]);
  const elapsed = performance.now() - t0;
  const written = sink.restore();

  assert.equal(code, 0);
  assert.ok(written >= 1 && written <= 10, `expected 1-10 result rows, got ${written}`);

  console.log(`PERF-02: ${(elapsed / 1000).toFixed(2)}s for 100K rows / GROUP BY`);
  assert.ok(elapsed < 5000, `PERF-02 budget: ${(elapsed / 1000).toFixed(2)}s < 5s`);
});

test('perf harness: confirms gated-by-env-var behavior', () => {
  if (RUN) {
    assert.ok(true, 'WHOODB_RUN_PERF is set — perf tests above ran');
  } else {
    assert.ok(true, 'perf tests skipped (set WHOODB_RUN_PERF=1 to enable)');
  }
});
```
