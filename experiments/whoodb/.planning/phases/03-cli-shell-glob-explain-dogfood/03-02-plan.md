Create `src/explain.js` exporting `printExplain(ast, plan)`. It writes a two-section text dump to `process.stdout` and returns nothing. Pure stdout — no filesystem I/O.

Output format:

```
=== Execution Plan ===
Source: <plan.scan.source>
Stages:
  1. SCAN        <source>
  2. FILTER      WHERE <stringified filter>           # only if plan.filter !== null
  3. GROUP BY    <comma-separated column names>       # only if plan.groupBy !== null
  4. AGGREGATE   <list of aggregate output names>     # only if any projection item is Aggregate
  5. PROJECT     <comma-separated projection names>
  6. SORT        <key direction>, ...                  # only if plan.sort !== null
  7. LIMIT       count=N offset=M                      # only if plan.limit !== null

=== AST ===
SelectStmt
  projection:
    <indented tree of each item>
  from: "<raw from string>"
  where: <indented tree | "(none)">
  groupBy: <indented tree | "(none)">
  orderBy: <indented tree | "(none)">
  limit: <indented or "(none)">
```

Implementation skeleton:

```js
// src/explain.js
const INDENT = '  ';

function pad2(n) { return String(n).padStart(2, ' '); }

function exprToString(node) {
  if (node === null || node === undefined) return '(none)';
  switch (node.type) {
    case 'Literal':
      if (typeof node.value === 'string') return `'${node.value}'`;
      return String(node.value);
    case 'ColumnRef': return node.name;
    case 'BinaryOp':  return `(${exprToString(node.left)} ${node.op} ${exprToString(node.right)})`;
    case 'UnaryOp':   return `(${node.op} ${exprToString(node.operand)})`;
    case 'InList':    return `${exprToString(node.expr)} IN (${node.values.map(exprToString).join(', ')})`;
    case 'BetweenRange': return `${exprToString(node.expr)} BETWEEN ${exprToString(node.lower)} AND ${exprToString(node.upper)}`;
    case 'LikePattern':  return `${exprToString(node.expr)} LIKE '${node.pattern}'`;
    default: return `<${node.type}>`;
  }
}

function projectionItemName(item) {
  if (item.type === 'Wildcard')  return '*';
  if (item.type === 'ColumnRef') return item.name;
  if (item.type === 'Alias')     return item.alias;
  if (item.type === 'Aggregate') {
    const arg = item.arg && item.arg.type === 'Star' ? '*' : (item.arg && item.arg.name) || '';
    return `${item.fn}(${arg})`;
  }
  return `<${item.type}>`;
}

function aggregateNames(projection) {
  const names = [];
  for (const item of projection) {
    const inner = item.type === 'Alias' ? item.expr : item;
    if (inner && inner.type === 'Aggregate') names.push(projectionItemName(item));
  }
  return names;
}

function writePlanSection(plan) {
  process.stdout.write('=== Execution Plan ===\n');
  process.stdout.write(`Source: ${plan.scan.source}\n`);
  process.stdout.write('Stages:\n');
  let n = 1;
  process.stdout.write(`${INDENT}${pad2(n++)}. SCAN        ${plan.scan.source}\n`);
  if (plan.filter !== null) {
    process.stdout.write(`${INDENT}${pad2(n++)}. FILTER      ${exprToString(plan.filter)}\n`);
  }
  if (plan.groupBy !== null && plan.groupBy.length > 0) {
    const cols = plan.groupBy.map((c) => c.name).join(', ');
    process.stdout.write(`${INDENT}${pad2(n++)}. GROUP BY    ${cols}\n`);
  } else if (plan.implicitGroupBy) {
    process.stdout.write(`${INDENT}${pad2(n++)}. GROUP BY    () (implicit single-bucket)\n`);
  }
  const aggNames = aggregateNames(plan.projection);
  if (aggNames.length > 0) {
    process.stdout.write(`${INDENT}${pad2(n++)}. AGGREGATE   ${aggNames.join(', ')}\n`);
  }
  const projNames = plan.projection.map(projectionItemName).join(', ');
  process.stdout.write(`${INDENT}${pad2(n++)}. PROJECT     ${projNames}\n`);
  if (plan.sort !== null) {
    const keys = plan.sort.map((s) => `${s.key} ${s.direction}`).join(', ');
    process.stdout.write(`${INDENT}${pad2(n++)}. SORT        ${keys}\n`);
  }
  if (plan.limit !== null) {
    process.stdout.write(`${INDENT}${pad2(n++)}. LIMIT       count=${plan.limit.count} offset=${plan.limit.offset}\n`);
  }
}

function writeAstSection(ast) {
  process.stdout.write('\n=== AST ===\n');
  process.stdout.write('SelectStmt\n');
  process.stdout.write('  projection:\n');
  for (const item of ast.projection) {
    process.stdout.write(`    ${itemTreeLine(item)}\n`);
  }
  process.stdout.write(`  from: "${ast.from}"\n`);
  process.stdout.write(`  where: ${exprToString(ast.where)}\n`);
  if (ast.groupBy === null) {
    process.stdout.write('  groupBy: (none)\n');
  } else {
    process.stdout.write(`  groupBy: ${ast.groupBy.map((c) => c.name).join(', ')}\n`);
  }
  if (ast.orderBy === null) {
    process.stdout.write('  orderBy: (none)\n');
  } else {
    process.stdout.write(`  orderBy: ${ast.orderBy.map((s) => `${s.key} ${s.direction}`).join(', ')}\n`);
  }
  if (ast.limit === null) {
    process.stdout.write('  limit: (none)\n');
  } else {
    process.stdout.write(`  limit: count=${ast.limit.count} offset=${ast.limit.offset}\n`);
  }
}

function itemTreeLine(item) {
  if (item.type === 'Wildcard')  return 'Wildcard *';
  if (item.type === 'ColumnRef') return `ColumnRef ${item.name}`;
  if (item.type === 'Alias') {
    const inner = item.expr;
    return `Alias "${item.alias}" → ${
      inner.type === 'Aggregate'
        ? `Aggregate ${inner.fn}(${inner.arg && inner.arg.type === 'Star' ? '*' : inner.arg.name})`
        : itemTreeLine(inner)
    }`;
  }
  if (item.type === 'Aggregate') {
    const arg = item.arg && item.arg.type === 'Star' ? '*' : (item.arg && item.arg.name) || '';
    return `Aggregate ${item.fn}(${arg})`;
  }
  return `${item.type}`;
}

export function printExplain(ast, plan) {
  writePlanSection(plan);
  writeAstSection(ast);
}
```

Key decisions:
- Stages are numbered in execution order (matches the architecture data flow).
- Stages are conditional — SORT line only appears if `plan.sort !== null`, etc. This keeps explain output relevant.
- Implicit GROUP BY () gets its own annotated line so the user can see why aggregates ran without an explicit GROUP BY.
- Output uses `process.stdout.write` (NOT console.log) for consistency with the formatter and to avoid double-newline buffering issues.
- No filesystem I/O — `printExplain` is sync and pure stdout.

```js
// src/formatter.js
import { styleText } from 'node:util';

/**
 * CLI-02: write one row per line as JSON.
 * Streams — call once per result row.
 */
export function jsonlRow(row) {
  process.stdout.write(JSON.stringify(row) + '\n');
}

/**
 * CLI-03: write all rows as a fixed-width table with bold headers
 * and a separator row. Buffers — column widths are computed from header
 * length + max cell width across all rows.
 *
 * Empty result set prints "(no rows)" instead of an empty table.
 */
export function formatTable(rows) {
  if (!Array.isArray(rows) || rows.length === 0) {
    process.stdout.write('(no rows)\n');
    return;
  }

  // Use the FIRST row's keys as the column set.
  // (Projector output is uniform — see research Pitfall 7.)
  const columns = Object.keys(rows[0]);
  if (columns.length === 0) {
    process.stdout.write('(no columns)\n');
    return;
  }

  // Compute width per column = max(header.length, max(cell.length)).
  const widths = columns.map((col) =>
    Math.max(
      col.length,
      ...rows.map((r) => String(r[col] ?? '').length),
    ),
  );

  // Header row — bold via util.styleText (auto-disabled when stdout not TTY).
  const header = columns
    .map((col, i) => styleText('bold', col.padEnd(widths[i])))
    .join(' | ');
  // Separator row — dashes per column joined by '-+-'.
  const sep = widths.map((w) => '-'.repeat(w)).join('-+-');

  process.stdout.write(header + '\n');
  process.stdout.write(sep + '\n');

  for (const row of rows) {
    const line = columns
      .map((col, i) => String(row[col] ?? '').padEnd(widths[i]))
      .join(' | ');
    process.stdout.write(line + '\n');
  }
}
```

Key decisions:
- `jsonlRow` is single-row (streaming). `formatTable` is whole-array (buffered) — required to compute widths.
- `String(r[col] ?? '')` handles null/undefined gracefully (empty string in cell, no crash).
- Separator uses `-+-` between columns (mirrors `' | '` between data cells).
- `styleText('bold', ...)` auto-disables ANSI when stdout is piped or NO_COLOR is set — no extra check needed.

import { test } from 'node:test';
import assert from 'node:assert/strict';
import { tokenize } from '../src/lexer.js';
import { parse } from '../src/parser.js';
import { analyze } from '../src/analyze.js';
import { printExplain } from '../src/explain.js';

function captureStdout(fn) {
  const original = process.stdout.write.bind(process.stdout);
  let buf = '';
  process.stdout.write = (chunk) => { buf += chunk; return true; };
  try { fn(); } finally { process.stdout.write = original; }
  return buf;
}

test('explain: simple SELECT * FROM file', () => {
  const ast = parse(tokenize("SELECT * FROM 'fixtures/aggregates.jsonl'"));
  const plan = analyze(ast);
  const out = captureStdout(() => printExplain(ast, plan));
  assert.match(out, /=== Execution Plan ===/);
  assert.match(out, /=== AST ===/);
  assert.match(out, /SCAN/);
  assert.match(out, /PROJECT/);
  assert.doesNotMatch(out, /SORT/);  // no ORDER BY
  assert.doesNotMatch(out, /LIMIT/); // no LIMIT
});

test('explain: WHERE + GROUP BY + ORDER BY + LIMIT (TEST-04 non-trivial query)', () => {
  const sql = "SELECT agent, COUNT(*) AS calls FROM 'f' WHERE level = 'error' GROUP BY agent ORDER BY calls DESC LIMIT 10";
  const ast = parse(tokenize(sql));
  const plan = analyze(ast);

import { test } from 'node:test';
import assert from 'node:assert/strict';
import { jsonlRow, formatTable } from '../src/formatter.js';

function captureStdout(fn) {
  const original = process.stdout.write.bind(process.stdout);
  let buf = '';
  process.stdout.write = (chunk) => { buf += chunk; return true; };
  try { fn(); } finally { process.stdout.write = original; }
  return buf;
}

test('jsonlRow writes JSON.stringify(row) + newline', () => {
  const out = captureStdout(() => jsonlRow({ a: 1, b: 'x' }));
  assert.equal(out, '{"a":1,"b":"x"}\n');
});

test('formatTable empty rows → "(no rows)"', () => {
  const out = captureStdout(() => formatTable([]));
  assert.equal(out, '(no rows)\n');
});

test('formatTable computes column widths from header + max cell width', () => {
  const rows = [{ name: 'a', count: 1 }, { name: 'longer', count: 1000 }];
  const out = captureStdout(() => formatTable(rows));
  const lines = out.split('\n').filter(Boolean);
  // Line 1 = header, line 2 = separator, lines 3+ = data.
  assert.equal(lines.length, 4);  // header + sep + 2 data rows
  // The separator uses '-+-' between columns and '-' fill.
  assert.match(lines[1], /-+\-\+\--+/);
  // Each data row should contain ' | ' as the column separator.
  assert.ok(lines[2].includes(' | '));
  assert.ok(lines[3].includes(' | '));
});

test('formatTable handles null/undefined cells without crashing', () => {
  const rows = [{ a: null, b: undefined }, { a: 1, b: 'x' }];
  const out = captureStdout(() => formatTable(rows));
  // Should not throw; should produce 4 lines (header, sep, 2 data rows).
  const lines = out.split('\n').filter(Boolean);
  assert.equal(lines.length, 4);
});
