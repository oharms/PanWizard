### validateType (TYPE-01) — handles Pitfall 1

```js
export function validateType(schema, data, path, errors) {
  if (schema.type === undefined) return;
  const types = Array.isArray(schema.type) ? schema.type : [schema.type];
  if (types.some(t => matchesType(t, data))) return;
  const expected = Array.isArray(schema.type) ? schema.type : schema.type;
  errors.push(makeError(path, 'type',
    `expected ${types.join(' or ')}, got ${describeType(data)}`,
    data, expected));
}

function matchesType(t, v) {
  switch (t) {
    case 'string':  return typeof v === 'string';
    case 'number':  return typeof v === 'number' && !Number.isNaN(v) && Number.isFinite(v);
    case 'integer': return typeof v === 'number' && !Number.isNaN(v) && Number.isFinite(v) && Number.isInteger(v);
    case 'boolean': return typeof v === 'boolean';
    case 'null':    return v === null;
    case 'array':   return Array.isArray(v);
    case 'object':  return typeof v === 'object' && v !== null && !Array.isArray(v);
    default:        return false; // unknown type — already warned at load
  }
}

function describeType(v) {
  if (v === null) return 'null';
  if (Array.isArray(v)) return 'array';
  if (typeof v === 'number' && Number.isNaN(v)) return 'NaN';
  if (typeof v === 'number' && !Number.isFinite(v)) return 'Infinity';
  return typeof v;
}
```

### validateString (TYPE-02) — handles Pitfall 8

```js
export function validateString(schema, data, path, errors) {
  if (typeof data !== 'string') return; // type mismatch already reported by validateType if applicable
  if (typeof schema.minLength === 'number' && [...data].length < schema.minLength) {
    errors.push(makeError(path, 'minLength',
      `string length ${[...data].length} below minimum ${schema.minLength}`,
      data, schema.minLength));
  }
  if (typeof schema.maxLength === 'number' && [...data].length > schema.maxLength) {
    errors.push(makeError(path, 'maxLength',
      `string length ${[...data].length} above maximum ${schema.maxLength}`,
      data, schema.maxLength));
  }
  if (typeof schema.pattern === 'string') {
    // Pattern was already validated at load — safe to compile here, no flags
    const re = new RegExp(schema.pattern);
    if (!re.test(data)) {
      errors.push(makeError(path, 'pattern',
        `string does not match pattern "${schema.pattern}"`,
        data, schema.pattern));
    }
  }
  if (Array.isArray(schema.enum) && !schema.enum.some(v => v === data)) {
    errors.push(makeError(path, 'enum',
      `value not in enum [${schema.enum.map(JSON.stringify).join(', ')}]`,
      data, schema.enum));
  }
}
```

Note on `enum`: also applies to non-string values. The simplest correct path is to **also** check enum for non-string data — but per architecture.md, validateString runs only when data is a string. The cleaner design is: the enum check belongs in a separate `validateEnum` or in a top-level branch in `validateNode` that handles `enum` regardless of type. **Implementation choice:** put a generic `validateEnum` in this file (applies to any data type via `JSON.stringify`-equality OR strict `===`):

```js
export function validateEnum(schema, data, path, errors) {
  if (!Array.isArray(schema.enum)) return;
  // Use canonical-key-sorted stringify for deep equality (same as uniqueItems below)
  const dataKey = canonicalize(data);
  if (!schema.enum.some(v => canonicalize(v) === dataKey)) {
    errors.push(makeError(path, 'enum',
      `value not in enum`,
      data, schema.enum));
  }
}
```

And **remove** the enum check from validateString. validateNode will dispatch into validateEnum when `schema.enum` is present, regardless of data type. Add `validateEnum` to the export list. (`canonicalize` is defined in validateArray's section below.)

### validateNumber (TYPE-03)

```js
export function validateNumber(schema, data, path, errors) {
  if (typeof data !== 'number' || Number.isNaN(data) || !Number.isFinite(data)) return;
  if (typeof schema.minimum === 'number' && data < schema.minimum) {
    errors.push(makeError(path, 'minimum', `value ${data} below minimum ${schema.minimum}`, data, schema.minimum));
  }
  if (typeof schema.maximum === 'number' && data > schema.maximum) {
    errors.push(makeError(path, 'maximum', `value ${data} above maximum ${schema.maximum}`, data, schema.maximum));
  }
  if (typeof schema.exclusiveMinimum === 'number' && data <= schema.exclusiveMinimum) {
    errors.push(makeError(path, 'exclusiveMinimum', `value ${data} not strictly above ${schema.exclusiveMinimum}`, data, schema.exclusiveMinimum));
  }
  if (typeof schema.exclusiveMaximum === 'number' && data >= schema.exclusiveMaximum) {
    errors.push(makeError(path, 'exclusiveMaximum', `value ${data} not strictly below ${schema.exclusiveMaximum}`, data, schema.exclusiveMaximum));
  }
  if (typeof schema.multipleOf === 'number' && schema.multipleOf > 0) {
    // Float-epsilon awareness for fractional divisors
    const ratio = data / schema.multipleOf;
    const rounded = Math.round(ratio);
    if (Math.abs(ratio - rounded) > 1e-9) {
      errors.push(makeError(path, 'multipleOf', `value ${data} is not a multiple of ${schema.multipleOf}`, data, schema.multipleOf));
    }
  }
}
```

### validateArray (TYPE-04) — handles Pitfall 7

```js
export function validateArray(schema, data, path, errors) {
  if (!Array.isArray(data)) return;
  if (typeof schema.minItems === 'number' && data.length < schema.minItems) {
    errors.push(makeError(path, 'minItems', `array length ${data.length} below minimum ${schema.minItems}`, data, schema.minItems));
  }
  if (typeof schema.maxItems === 'number' && data.length > schema.maxItems) {
    errors.push(makeError(path, 'maxItems', `array length ${data.length} above maximum ${schema.maxItems}`, data, schema.maxItems));
  }
  if (schema.uniqueItems === true) {
    const seen = new Map();
    for (let i = 0; i < data.length; i++) {
      const key = canonicalize(data[i]);
      if (seen.has(key)) {
        errors.push(makeError(path, 'uniqueItems', `duplicate item at index ${i} (also at ${seen.get(key)})`, data[i], true));
      } else {
        seen.set(key, i);
      }
    }
  }
  // items
  if (schema.items !== undefined) {
    if (Array.isArray(schema.items)) {
      // Tuple form: validate each item against its corresponding sub-schema
      for (let i = 0; i < data.length && i < schema.items.length; i++) {
        const sub = validateNode(schema.items[i], data[i], appendIndex(path, i));
        for (const e of sub) errors.push(e);
      }
      // Items beyond the tuple length are unconstrained in Phase 1 (additionalItems is a Phase 2 concern, not even listed in idea.md)
    } else {
      // Single schema: validate every item against it
      for (let i = 0; i < data.length; i++) {
        const sub = validateNode(schema.items, data[i], appendIndex(path, i));
        for (const e of sub) errors.push(e);
      }
    }
  }
}

// Canonical-key-sorted JSON stringify for deep equality (Pitfall 7).
// Recursively sorts object keys so {a:1,b:2} and {b:2,a:1} produce identical strings.
export function canonicalize(value) {
  if (value === null || typeof value !== 'object') return JSON.stringify(value);
  if (Array.isArray(value)) return '[' + value.map(canonicalize).join(',') + ']';
  const keys = Object.keys(value).sort();
  return '{' + keys.map(k => JSON.stringify(k) + ':' + canonicalize(value[k])).join(',') + '}';
}
```

### validateObject (TYPE-05) — handles Pitfall 4

```js
export function validateObject(schema, data, path, errors) {
  if (typeof data !== 'object' || data === null || Array.isArray(data)) return;

  // required
  if (Array.isArray(schema.required)) {
    for (const k of schema.required) {
      if (!(k in data)) {
        errors.push(makeError(path, 'required', `missing required property "${k}"`, undefined, k));
      }
    }
  }

  // properties — recurse for keys that exist in data
  if (schema.properties) {
    for (const [k, subSchema] of Object.entries(schema.properties)) {
      if (k in data) {
        const sub = validateNode(subSchema, data[k], appendKey(path, k));
        for (const e of sub) errors.push(e);
      }
    }
  }

  // patternProperties — apply each pattern to every matching key
  const compiledPatterns = [];
  if (schema.patternProperties) {
    for (const [pattern, subSchema] of Object.entries(schema.patternProperties)) {
      const re = new RegExp(pattern); // safe — already validated at load
      compiledPatterns.push({ re, subSchema });
      for (const k of Object.keys(data)) {
        if (re.test(k)) {
          const sub = validateNode(subSchema, data[k], appendKey(path, k));
          for (const e of sub) errors.push(e);
        }
      }
    }
  }

  // additionalProperties — only true/false form in Phase 1 (sub-schema form deferred to Phase 2)
  if (schema.additionalProperties === false) {
    const propKeys = new Set(Object.keys(schema.properties || {}));
    for (const k of Object.keys(data)) {
      if (propKeys.has(k)) continue;
      if (compiledPatterns.some(({ re }) => re.test(k))) continue;
      errors.push(makeError(appendKey(path, k), 'additionalProperties',
        `additional property "${k}" not allowed`,
        data[k], false));
    }
  }
  // schema.additionalProperties === true OR a sub-schema OR undefined → permissive (no error)
}
```

**Notes:**
- For `additionalProperties` errors, the path points to the offending property (`appendKey(path, k)`) per Pitfall 11 reasoning — the caller wants to know WHICH property, not just that the parent had extras.
- For `required` errors, `value` is `undefined` (the missing-key case); `makeError` will OMIT the key entirely from the error object per Plan 01 contract. The `expected` field carries the missing key name.
- `canonicalize` is exported (used by validateEnum and by validateArray.uniqueItems internally).

Use `[...data].length` for string lengths (Unicode code-point counting) rather than `data.length` (UTF-16 code units). This matches the JSON Schema spec which says length is character count.

```js
// src/validator-core.js — validateNode dispatch
import {
  validateType,
  validateString,
  validateNumber,
  validateArray,
  validateObject,
  validateEnum
} from './keyword-handlers.js';

export function validateNode(schema, data, path) {
  const errors = [];
  if (schema === null || typeof schema !== 'object') return errors;

  // Order matters for readability of error output, but every handler is independent.
  // Type check first so other handlers can rely on it (or skip if data type doesn't match — each handler self-guards).
  validateType(schema, data, path, errors);
  validateEnum(schema, data, path, errors);

  // Type-specific keyword handlers — each one no-ops if data isn't its type.
  validateString(schema, data, path, errors);
  validateNumber(schema, data, path, errors);
  validateArray(schema, data, path, errors);
  validateObject(schema, data, path, errors);

  return errors;
}
```

**Why each handler self-guards on data type:** keeps validateNode a flat dispatcher rather than a type-switching tree. If `data` is a string, only `validateType` and `validateString` (and `validateEnum`) actually do work; the others all return early. This is correct and readable.

**Phase 2 extension hook:** The composition handlers (`oneOf`/`anyOf`/`allOf`/`not`) and `$ref` will be added here in Phase 2 — they recurse via `validateNode` themselves. Phase 1 simply omits them; an unknown-keyword warn was already emitted at load (or rather, `oneOf` etc. ARE in KNOWN_KEYWORDS so they don't warn — they just no-op in Phase 1).

Do NOT export `makeError`/`appendKey`/`appendIndex` from here anymore — Plan 01's shell did, but those re-exports were courtesy. Plan 02 keyword-handlers.js imports directly from `./error-utils.js`. Remove the re-export lines if present.

```js
// src/validate.js — public API for whooschema
// validate(schema, data) → { valid: boolean, errors: ValidationError[] }
// Errors are sorted with a numeric-aware path comparator and contain
// {path, rule, message, value, expected} (expected omitted when not provided).
import { loadSchema } from './schema-loader.js';
import { validateNode } from './validator-core.js';
import { sortErrors } from './error-utils.js';

export function validate(schema, data) {
  // Pre-walk the schema: throws on invalid pattern (LOAD-01); warns on unknown keywords (LOAD-02).
  // The throw happens before any data is checked — required by SC-4 (idea.md).
  loadSchema(schema);
  const raw = validateNode(schema, data, '$');
  const errors = sortErrors(raw);
  return { valid: errors.length === 0, errors };
}
```

**Order matters:** `loadSchema` MUST be called BEFORE `validateNode`. If a schema has `pattern: "["` and `loadSchema` is called second, validateNode will hit the bad pattern when it tries to compile during validateString — that's a runtime error, not a load-time error. Per LOAD-01, the error must surface BEFORE any data is checked. Tests in Plan 03 will verify this ordering with a schema that has both a bad pattern and would otherwise fail validation — the bad-pattern error must come first.

**Caching consideration:** `loadSchema` runs every call. For Phase 1 this is acceptable (research notes that caching is a YAGNI optimization). Future: a WeakMap on the schema object could memoize.

**Do NOT export anything else from this file.** `loadSchema`, `validateNode`, and the helpers are internals. Only `validate` is the public surface (per index.d.ts in Plan 01).
