  assert.equal(validate({ type: 'number' }, NaN).valid, false);
  assert.equal(validate({ type: 'number' }, Infinity).valid, false);
  assert.equal(validate({ type: 'number' }, -Infinity).valid, false);

test('type:integer rejects 1.5 but accepts 1.0', () => {
  assert.equal(validate({ type: 'integer' }, 1.5).valid, false);
  assert.equal(validate({ type: 'integer' }, 1.0).valid, true);
  assert.equal(validate({ type: 'integer' }, 1).valid, true);
});

test('type:object rejects null and arrays', () => {
  assert.equal(validate({ type: 'object' }, null).valid, false);
  assert.equal(validate({ type: 'object' }, []).valid, false);
  assert.equal(validate({ type: 'object' }, {}).valid, true);
});

test('type:array rejects {} and accepts []', () => {
  assert.equal(validate({ type: 'array' }, {}).valid, false);
  assert.equal(validate({ type: 'array' }, []).valid, true);

  assert.equal(validate({ type: 'null' }, null).valid, true);
  assert.equal(validate({ type: 'null' }, undefined).valid, false);
  assert.equal(validate({ type: 'null' }, 0).valid, false);
});

test('type as array of types: at least one must match', () => {
  assert.equal(validate({ type: ['string', 'number'] }, 'hi').valid, true);
  assert.equal(validate({ type: ['string', 'number'] }, 5).valid, true);

  assert.equal(validate({ maxLength: 3 }, 'abc').valid, true);

test('pattern matches anywhere in string per spec (Pitfall 8)', () => {
  // Substring match — pattern is found inside the string
  assert.equal(validate({ pattern: '[0-9]+' }, 'abc123def').valid, true);
  // Anchored pattern requires full match
  assert.equal(validate({ pattern: '^[0-9]+$' }, 'abc123def').valid, false);
});

test('pattern does not enable multiline flag (Pitfall 8)', () => {
  // Without /m, ^/$ anchor at start/end of string, not lines
  assert.equal(validate({ pattern: '^[0-9]+$' }, '123\\nmore').valid, false);
});

test('enum requires deep equality match', () => {
  assert.equal(validate({ enum: ['a', 'b', 'c'] }, 'b').valid, true);
  assert.equal(validate({ enum: ['a', 'b', 'c'] }, 'd').valid, false);
  assert.equal(validate({ enum: [{a:1,b:2}] }, {b:2,a:1}).valid, true); // canonical equality

import { test } from 'node:test';
import assert from 'node:assert/strict';
import { validate } from '../src/validate.js';

// SC-2: A document with five distinct primitive violations returns all five errors in one call.
// Schema designed so each constraint targets a different rule on a different path.
test('SC-2: validate aggregates 5 distinct primitive violations in one call', () => {
  const schema = {
    type: 'object',
    required: ['name'],
    properties: {
      title:   { type: 'string', minLength: 3 },          // -> minLength
      age:     { type: 'number', minimum: 18 },           // -> minimum
      tags:    { type: 'array',  minItems: 2 },           // -> minItems
      kind:    { type: 'string' }                         // -> type (data is a number)
    }
  };
  const data = {
    title: 'X',          // minLength fails (length 1 < 3)
    age: 5,              // minimum fails (5 < 18)
    tags: ['only-one'],  // minItems fails (1 < 2)
    kind: 42             // type fails (not a string)
    // 'name' is missing → required fails
  };
  const r = validate(schema, data);
  assert.equal(r.valid, false);
  const rules = r.errors.map(e => e.rule).sort();
  // Expect exactly: required, minLength, minimum, minItems, type
  assert.deepEqual(rules, ['minItems', 'minLength', 'minimum', 'required', 'type']);
  assert.equal(r.errors.length, 5, 'all 5 violations must surface; got: ' + JSON.stringify(rules));
});

// SC-3a: Numeric-aware sort: $.users[2].x must come before $.users[10].x
test('SC-3: errors sort with numeric-aware path comparator (Pitfall 11)', () => {
  // Build a schema/data pair that produces errors at $.users[2].email and $.users[10].name.
  // 11 entries; entry 2 has a non-string email, entry 10 has a non-string name.
  const schema = {
    properties: {
      users: {
        items: {
          properties: {
            email: { type: 'string' },
            name:  { type: 'string' }
          }
        }
      }
    }
  };
  const users = [];
  for (let i = 0; i < 11; i++) {
    users.push({
      email: i === 2 ? 99 : 'ok@x.io',
      name:  i === 10 ? 99 : 'ok'
    });
  }
  const r = validate(schema, { users });
  assert.equal(r.valid, false);
  assert.equal(r.errors.length, 2);
  // Numeric-aware: [2] before [10]
  assert.equal(r.errors[0].path, '$.users[2].email');
  assert.equal(r.errors[1].path, '$.users[10].name');
});

// SC-3b: Byte-identical determinism — running twice produces identical JSON.stringify output
test('SC-3: identical inputs produce byte-identical errors arrays', () => {
  const schema = {
    type: 'object',
    properties: {
      items: { type: 'array', items: { type: 'integer' } },
      score: { type: 'number', minimum: 0 }
    }
  };
  const data = { items: [1, 'two', 3, 'four'], score: -5 };
  const a = JSON.stringify(validate(schema, data).errors);
  const b = JSON.stringify(validate(schema, data).errors);
  assert.equal(a, b, 'byte-identical determinism violated');
});

// SC-3c: Plain lex sort would put [10] before [2] — verify our comparator beats it.
test('SC-3: sort is NOT plain lexicographic — [10] never precedes [2]', () => {
  const schema = { items: { type: 'string' } };
  const data = ['ok', 'ok', 99, 'ok', 'ok', 'ok', 'ok', 'ok', 'ok', 'ok', 99];
  // Errors at indices 2 and 10
  const r = validate(schema, data);
  assert.equal(r.errors.length, 2);
  assert.equal(r.errors[0].path, '$[2]');
  assert.equal(r.errors[1].path, '$[10]');
});

import { test } from 'node:test';
import assert from 'node:assert/strict';
import { validate } from '../src/validate.js';

// SC-4: An invalid pattern regex throws a clear load-time error before any data is checked.
test('SC-4: invalid pattern regex throws at load time', () => {
  let threw = false;
  let message = '';
  try {
    validate({ pattern: '[' }, 'any data');
  } catch (e) {
    threw = true;
    message = e.message;
  }
  assert.equal(threw, true, 'validate must throw on invalid pattern');
  assert.ok(message.includes('Invalid pattern'), 'error message must say "Invalid pattern": ' + message);
  assert.ok(message.includes('['), 'error message must reference the bad regex: ' + message);
});

test('SC-4: invalid pattern in nested schema also throws at load time', () => {
  const schema = {
    type: 'object',
    properties: {
      tag: { type: 'string', pattern: '(unclosed' }
    }
  };
  assert.throws(() => validate(schema, { tag: 'abc' }), /Invalid pattern/);
});

test('SC-4: bad pattern throws BEFORE data validation runs', () => {
  // Schema has both a bad pattern AND a violation — the throw must come from the pattern.
  // If the data was processed first, the test would see {valid:false, errors:[...type error...]}
  // instead of an exception.
  const schema = {
    type: 'string',           // would yield a type error for 42
    pattern: '['              // bad regex — must throw first
  };
  assert.throws(() => validate(schema, 42), /Invalid pattern/);
});

// SC-5: A schema with an unknown keyword loads without throwing and emits a console warning.
test('SC-5: unknown keyword emits exactly one console.warn', () => {
  const warns = [];
  const orig = console.warn;
  console.warn = (...args) => warns.push(args.join(' '));
  try {
    const r = validate({ type: 'string', 'x-custom': true }, 'hello');
    assert.equal(r.valid, true, 'validation must succeed despite unknown keyword');
    const xCustomWarns = warns.filter(w => w.includes('x-custom'));
    assert.equal(xCustomWarns.length, 1, 'expected exactly one x-custom warn, got: ' + warns.join(' | '));
    assert.ok(xCustomWarns[0].includes('whooschema'), 'warn should be branded as whooschema: ' + xCustomWarns[0]);
  } finally {
    console.warn = orig;
  }
});

test('SC-5: unknown keyword is deduped across multiple validate() calls', () => {
  const warns = [];
  const orig = console.warn;
  console.warn = (...args) => warns.push(args.join(' '));
  try {
    // NOTE: use a unique keyword name to avoid collision with the prior test's dedup state
    validate({ 'x-deduptest': 1 }, 'hi');
    validate({ 'x-deduptest': 1 }, 'hi');
    validate({ 'x-deduptest': 1 }, 'hi');
    const matched = warns.filter(w => w.includes('x-deduptest'));
    assert.equal(matched.length, 1, 'unknown keyword must be warned once across N calls, got ' + matched.length);
  } finally {
    console.warn = orig;
  }
});

test('LOAD-02: known Phase 2/3 keywords (oneOf, $ref, format) do NOT emit warnings', () => {
  const warns = [];
  const orig = console.warn;
  console.warn = (...args) => warns.push(args.join(' '));
  try {
    // These keywords are listed in KNOWN_KEYWORDS so they do not warn;
    // they're simply no-ops in Phase 1's validateNode.
    validate({ oneOf: [{ type: 'string' }] }, 'hi');
    validate({ $ref: '#/definitions/foo' }, 'hi');
    validate({ format: 'email' }, 'a@b.c');
    for (const w of warns) {
      for (const kw of ['oneOf', '$ref', 'format']) {
        assert.ok(!w.includes(`unknown keyword "${kw}"`), `${kw} must not warn — listed as known: ${w}`);
