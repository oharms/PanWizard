Purpose: STATE-05 says resume is a pure pre-scheduler merge step. Plan 04 wires CLI --resume to call mergeState; Plan 04 also adds the scheduler's success-task pre-decrement loop. Plan 02 (skip.js) is independent of this plan because skip-downstream operates on already-running scheduler state, not on the merge.

```js
// Phase 3 Plan 01 — pure pre-scheduler resume merge step.
//
// STATE-04 + STATE-05: mergeState(flow, oldState) takes the validated flow and the
// prior state file's contents and returns a NEW state object ready for runFlow().
// The scheduler is oblivious to fresh-vs-resume mode because the output shape is
// identical to buildInitialState's output. By keeping this module pure (no I/O, no
// scheduler imports), the "resume is a pre-scheduler merge step" guarantee is
// structural, not policy.
//
// Pitfall C5 (running misinterpreted): tasks in 'running' status are reset to
// 'pending' with their attempts counter preserved — the partial run counted as one
// attempt. Never silently coerce 'running' to 'success' (silent loss of work) or
// to 'failed' (premature retry exhaustion).
//
// Pitfall (state-flow drift): tasks referenced by oldState but absent from
// flow.tasks throw a clear named error rather than being silently dropped — the
// most likely cause is the user edited flow.json between runs.

import { RuntimeError } from './errors.js';

/**
 * mergeState(flow, oldState) -> newState
 *
 * Pure function. No fs, no Date.now(), no process.*.
 *
 * @param {Object} flow      validatedFlow from src/validator.js validateFlow
 * @param {Object} oldState  parsed prior whooflow.state.json contents
 * @returns {Object}         new state object — same shape as buildInitialState output
 * @throws {RuntimeError}    if oldState references a task id absent from flow.tasks
 */
export function mergeState(flow, oldState) {
  // Drift check: every id in oldState.tasks must exist in flow.tasks.
  // Surface a clear error rather than silently dropping a task.
  for (const id of Object.keys(oldState.tasks ?? {})) {
    if (!(id in flow.tasks)) {
      throw new RuntimeError(
        `state references unknown task: ${id}. Was flow.json edited after the previous run?`,
      );
    }
  }

  const newTasks = {};
  for (const id of Object.keys(flow.tasks)) {
    const oldTask = oldState.tasks?.[id];
    if (!oldTask) {
      // New task added to flow since prior run — initialize fresh pending.
      newTasks[id] = freshPending();
      continue;
    }

    const status = oldTask.status;
    if (status === 'success') {
      // Preserved — scheduler's seed loop pre-decrements children's indegrees.
      newTasks[id] = {
        status: 'success',
        attempts: oldTask.attempts ?? 0,
        started_at: oldTask.started_at ?? null,
        ended_at: oldTask.ended_at ?? null,
        exit_code: oldTask.exit_code ?? 0,
      };
    } else if (status === 'failed') {
      const attemptsAllowed = flow.tasks[id].retry?.attempts ?? 1;
      const attemptsUsed = oldTask.attempts ?? 0;
      const retriesRemaining = attemptsAllowed - attemptsUsed;
      if (retriesRemaining > 0) {
        // Reset to pending — retries still available. Preserve attempts counter.
        newTasks[id] = resetToPending(oldTask);
      } else {
        // Terminal failed — left for skip-downstream BFS on scheduler startup.
        newTasks[id] = {
          status: 'failed',
          attempts: oldTask.attempts ?? 0,
          started_at: oldTask.started_at ?? null,
          ended_at: oldTask.ended_at ?? null,
          exit_code: oldTask.exit_code ?? -1,
          ...(oldTask.signal ? { signal: oldTask.signal } : {}),
        };
      }
    } else if (status === 'running') {
      // Interrupted — partial run counted as one attempt. Reset to pending,
      // preserve attempts. Pitfall C5: never coerce 'running' to anything else.
      newTasks[id] = resetToPending(oldTask);
    } else if (status === 'skipped') {
      // Re-evaluate on this run: if the parent now succeeds, this task should run.
      // mergeState resets to pending; the scheduler's startup BFS may re-skip it
      // if the parent is still in terminal failed state.
      newTasks[id] = resetToPending(oldTask);
    } else if (status === 'pending') {
      // Already pending — leave as-is, but normalize shape (drop transient fields).
      newTasks[id] = resetToPending(oldTask);
    } else {
      // Unknown status — defensive. Treat as pending so the scheduler can run it.
      newTasks[id] = resetToPending(oldTask);
    }
  }

  return {
    flow_file: oldState.flow_file ?? null,
    started_at: oldState.started_at ?? null,
    ended_at: null, // cleared — merged run is restarting
    tasks: newTasks,
  };
}

function freshPending() {
  return {
    status: 'pending',
    attempts: 0,
    started_at: null,
    ended_at: null,
    exit_code: null,
  };
}

function resetToPending(oldTask) {
  return {
    status: 'pending',
    attempts: oldTask.attempts ?? 0, // preserve — partial run counted as attempt
    started_at: null,
    ended_at: null,
    exit_code: null,
    // Deliberately drop oldTask.signal and oldTask.skip_reason — those belong to
    // the prior terminal state, not the reset state.
  };
}
```

**DO NOT** call `Date.now()`, `new Date()`, or `process.*` inside `mergeState`. Tests must be able to assert exact equality on the returned object without time-fudging.

**DO NOT** mutate the `oldState` argument. Always return a fresh object — callers may keep the old reference for diagnostics.

**DO NOT** add a "resume mode" flag the scheduler reads. STATE-05 says scheduler is oblivious; honor that structurally.

import assert from 'node:assert/strict';
import { mergeState } from '../src/resume.js';

// Helper: minimal validatedFlow shape for tests (only the keys mergeState reads).
function flowOf(tasksObj) {
  return {
    tasks: tasksObj,
    children: {},
    topoOrder: Object.keys(tasksObj),
  };
}

// Helper: minimal oldState shape.
function oldStateOf(tasksObj, opts = {}) {
  return {
    flow_file: opts.flow_file ?? '/tmp/flow.json',
    started_at: opts.started_at ?? '2026-05-02T00:00:00.000Z',
    ended_at: opts.ended_at ?? null,
    tasks: tasksObj,
  };
}

test('STATE-04 R-1: success → preserved as success (scheduler will pre-decrement children)', () => {
  const flow = flowOf({
    a: { id: 'a', cmd: 'true', depends_on: [] },
    b: { id: 'b', cmd: 'true', depends_on: ['a'] },
  });
  const old = oldStateOf({
    a: { status: 'success', attempts: 1, started_at: 't1', ended_at: 't2', exit_code: 0 },
    b: { status: 'pending', attempts: 0, started_at: null, ended_at: null, exit_code: null },
  });
  const merged = mergeState(flow, old);
  assert.equal(merged.tasks.a.status, 'success');
  assert.equal(merged.tasks.a.attempts, 1);
  assert.equal(merged.tasks.a.exit_code, 0);
  assert.equal(merged.tasks.b.status, 'pending');
});

test('STATE-04 R-2: failed with retries remaining → reset to pending, preserve attempts', () => {
  const flow = flowOf({
    a: { id: 'a', cmd: 'false', depends_on: [], retry: { attempts: 3, backoff_ms: 10 } },
  });
  const old = oldStateOf({
    a: { status: 'failed', attempts: 1, started_at: 't1', ended_at: 't2', exit_code: 1 },

  const merged = mergeState(flow, old);
  assert.equal(merged.tasks.a.status, 'pending');
  assert.equal(merged.tasks.a.attempts, 1, 'attempts must be preserved');
  assert.equal(merged.tasks.a.started_at, null);
  assert.equal(merged.tasks.a.ended_at, null);

    a: { status: 'failed', attempts: 2, started_at: 't1', ended_at: 't2', exit_code: 1 },
