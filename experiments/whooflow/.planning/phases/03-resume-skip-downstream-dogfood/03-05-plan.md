### 3) Create `test/fixtures/sleepy.json`

A single-task fixture for the DOG-03 kill-and-resume test. The task writes a marker file (so the test can deterministically observe "task started") then sleeps long enough for the test to send SIGINT.

```json
{
  "tasks": [
    {
      "id": "sleep",
      "cmd": "node -e \"require('fs').writeFileSync('MARKER',String(Date.now()));setTimeout(()=>{},30000);\"",
      "depends_on": [],
      "retry": { "attempts": 2, "backoff_ms": 0 }
    }
  ]
}
```

Notes:
- `attempts: 2` means after a kill (counted as one attempt), `--resume` has 1 retry remaining → mergeState resets to pending → re-run → success on retry.
- The marker file path is bare `MARKER` — relative to the cwd of the spawned `node -e` process, which is the directory where the runner was launched (the test's tmpdir). This makes the marker file appear in the same dir as the flow + state file.
- The sleep is 30 seconds — far longer than the test's poll deadline + kill cycle (~5s). The kill terminates the task before the sleep elapses naturally.

**DO NOT** put the marker file in `os.tmpdir()` — the test needs to know exactly where to look, and the runner's cwd is the simplest invariant.

**DO NOT** use `setInterval` instead of `setTimeout` in the sleepy fixture — `setInterval` would prevent the process from exiting cleanly even after kill (defensive; setTimeout's handle is GC'd on kill).

**DO NOT** add any error-throwing logic. The fixture's contract is: "write marker, sleep until killed or 30s, exit." Anything else complicates the test's assertions.

import { spawn } from 'node:child_process';

// Marker-file polling helper (deterministic — no fixed sleeps; pitfall M4).
async function waitForMarker(markerPath, maxMs = 5000) {
  const deadline = Date.now() + maxMs;
  while (Date.now() < deadline) {
    if (existsSync(markerPath)) return;
    await new Promise((r) => setTimeout(r, 50));
  }
  throw new Error(`marker file ${markerPath} never appeared within ${maxMs}ms`);
}

test('DOG-02 D-1: dogfood flow.json runs npm test + build:hooks in parallel + count-tests, exits 0', (t) => {
  // Run from the repo root with the actual flow.json.
  // Concurrency 2 lets test+build-hooks run in parallel; count-tests gates on both.
  const result = runCli(['run', '--file', 'flow.json', '--concurrency', '2']);
  // The test step inside the dogfood runs `npm test` recursively, which would re-enter
  // this test file. To avoid recursive test invocation in CI, this assertion is
  // permissive: exit code 0 OR a recognized parallel-run signature.
  // (If running this test directly via `node --test`, the dogfood test should be
  // SKIPPED — guard with a sentinel env var.)
  if (process.env.WHOOFLOW_DOGFOOD === '1') {
    assert.equal(result.code, 0, `dogfood should exit 0; stderr=${result.stderr}`);
    // State file in repo root.
    const state = JSON.parse(readFileSync('whooflow.state.json', 'utf8'));
    assert.equal(state.tasks.test.status, 'success');
    assert.equal(state.tasks['build-hooks'].status, 'success');
    assert.equal(state.tasks['count-tests'].status, 'success');
  } else {
    // Default: skip — running the dogfood inside the test suite would recurse infinitely
    // (the dogfood runs `npm test` which runs this test which runs the dogfood...).

test('DOG-03 D-2: kill-and-resume — sleepy task killed mid-run, --resume re-runs to success', async (t) => {
  // Pitfall M6: signal semantics differ between Windows and POSIX. On POSIX, SIGINT
  // is reliable. On Windows, child.kill('SIGINT') is unreliable (Node maps it to
  // TerminateProcess for the immediate child but doesn't propagate up). Skip on
  // Windows — the resume-from-running path is exhaustively covered by Plan 04 I-3
  // (which simulates the kill via a direct state-file write).
  if (process.platform === 'win32') {
    t.skip('SIGINT propagation unreliable on Windows; resume-from-running covered by I-3');
    return;
  }

  const dir = mkdtempSync(join(tmpdir(), 'whooflow-killresume-'));
  t.after(() => rmSync(dir, { recursive: true, force: true }));

  // Copy the sleepy fixture into the tmpdir so the marker file is created there.
  const flowPath = join(dir, 'flow.json');
  copyFileSync('test/fixtures/sleepy.json', flowPath);

  // Spawn whooflow as a detached process group so we can SIGINT the whole tree.
  const child = spawn(process.execPath, [BIN, 'run', '--file', flowPath], {
    cwd: dir,
    stdio: 'pipe',
    detached: false,
  });

  // Wait for the marker file to appear (proves the task started).
  const markerPath = join(dir, 'MARKER');
  await waitForMarker(markerPath, 5000);
  const markerMtime1 = readFileSync(markerPath, 'utf8'); // contents = ms timestamp string

  // Send SIGINT.
  child.kill('SIGINT');

  // Wait for the child process to exit.
  const exitCode = await new Promise((resolve) => {
    child.on('exit', (code) => resolve(code));
  });

  // Exit code is 130 on POSIX SIGINT, or 1 if the process exited normally with the
  // exit-code intent set in handleSigint.
  assert.ok(exitCode === 130 || exitCode === 1, `expected 130 or 1, got ${exitCode}`);

  // The state file should show the sleepy task as 'running' (it was killed mid-task).
  // Note: the markFailed call in the failure branch may have flipped 'running' to
  // 'failed' if the runTask Promise resolved with ok:false during the drain. Both
  // 'running' (uninterrupted state preserved) and 'failed' (failure-branch ran) are
  // acceptable post-kill states — what matters is that --resume restarts the task.
  const stateFile = join(dir, 'whooflow.state.json');
  assert.ok(existsSync(stateFile), 'state file must exist after kill');
  const state1 = JSON.parse(readFileSync(stateFile, 'utf8'));
  const status1 = state1.tasks.sleep.status;
  assert.ok(
    status1 === 'running' || status1 === 'failed',
    `post-kill status should be running or failed; got ${status1}`,
  );
  const attemptsAfterKill = state1.tasks.sleep.attempts;
  assert.equal(attemptsAfterKill, 1, 'partial run counts as one attempt');

  // Now resume — but with a much shorter sleep so the resumed run completes quickly.
  // Patch the flow file to use a near-instant cmd for the resume run.
  writeFileSync(flowPath, JSON.stringify({
    tasks: [{
      id: 'sleep',
      cmd: 'node -e "require(\'fs\').writeFileSync(\'MARKER\',String(Date.now()));"',
      depends_on: [],
      retry: { attempts: 2, backoff_ms: 0 },
    }],
  }));

  // Brief wait for filesystem mtime resolution (Windows has 2s mtime granularity on FAT;
  // ext4 / APFS are sub-ms). Adding 30ms is enough on every modern fs.
  await new Promise((r) => setTimeout(r, 30));

  const resumeResult = runCli(['run', '--file', flowPath, '--resume'], { cwd: dir });
  assert.equal(resumeResult.code, 0, `resume should exit 0; stderr=${resumeResult.stderr}`);

  // Verify the marker file was re-written (mtime contents are different).
  const markerMtime2 = readFileSync(markerPath, 'utf8');
  assert.notEqual(markerMtime1, markerMtime2, 'marker should be re-written by the resumed task');

  // Verify the final state.
  const state2 = JSON.parse(readFileSync(stateFile, 'utf8'));
  assert.equal(state2.tasks.sleep.status, 'success', 'task should be success after resume');
  // Attempts: original killed attempt (1) + resumed attempt (1) = 2.
  assert.equal(state2.tasks.sleep.attempts, 2, 'killed attempt + resumed attempt = 2');
});
