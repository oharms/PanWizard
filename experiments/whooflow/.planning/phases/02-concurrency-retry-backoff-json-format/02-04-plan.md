<objective>
Add the pre-flight ergonomics (`--dry-run` and `--list`) so users can sanity-check the DAG without running it, and establish the single-SIGINT-handler-at-startup discipline that pitfall M3 demands. Both flags route through `src/preflight.js` (~60 LOC) — pure functions over the validated flow. Dispatch happens in `src/cli.js` after `loadFlow + validateFlow` succeed but BEFORE `assertWritable + buildInitialState` (so a `--list` on a read-only directory still works — the preflight commands are read-only against the filesystem). Phase 3 will replace the SIGINT handler body with a SIGTERM-then-SIGKILL drain protocol; Plan 04 establishes only the registration shape (single named handler, registered once) so no listener leak ever materializes.

Purpose: This is the last layer that makes Phase 2's parallel runner usable. `--list` prints the task graph for visual inspection; `--dry-run` prints the planned execution waves so users can spot mistakes before kicking off a long parallel run. The depth-grouping algorithm is pure (depth[id] = max(depth[parents]) + 1 over topoOrder) and produces a deterministic output regardless of `--concurrency`. The "single SIGINT handler at startup" discipline is the resource-hygiene capstone of Phase 2 — it prevents the MaxListenersExceededWarning that pitfall M3 calls out, and gives Phase 3 a clean place to add the drain protocol without reorganizing the CLI.

Output: A `whooflow run --file flow.json --list` and `whooflow run --file flow.json --dry-run` that print structural/predictive views of the flow and exit 0 (or 2 on validation/cycle error), plus a CLI that registers exactly one SIGINT and one SIGTERM listener at startup.
</objective>

```js
// Phase 2 Plan 04 — pre-flight commands (--list, --dry-run).
//
// CLI-05 (--dry-run): print planned execution waves and exit 0 without spawning.
//   The "waves" are depth-grouped from the readiness graph (NOT a runtime artifact —
//   the actual scheduler is slot-based per architecture.md Pattern 1). Pure rendering
//   over the validated flow's topoOrder + depends_on.
//
// CLI-06 (--list): print task ids with their depends_on, exit 0.
//
// Both:
//   - Pure functions over validatedFlow (output of validator.validateFlow).
//   - Output to a stream parameter (defaults to process.stdout) for testability.
//   - Validation/cycle errors are surfaced by validateFlow BEFORE these functions run;
//     the CLI catches those errors and exits 2 — consistent with run-mode error path.

/**
 * computeDepthWaves(validatedFlow) -> id[][]
 *
 * Depth-grouping over the readiness graph:
 *   - Roots (no parents) at depth 0.
 *   - Each child at depth = max(depth[parent]) + 1.
 *   - waves[d] = array of task ids whose depth is d.
 *
 * Iterating in topoOrder guarantees parents are visited before children (validator's
 * SCHED-05 contract), so depth[parent] is always populated when we read it.
 */
export function computeDepthWaves(validatedFlow) {
  const depth = Object.create(null);
  const waves = [];

  for (const id of validatedFlow.topoOrder) {
    const parents = validatedFlow.tasks[id].depends_on;
    let d = 0;
    for (const p of parents) {
      // depth[p] is guaranteed defined: validator's topoOrder is parent-before-child.
      if (depth[p] + 1 > d) d = depth[p] + 1;
    }
    depth[id] = d;
    if (!waves[d]) waves[d] = [];
    waves[d].push(id);
  }
  return waves;
}

/**
 * printDryRun(validatedFlow, out)
 *
 * Output format:
 *   wave 1: a, b, c
 *   wave 2: d
 *   wave 3: e, f
 *
 * Note: the runtime scheduler is slot-based (NOT wave-based). Waves here are a
 * display abstraction: tasks at the same depth COULD run in parallel at unbounded
 * concurrency. The actual run will start a child task the instant its specific
 * parents finish, regardless of the wave grouping.
 */
export function printDryRun(validatedFlow, out = process.stdout) {
  const waves = computeDepthWaves(validatedFlow);
  for (let i = 0; i < waves.length; i++) {
    out.write(`wave ${i + 1}: ${waves[i].join(', ')}\n`);
  }
}

/**
 * printList(validatedFlow, out)
 *
 * Output format:
 *   fetch
 *   parse
 *     depends on: fetch
 *   report
 *     depends on: parse
 *
 * Iteration order: topoOrder (deterministic per SCHED-05).
 */
export function printList(validatedFlow, out = process.stdout) {
  for (const id of validatedFlow.topoOrder) {
    out.write(`${id}\n`);
    const deps = validatedFlow.tasks[id].depends_on;
    if (deps.length > 0) {
      out.write(`  depends on: ${deps.join(', ')}\n`);
    }
  }
}
```

### Create `test/preflight.test.js`

Use a tiny in-memory writable stream (a Writable that captures chunks) to test the print functions without reaching for process.stdout.

```js
import { test } from 'node:test';
import assert from 'node:assert/strict';
import { mkdtempSync, rmSync, writeFileSync } from 'node:fs';
import { join } from 'node:path';
import { tmpdir } from 'node:os';
import { loadFlow } from '../src/loader.js';
import { validateFlow } from '../src/validator.js';
import { printList, computeDepthWaves, printDryRun } from '../src/preflight.js';

function captureWriter() {
  const chunks = [];
  return {
    out: { write: (s) => { chunks.push(String(s)); return true; } },
    text: () => chunks.join(''),
  };
}

function makeFlow(t, taskList) {
  const dir = mkdtempSync(join(tmpdir(), 'whooflow-preflight-test-'));
  t.after(() => rmSync(dir, { recursive: true, force: true }));
  const flowPath = join(dir, 'flow.json');
  writeFileSync(flowPath, JSON.stringify({ tasks: taskList }));
  return validateFlow(loadFlow(flowPath));
}

test('CLI-06: printList on linear chain prints id then deps', (t) => {
  const flow = makeFlow(t, [
    { id: 'fetch', cmd: 'node -e ""' },
    { id: 'parse', cmd: 'node -e ""', depends_on: ['fetch'] },
    { id: 'report', cmd: 'node -e ""', depends_on: ['parse'] },
  ]);
  const w = captureWriter();
  printList(flow, w.out);
  const text = w.text();
  // Expected:
  //   fetch
  //   parse
  //     depends on: fetch
  //   report
  //     depends on: parse
  assert.match(text, /^fetch\n/);
  assert.match(text, /parse\n\s+depends on: fetch/);
  assert.match(text, /report\n\s+depends on: parse/);
});
```
