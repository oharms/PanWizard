Phase 1 deliberately structured the scheduler as a slot-based ready-queue at cap=1 specifically so this plan would be a one-line semantic change plus rigorous invariant testing. The `seq` plumbing is added here (not in Plan 03) because the scheduler is the single source of monotonic ordering — the JSON formatter is just a renderer.

Output: A scheduler that runs N tasks concurrently, a CLI that respects `--concurrency N`, and a parallel.test.js suite that proves SCHED-02/SCHED-03/SCHED-04 hold (no wave barriers, parallelizable wall-clock, 100-task perf gate).

**Change 2: Add monotonic seq counter, inject seq into every emitted event.**

Add at the top of `runFlow` (just after the line declaring `running`):
```js
let seqCounter = 0;
function nextSeq() { return ++seqCounter; }
```

Then at every `onEvent({...})` call site, add `seq: nextSeq()` as a property of the event object. There are 4 emit sites in the current scheduler.js:

1. Line 59 (empty-flow run-complete): `onEvent({ type: 'run', status: 'complete', summary, seq: nextSeq() });`
2. Line 78 (normal run-complete in maybeFinish): `onEvent({ type: 'run', status: 'complete', summary, seq: nextSeq() });`
3. Line 92 (task running): add `seq: nextSeq()` to the object literal.
4. Line 109 (task success): add `seq: nextSeq()` to the object literal.
5. Line 125 (task failed): add `seq: nextSeq()` to the object literal.
6. Line 148 (defensive failure path): add `seq: nextSeq()` to the object literal.

The `seq` field MUST be a strictly increasing integer starting from 1 within a single `runFlow` call. Because all `onEvent` calls happen synchronously inside `.then` continuations (which run on the JS microtask queue, single-threaded), there is no race condition on `++seqCounter` — pitfall CC-5 #4 is mitigated by the JS event-loop model itself.

**Add this comment block** above the `seqCounter` declaration so a future reader doesn't try to "simplify" by moving seq generation into the formatter:
```js
// Pitfall CC-5 #4: monotonic seq under interleaved parallelism. seq is generated

// debug log would also want seq for ordering. The formatter is the renderer; the
// scheduler is the canonical event surface. Race-free because .then continuations
```

```js
import { test } from 'node:test';
import assert from 'node:assert/strict';
import { mkdtempSync, rmSync, writeFileSync } from 'node:fs';

function scratchDir(t) {
  const dir = mkdtempSync(join(tmpdir(), 'whooflow-parallel-test-'));
  t.after(() => rmSync(dir, { recursive: true, force: true }));
  return dir;
}

function writeFlowFile(dir, flowObj) {
  const flowPath = join(dir, 'flow.json');
  writeFileSync(flowPath, JSON.stringify(flowObj));
  return flowPath;
}

test('SCHED-04: 100 trivial tasks finish under 3000ms at --concurrency 4 (perf gate)', async (t) => {
  const dir = scratchDir(t);
  // Build 100 independent trivial tasks programmatically — don't commit a 100-task fixture.
  const tasks = [];
  for (let i = 0; i < 100; i++) {
    tasks.push({ id: `t${i}`, cmd: 'node -e ""' });
  }
  const flowPath = writeFlowFile(dir, { tasks });
  const flow = validateFlow(loadFlow(flowPath));
  const state = buildInitialState(flowPath, flow);

  const startMs = Date.now();
  const { summary } = await runFlow(flow, state, { concurrency: 4 });
  const elapsedMs = Date.now() - startMs;

  assert.equal(summary.success, 100, 'all 100 tasks should succeed');
  assert.equal(summary.failed, 0);
  // Generous upper bound per pitfall M4 — actual measured wall-clock should be well under 3000ms
  // on any reasonable machine; budget allows for CI jitter.
  assert.ok(elapsedMs < 3000, `100 trivial tasks took ${elapsedMs}ms at --concurrency 4 — SCHED-04 violated (budget: 3000ms)`);
});

test('SCHED-03 / pitfall M2: 1 slow + 4 fast at --concurrency 2 finishes in ~slow-task-duration (NOT slow + fast/2)', async (t) => {
  const dir = scratchDir(t);
  // 1 task takes ~200ms, 4 tasks take ~50ms each. All independent.
  // Slot-based scheduler: slow task occupies one slot continuously, four fast tasks share the other slot serially.
  // Expected wall-clock: max(slow=200, fast*4/2=100) = ~200ms.
  // Wave-based scheduler (the bug): would be 250ms (slow waits 50ms then runs alone).
  const tasks = [
    { id: 'slow',  cmd: 'node -e "setTimeout(()=>{}, 200)"' },
    { id: 'fast1', cmd: 'node -e "setTimeout(()=>{}, 50)"' },
    { id: 'fast2', cmd: 'node -e "setTimeout(()=>{}, 50)"' },
    { id: 'fast3', cmd: 'node -e "setTimeout(()=>{}, 50)"' },
    { id: 'fast4', cmd: 'node -e "setTimeout(()=>{}, 50)"' },
  ];
  const flowPath = writeFlowFile(dir, { tasks });
  const flow = validateFlow(loadFlow(flowPath));
  const state = buildInitialState(flowPath, flow);

  const startMs = Date.now();
  const { summary } = await runFlow(flow, state, { concurrency: 2 });
  const elapsedMs = Date.now() - startMs;

  assert.equal(summary.success, 5);
  // Generous upper bound (pitfall M4): real wall-clock ~200ms + spawn overhead.
  // 600ms budget absorbs Windows process-spawn overhead (typically 50-150ms per spawn).
  // The KEY signal: this should be substantially less than 5*50 = 250ms + slow_overhead in a
  // serialized run at concurrency=1. We're not asserting a tight equality — just slot-based not
  // grossly worse than serial. The 100-task perf gate covers the tight bound.
  assert.ok(elapsedMs < 1000, `slow+fast took ${elapsedMs}ms at --concurrency 2 — should be ~200ms + spawn overhead, under 1000ms budget`);
});

test('SCHED-02: --concurrency 1 5-task linear chain still runs sequentially (regression for cap=1 path after lift)', async (t) => {
  const dir = scratchDir(t);

    { id: 'a', cmd: 'node -e ""' },

test('CC-5 #4: every emitted event carries a strictly-increasing seq integer starting from 1', async (t) => {
  const dir = scratchDir(t);
  // Diamond keeps it simple but exercises both task and run events.
  const tasks = [
    { id: 'A', cmd: 'node -e ""' },
    { id: 'B', cmd: 'node -e ""', depends_on: ['A'] },
    { id: 'C', cmd: 'node -e ""', depends_on: ['A'] },
    { id: 'D', cmd: 'node -e ""', depends_on: ['B', 'C'] },
  ];
  const flowPath = writeFlowFile(dir, { tasks });
  const flow = validateFlow(loadFlow(flowPath));
  const state = buildInitialState(flowPath, flow);
  const events = [];

  await runFlow(flow, state, { concurrency: 4, onEvent: (e) => events.push(e) });

  // Every event should have a seq.
  for (const e of events) {
    assert.equal(typeof e.seq, 'number', `event ${JSON.stringify(e)} missing seq field`);
    assert.ok(Number.isInteger(e.seq), `event seq must be integer, got ${e.seq}`);
  }
  // seq values must be strictly monotonic increasing starting from 1.
  const seqs = events.map((e) => e.seq);
  assert.equal(seqs[0], 1, `first event seq should be 1, got ${seqs[0]}`);
  for (let i = 1; i < seqs.length; i++) {
    assert.ok(seqs[i] > seqs[i - 1], `seq must strictly increase: index ${i - 1}=${seqs[i - 1]}, ${i}=${seqs[i]}`);
  }
  // Diamond: A:running, A:success, B:running, C:running, B:success, C:success, D:running, D:success, run:complete
  // = 9 events. Seq should be 1..9.
  assert.equal(seqs[seqs.length - 1], events.length);
});

test('parallelism: 4 independent tasks at --concurrency 4 run concurrently (running events overlap before any success)', async (t) => {
  const dir = scratchDir(t);
  // Each task sleeps 100ms. With concurrency 4, all four should be in 'running' state simultaneously
  // before any reaches 'success'.
  const tasks = [
    { id: 't1', cmd: 'node -e "setTimeout(()=>{}, 100)"' },
    { id: 't2', cmd: 'node -e "setTimeout(()=>{}, 100)"' },
    { id: 't3', cmd: 'node -e "setTimeout(()=>{}, 100)"' },
    { id: 't4', cmd: 'node -e "setTimeout(()=>{}, 100)"' },
  ];
  const flowPath = writeFlowFile(dir, { tasks });
  const flow = validateFlow(loadFlow(flowPath));
  const state = buildInitialState(flowPath, flow);
  const events = [];

  await runFlow(flow, state, { concurrency: 4, onEvent: (e) => events.push(e) });

  // Find indices of all running events and the first success event.
  const runningSeqs = events.filter((e) => e.status === 'running').map((e) => e.seq);
  const firstSuccessSeq = events.find((e) => e.status === 'success').seq;

  // All 4 running events should be emitted before the first success event.
  assert.equal(runningSeqs.length, 4);
  for (const rs of runningSeqs) {
    assert.ok(rs < firstSuccessSeq, `running event seq=${rs} should precede first success seq=${firstSuccessSeq} (proves slot-based parallel dispatch)`);
  }
});
```
