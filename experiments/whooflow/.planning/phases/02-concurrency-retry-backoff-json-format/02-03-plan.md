<objective>
Add `--format json` for machine consumers. The implementation is three orthogonal pieces: (1) `createJsonFormatter()` as a sibling factory in `src/formatter.js` — same event-subscriber shape as the text formatter, but writes one NDJSON line per event using `JSON.stringify(structuredClone(event)) + '\n'`; (2) BigInt → Number conversion of `duration_ns` at the scheduler's emit sites so JSON.stringify can serialize the field (BigInt throws); (3) CLI plumbing — a `--format text|json` flag, validated, with the chosen formatter installed as the sole event subscriber. Diagnostic output stays on stderr (Phase 1 already established this discipline; Plan 03 verifies no regression under JSON mode).

Purpose: Phase 2's third orthogonal concern. The JSON event stream is what differentiates whooflow from "Just" or "go-task" — machine consumers (CI, dashboards, log aggregators) can parse the run programmatically. Plan 01 already produced the `seq` field on every event (single source of monotonicity in the scheduler); this plan is the rendering boundary. `structuredClone` at the formatter is defensive — Phase 2's events don't currently embed mutable refs, but applying the clone now means a future event-shape change (e.g., embedding `currentState` for richer dashboards) cannot introduce subtle mutation bugs.

Output: A `--format json` CLI mode that emits NDJSON-clean output on stdout, validated by both unit tests (NDJSON discipline, seq monotonicity, structuredClone safety) and an integration test (full subprocess run with `--format json`, stdout = all-NDJSON, stderr = text-only).
</objective>

```js
/**
 * CLI-04 + pitfall N1: NDJSON event subscriber for machine consumers.
 *
 * Emits exactly one `JSON.stringify(structuredClone(event)) + '\n'` per call to
 * process.stdout.write. structuredClone is defensive — Phase 2's events don't embed
 * mutable refs, but a future event-shape change (e.g., embedding currentState) cannot
 * introduce mutation-after-emit bugs.
 *
 * Hard rules (pitfall N1):
 *   - Exactly one process.stdout.write per event. No console.log (different newline
 *     semantics on some terminals; possible interleaving).
 *   - No pretty-printing. JSON.stringify(event) (no indent argument) — single line.
 *   - '\n' literal — NDJSON spec is \n, never \r\n, regardless of platform.
 *   - All diagnostic output goes to process.stderr — established in Phase 1, this
 *     formatter does NOT touch stderr.
 *   - When --format json is in effect, this formatter is the SOLE subscriber. No
 *     text formatter runs in parallel.
 *
 * Pitfall CC-5 #4: seq is generated in scheduler.js (Plan 01); this formatter just
 * passes it through.
 *
 * Pitfall N4: duration_ns arrives as Number (converted from BigInt at scheduler emit
 * site — see Plan 03 Task 1 Part B). BigInt would throw on JSON.stringify.
 */
export function createJsonFormatter() {
  return function onEvent(event) {
    // structuredClone defensively — even though Phase 2's events are flat primitives,
    // future shape changes (embedded state snapshots, nested error objects) get this
    // protection for free.
    const cloned = structuredClone(event);
    process.stdout.write(JSON.stringify(cloned) + '\n');
  };
}
```

```js
function formatDuration(durationNs) {
  // durationNs may be BigInt (direct call from text path) or Number (post-Plan 03 conversion).
  // BigInt support kept for back-compat in case a test calls the formatter directly.
  if (durationNs === undefined || durationNs === null) return '';
  let nsBig;
  if (typeof durationNs === 'bigint') {
    nsBig = durationNs;
  } else if (typeof durationNs === 'number' && Number.isFinite(durationNs)) {

}
```

```js
import { test } from 'node:test';
import assert from 'node:assert/strict';
import { mkdtempSync, rmSync, writeFileSync } from 'node:fs';
import { join } from 'node:path';
import { tmpdir } from 'node:os';
import { createJsonFormatter } from '../src/formatter.js';
import { loadFlow } from '../src/loader.js';
import { validateFlow } from '../src/validator.js';
import { buildInitialState } from '../src/state.js';
import { runFlow } from '../src/scheduler.js';

function captureStdout(fn) {
  const orig = process.stdout.write.bind(process.stdout);
  const out = [];
  process.stdout.write = (chunk) => {
    out.push(String(chunk));
    return true;
  };
  try { return fn(); } finally { process.stdout.write = orig; }
}

async function captureStdoutAsync(fn) {
  const orig = process.stdout.write.bind(process.stdout);
  const out = [];
  process.stdout.write = (chunk) => {
    out.push(String(chunk));
    return true;
  };
  try { await fn(); } finally { process.stdout.write = orig; }
  return out.join('');
}

test('N1: NDJSON discipline — every line is exactly one JSON.stringify(event) + \\n', () => {
  const fmt = createJsonFormatter();
  const out = captureStdout(() => {
    fmt({ type: 'task', id: 'a', status: 'running', attempts: 1, seq: 1 });
    fmt({ type: 'task', id: 'a', status: 'success', attempts: 1, exit_code: 0, duration_ns: 50_000_000, seq: 2 });
    fmt({ type: 'run', status: 'complete', summary: { success: 1, failed: 0, skipped: 0, total: 1 }, seq: 3 });
  });
  const joined = out.join('');
  // Split on \n and filter empties — every non-empty line must JSON.parse.
  const lines = joined.split('\n').filter((l) => l.length > 0);
  assert.equal(lines.length, 3, `expected 3 NDJSON lines, got ${lines.length}: ${joined}`);
  for (const line of lines) {
    // No \r (NDJSON spec — \n not \r\n).
    assert.equal(line.includes('\r'), false, `line should not contain \\r: ${line}`);
    // Single-line JSON — no embedded newlines.
    assert.ok(!line.includes('\n'), `line should be single-line: ${line}`);
    // Each line must parse.
    const parsed = JSON.parse(line);
    assert.ok(typeof parsed === 'object');
  }
  // Verify shape of the first event.
  const first = JSON.parse(lines[0]);
  assert.equal(first.type, 'task');
  assert.equal(first.id, 'a');
  assert.equal(first.seq, 1);
});

test('CC-5 #4: seq is monotonic across all events of a real diamond run with --format json', async (t) => {
  const dir = mkdtempSync(join(tmpdir(), 'whooflow-jsonfmt-test-'));
  t.after(() => rmSync(dir, { recursive: true, force: true }));
  const flowPath = join(dir, 'flow.json');
  writeFileSync(flowPath, JSON.stringify({
    tasks: [
      { id: 'A', cmd: 'node -e ""' },
      { id: 'B', cmd: 'node -e ""', depends_on: ['A'] },
      { id: 'C', cmd: 'node -e ""', depends_on: ['A'] },
      { id: 'D', cmd: 'node -e ""', depends_on: ['B', 'C'] },
    ]
  }));
  const flow = validateFlow(loadFlow(flowPath));
  const state = buildInitialState(flowPath, flow);

  const captured = await captureStdoutAsync(async () => {
    await runFlow(flow, state, { concurrency: 4, onEvent: createJsonFormatter() });
  });

  const lines = captured.split('\n').filter((l) => l.length > 0);
  const parsed = lines.map((l) => JSON.parse(l));
  // Every event has integer seq, strictly monotonic, starting at 1.
  for (const e of parsed) {
    assert.equal(typeof e.seq, 'number', `event missing seq: ${JSON.stringify(e)}`);
    assert.ok(Number.isInteger(e.seq), `seq must be integer: ${e.seq}`);
  }
  const seqs = parsed.map((e) => e.seq);
  assert.equal(seqs[0], 1, `first seq should be 1, got ${seqs[0]}`);
  for (let i = 1; i < seqs.length; i++) {
    assert.ok(seqs[i] > seqs[i - 1], `seq monotonic violation at index ${i}: ${seqs[i - 1]} -> ${seqs[i]}`);
  }
});

test('structuredClone safety: mutating the original event after emit does not affect emitted JSON', () => {
  const fmt = createJsonFormatter();
  const eventWithNested = {
    type: 'task',
    id: 'x',
    status: 'success',
    attempts: 1,
    exit_code: 0,
    seq: 1,
    nested: { stillRunning: true, list: [1, 2, 3] },
  };
  const out = captureStdout(() => {
    fmt(eventWithNested);
    // Mutate AFTER emit. structuredClone should have insulated the emitted JSON.
    eventWithNested.nested.stillRunning = false;
    eventWithNested.nested.list.push(99);
  });
  const line = out.join('').trim();
  const parsed = JSON.parse(line);
  // The emitted JSON snapshot should reflect the state AT emit time, not after mutation.
  assert.equal(parsed.nested.stillRunning, true, 'structuredClone should preserve emit-time value');
  assert.deepEqual(parsed.nested.list, [1, 2, 3], 'structuredClone should preserve emit-time array');
});

test('N4: duration_ns is a finite Number (not BigInt) — JSON.stringify does not throw', async (t) => {
  // Run a real flow and confirm duration_ns serializes successfully and is a Number.
  const dir = mkdtempSync(join(tmpdir(), 'whooflow-jsonfmt-dur-'));
  t.after(() => rmSync(dir, { recursive: true, force: true }));
  const flowPath = join(dir, 'flow.json');
  writeFileSync(flowPath, JSON.stringify({
    tasks: [{ id: 'one', cmd: 'node -e "setTimeout(()=>{}, 30)"' }],
  }));
  const flow = validateFlow(loadFlow(flowPath));
  const state = buildInitialState(flowPath, flow);

  const captured = await captureStdoutAsync(async () => {
    await runFlow(flow, state, { concurrency: 1, onEvent: createJsonFormatter() });
  });

  const lines = captured.split('\n').filter((l) => l.length > 0);
  const successEvent = lines.map((l) => JSON.parse(l)).find((e) => e.status === 'success');
  assert.ok(successEvent, 'should have a success event');
  assert.equal(typeof successEvent.duration_ns, 'number', `duration_ns should be Number, got ${typeof successEvent.duration_ns}`);
  assert.ok(Number.isFinite(successEvent.duration_ns), `duration_ns should be finite, got ${successEvent.duration_ns}`);
  assert.ok(successEvent.duration_ns > 0, 'duration_ns should be positive for a 30ms task');
});

test('JSON formatter does NOT filter running events (unlike text formatter)', () => {
  const fmt = createJsonFormatter();
  const out = captureStdout(() => {
    fmt({ type: 'task', id: 'a', status: 'running', attempts: 1, seq: 1 });
  });
  const line = out.join('').trim();
  assert.ok(line.length > 0, 'JSON formatter should emit running events (text formatter filters them)');
  const parsed = JSON.parse(line);
  assert.equal(parsed.status, 'running');
});
```
