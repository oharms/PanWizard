### `atomicWriteJson(targetPath, value)` — full prescription

```js
export function atomicWriteJson(targetPath, value) {
  // Same-directory tmp file (NEVER os.tmpdir() — pitfall C4 EXDEV).
  const tmpPath = join(
    dirname(targetPath),
    `.${basename(targetPath)}.tmp.${process.pid}.${Date.now()}`
  );
  const data = JSON.stringify(value, null, 2);

  // 1) Write data to tmp.
  writeFileSync(tmpPath, data, 'utf8');

  // 2) fsync the tmp file — without this, writeFileSync just queued data to the kernel
  //    page cache (pitfall C4: 0-byte tmp on power-cycle between write and rename).
  let fd;
  try {
    fd = openSync(tmpPath, 'r+');
    fsyncSync(fd);
  } finally {
    if (fd !== undefined) closeSync(fd);
  }

  // 3) Atomic rename, with Windows retry on EPERM/EBUSY/EACCES (pitfall C4: AV/OneDrive
  //    can briefly hold the new file open after rename, causing transient errors).
  renameWithRetry(tmpPath, targetPath);
}

function renameWithRetry(from, to) {
  const maxAttempts = 3;
  let lastErr;
  for (let attempt = 0; attempt < maxAttempts; attempt++) {
    try {
      renameSync(from, to);
      return;
    } catch (err) {
      lastErr = err;
      // Only retry on Windows for transient EPERM/EBUSY/EACCES.
      const transient = err.code === 'EPERM' || err.code === 'EBUSY' || err.code === 'EACCES';
      if (process.platform !== 'win32' || !transient || attempt === maxAttempts - 1) {
        throw err;
      }
      // ~50ms busy wait — synchronous, brief, deterministic.
      const until = Date.now() + 50;
      while (Date.now() < until) { /* burn ~50ms */ }
    }
  }
  throw lastErr;
}
```

```js
export function markRunning(state, id) {
  const t = state.tasks[id];
  if (!t) throw new RuntimeError(`markRunning: unknown task ${id}`);
  t.status = 'running';
  t.attempts = (t.attempts ?? 0) + 1;
  t.started_at = new Date().toISOString();

export function markSuccess(state, id, exitCode) {
  const t = state.tasks[id];
  if (!t) throw new RuntimeError(`markSuccess: unknown task ${id}`);
  t.status = 'success';
  t.exit_code = exitCode;
  t.ended_at = new Date().toISOString();

export function markFailed(state, id, exitCode, signal) {
  const t = state.tasks[id];
  if (!t) throw new RuntimeError(`markFailed: unknown task ${id}`);
  t.status = 'failed';
  t.exit_code = exitCode ?? -1;
  if (signal) t.signal = signal;
  t.ended_at = new Date().toISOString();

export function markSkipped(state, id, reason) {
  const t = state.tasks[id];
  if (!t) throw new RuntimeError(`markSkipped: unknown task ${id}`);
  t.status = 'skipped';
  t.ended_at = new Date().toISOString();
  if (reason) t.skip_reason = reason;
```

```js
import { spawn } from 'node:child_process';

/**
 * EXEC-01 + EXEC-02 + pitfall C3 + pitfall C6:
 *   - shell: true for cross-platform cmd execution (cmd.exe on Windows, /bin/sh on POSIX)
 *   - resolve on 'close' (NOT 'exit') — 'close' fires after stdio streams drain,
 *     'exit' fires earlier and can truncate trailing output. DO NOT "simplify" to 'exit'.
 *   - null exit code (signal-killed) is treated as failure with exit_code = -1.
 *   - windowsHide: true suppresses the conhost flash on Windows.
 *   - stdio: 'inherit' so child output streams live to the user's terminal —
 *     this is correct for Phase 1's text format. Phase 2's --format json may
 *     need to capture stdout/stderr; that's a Phase 2 concern.
 */
export function runTask(task) {
  return new Promise((resolve) => {
    let resolved = false;
    let spawnErr = null;

    const child = spawn(task.cmd, {
      shell: true,
      stdio: 'inherit',
      windowsHide: true,
    });

    child.on('error', (err) => {
      // Spawn-time error (e.g., ENOENT for the shell itself, which would be unusual).
      // Defer resolution to 'close' if the child somehow still emits one; otherwise
      // resolve here as a fallback.
      spawnErr = err;
    });

    child.on('close', (code, signal) => {
      if (resolved) return;
      resolved = true;
      if (spawnErr) {
        resolve({ ok: false, exit_code: -1, signal: null, error: spawnErr.message });
        return;
      }
      if (code === null) {
        // Signal-killed (e.g., SIGTERM, OOM). Treat as failure per EXEC-02.
        resolve({ ok: false, exit_code: -1, signal: signal ?? null });
        return;
      }
      resolve({ ok: code === 0, exit_code: code, signal: null });
    });
  });
}
```

The function does NOT update state directly — the scheduler (Plan 04) is the single state writer. `runTask` returns a result object; the scheduler is responsible for calling `markRunning` before, and `markSuccess`/`markFailed` after.

```js
test('EXEC-02 + C3: large stdout output (>64KB) does not truncate / is not lost', async () => {
  // 64KB+ ensures we exercise the pipe-buffer drain that 'close' (not 'exit') waits for.
  // A task that writes 100k chars and exits 0 must report ok: true.
  const result = await runTask({
    id: 't',
    cmd: 'node -e "for (let i = 0; i < 10000; i++) console.log(i.toString().padStart(7, \'0\'));"',
  });
  assert.equal(result.ok, true);
  assert.equal(result.exit_code, 0);
});

test('runTask: spawn-error path returns ok: false with exit_code -1', async () => {
  // A truly impossible cmd. With shell:true the shell will be invoked; it'll print
  // an error and exit non-zero. But if the shell itself is missing (extremely rare),
  // we should still resolve cleanly.
  const result = await runTask({ id: 't', cmd: 'this-command-definitely-does-not-exist-xyzzy 2>/dev/null' });
  // Cross-platform: cmd.exe returns 1 for unknown commands; sh returns 127.
  // Both are non-zero, so ok must be false.
  assert.equal(result.ok, false);
  assert.notEqual(result.exit_code, 0);
});
```

The 64KB-output test is the core C3 verification: if executor listens on `'exit'` instead of `'close'`, this test will sometimes report `ok: true` but with stale/missing output downstream. With `'close'`, the resolution waits for the OS pipe buffer to drain — and on a 100KB stream it WILL be a measurable wait, so the test exercises the pitfall directly.
