### `bin/whooflow.js`

The entry shebang. As small as possible — just imports `main` and pipes to `process.exit`:

```js
#!/usr/bin/env node
import { main } from '../src/cli.js';

main(process.argv.slice(2)).then((code) => {
  process.exitCode = code;
}).catch((err) => {
  // Last-resort safety net — main() should always resolve, never reject.
  // pitfall N1: diagnostics to stderr.
  process.stderr.write(`whooflow: unexpected error: ${err && err.stack || err}\n`);
  process.exitCode = 1;
});
```

After writing, on POSIX run `chmod +x bin/whooflow.js`. (The Phase 1 install path is `node bin/whooflow.js` directly; npm-installed `whooflow` invocation is bin-mapped via package.json and works automatically.)

### `src/cli.js`

```js
import { parseArgs } from 'node:util';
import { readFileSync } from 'node:fs';
import { dirname, resolve, join } from 'node:path';
import { fileURLToPath } from 'node:url';
import { ValidationError, CycleError, RuntimeError, exitCodeFor } from './errors.js';
import { loadFlow } from './loader.js';
import { validateFlow } from './validator.js';
import { buildInitialState, stateFilePath, assertWritable } from './state.js';
import { runFlow } from './scheduler.js';
import { createTextFormatter } from './formatter.js';

// Read package.json for --version. Use import.meta.url to locate ourselves on disk.
function readVersion() {
  try {
    const here = fileURLToPath(import.meta.url);
    const pkgPath = resolve(dirname(here), '..', 'package.json');
    const pkg = JSON.parse(readFileSync(pkgPath, 'utf8'));
    return pkg.version || '0.0.0';
  } catch {
    return '0.0.0';
  }
}

export function printHelp() {
  // Goes to stdout (this is documentation output, not a diagnostic).
  process.stdout.write([
    'whooflow — zero-dependency Node.js DAG task runner',
    '',
    'Usage:',
    '  whooflow run --file <path> [--concurrency N]',
    '  whooflow --help',
    '  whooflow --version',
    '',
    'Options:',
    '  -f, --file <path>        Path to the JSON flow file (required for run)',
    '  -c, --concurrency <N>    Maximum parallel tasks (default: 4) [Phase 2 — Phase 1 ignores]',
    '  -h, --help               Show this help and exit',
    '  -V, --version            Print version and exit',
    '',
    'Exit codes:',
    '  0  All tasks succeeded',
    '  1  At least one task failed',
    '  2  Validation, cycle, or input error (no tasks ran)',
    '',
  ].join('\n'));
}

export function printVersion() {
  process.stdout.write(`whooflow ${readVersion()}\n`);
}

export async function main(argv) {
  let parsed;
  try {
    parsed = parseArgs({
      args: argv,
      options: {
        file:        { type: 'string',  short: 'f' },
        concurrency: { type: 'string',  short: 'c', default: '4' },
        help:        { type: 'boolean', short: 'h' },
        version:     { type: 'boolean', short: 'V' },
      },
      allowPositionals: true,
      strict: true,
    });
  } catch (err) {
    // parseArgs throws on unknown options or missing required values.
    // pitfall N1: diagnostics to stderr.
    process.stderr.write(`whooflow: ${err.message}\n`);
    process.stderr.write(`Try 'whooflow --help' for usage.\n`);
    return 2;
  }

  if (parsed.values.help) {
    printHelp();
    return 0;
  }
  if (parsed.values.version) {
    printVersion();
    return 0;
  }

  // Subcommand dispatch. Phase 1 only ships 'run'.
  const sub = parsed.positionals[0];
  if (sub === undefined) {
    printHelp();
    return 0;
  }
  if (sub !== 'run') {
    process.stderr.write(`whooflow: unknown subcommand '${sub}'\n`);
    process.stderr.write(`Try 'whooflow --help' for usage.\n`);
    return 2;
  }

  // 'run' requires --file.
  const filePath = parsed.values.file;
  if (!filePath) {
    process.stderr.write(`whooflow run: --file <path> is required\n`);
    return 2;
  }

  // Concurrency is parsed but not enforced in Phase 1 (cap=1 hardcoded in scheduler).
  // We DO validate the value here so the user gets feedback on a bad flag now.
  const concurrencyN = Number(parsed.values.concurrency);
  if (!Number.isInteger(concurrencyN) || concurrencyN < 1) {
    process.stderr.write(`whooflow: --concurrency must be a positive integer (got '${parsed.values.concurrency}')\n`);
    return 2;
  }

  // The whole pipeline is wrapped so any thrown error gets mapped to an exit code.
  try {
    const absPath = resolve(filePath);
    // pitfall N3: write-test before any task spawns.
    assertWritable(dirname(absPath));

    const rawFlow = loadFlow(absPath);
    const validatedFlow = validateFlow(rawFlow);
    const state = buildInitialState(absPath, validatedFlow);

    const onEvent = createTextFormatter();
    // Phase 1 forces concurrency=1 (sequential per SCHED-01). The flag is parsed/validated
    // but not propagated yet — Phase 2 will pass concurrencyN through to runFlow.
    const { summary } = await runFlow(validatedFlow, state, { onEvent, concurrency: 1 });

    // CLI-10: 0 on full success, 1 on any failure.
    return summary.failed > 0 ? 1 : 0;
  } catch (err) {
    // Map error to exit code per CLI-10 / src/errors.js exitCodeFor.
    if (err instanceof ValidationError || err instanceof CycleError || err instanceof RuntimeError) {
      // Print a readable error line to stderr (pitfall N1).
      process.stderr.write(`whooflow: ${err.message}\n`);
      // For aggregated ValidationError, also print details.
      if (err instanceof ValidationError && Array.isArray(err.details) && err.details.length > 0) {
        for (const d of err.details) {
          process.stderr.write(`  - ${d.path}: ${d.issue}\n`);
        }
      }
    } else {
      // Unexpected error — full stack to aid triage.
      process.stderr.write(`whooflow: unexpected error: ${err && err.stack || err}\n`);
    }
    return exitCodeFor(err);
  }
}
```

### `test/cli.test.js` — unit tests for `main()` direct invocation

These tests call `main(argv)` as a function and capture stderr/stdout, avoiding spawn overhead.

```js
import { test } from 'node:test';
import assert from 'node:assert/strict';
import { main } from '../src/cli.js';

function captureStreams(fn) {
  const origStdout = process.stdout.write.bind(process.stdout);
  const origStderr = process.stderr.write.bind(process.stderr);
  const stdout = [];
  const stderr = [];
  process.stdout.write = (chunk) => { stdout.push(String(chunk)); return true; };
  process.stderr.write = (chunk) => { stderr.push(String(chunk)); return true; };
  return Promise.resolve(fn()).finally(() => {
    process.stdout.write = origStdout;
    process.stderr.write = origStderr;
  }).then((code) => ({ code, stdout: stdout.join(''), stderr: stderr.join('') }));
}

test('CLI-07: --help prints usage to stdout and exits 0', async () => {
  const { code, stdout } = await captureStreams(() => main(['--help']));
  assert.equal(code, 0);
  assert.match(stdout, /Usage:/);
  assert.match(stdout, /whooflow run --file/);
});

test('CLI-07: --version prints version to stdout and exits 0', async () => {
  const { code, stdout } = await captureStreams(() => main(['--version']));
  assert.equal(code, 0);
  assert.match(stdout, /^whooflow \d+\.\d+\.\d+/);
});

test('no args prints help and exits 0', async () => {
  const { code, stdout } = await captureStreams(() => main([]));
  assert.equal(code, 0);
  assert.match(stdout, /Usage:/);
});

test('unknown subcommand exits 2 with stderr message', async () => {
```

### `test/integration.test.js` — end-to-end via subprocess invocation

These tests spawn `node bin/whooflow.js ...` as a child process. They verify the actual exit codes and the OS-level integration.

```js
import { test } from 'node:test';
import assert from 'node:assert/strict';
import { spawnSync } from 'node:child_process';
import { mkdtempSync, rmSync, copyFileSync, readFileSync, existsSync, writeFileSync } from 'node:fs';
import { join, resolve } from 'node:path';
import { tmpdir } from 'node:os';

const BIN = resolve('bin/whooflow.js');

function runCli(argv, opts = {}) {
  // Use Node to spawn the CLI — works on Windows even without chmod +x.
  const result = spawnSync(process.execPath, [BIN, ...argv], { encoding: 'utf8', ...opts });
  return {
    code: result.status,
    stdout: result.stdout,
    stderr: result.stderr,
  };
}

function isolated(srcRelPath, t) {
  const dir = mkdtempSync(join(tmpdir(), 'whooflow-int-test-'));
  t.after(() => rmSync(dir, { recursive: true, force: true }));
  const dst = join(dir, 'flow.json');
  copyFileSync(srcRelPath, dst);
  return { dir, flow: dst };
}

test('CLI-10: linear chain success exits 0', (t) => {
  const { flow, dir } = isolated('test/fixtures/linear.json', t);
  const { code } = runCli(['run', '--file', flow]);
  assert.equal(code, 0);
  // State file written next to the flow file.
  const stateFile = join(dir, 'whooflow.state.json');
  assert.ok(existsSync(stateFile));
  const state = JSON.parse(readFileSync(stateFile, 'utf8'));
  assert.equal(state.tasks.a.status, 'success');
  assert.equal(state.tasks.c.status, 'success');
  assert.ok(state.ended_at);
});

test('CLI-10: single failing task exits 1', (t) => {
  const { flow } = isolated('test/fixtures/single-fail.json', t);
  const { code } = runCli(['run', '--file', flow]);
  assert.equal(code, 1);
});

test('CLI-10: cycle exits 2 with cycle path on stderr', (t) => {
  const { flow } = isolated('test/fixtures/cycle-2.json', t);
  const { code, stderr } = runCli(['run', '--file', flow]);
  assert.equal(code, 2);
  assert.match(stderr, /cycle detected/);
  // Path must include both ids.
  assert.match(stderr, /a/);
  assert.match(stderr, /b/);
});

test('CLI-10: malformed JSON exits 2 with named error on stderr', (t) => {
  const { flow } = isolated('test/fixtures/malformed.json', t);
  const { code, stderr } = runCli(['run', '--file', flow]);
  assert.equal(code, 2);
  assert.match(stderr, /malformed JSON/);
});

test('CLI-10: nonexistent file exits 2 (not 1)', () => {
  const { code, stderr } = runCli(['run', '--file', '/nonexistent/flow.json']);
  assert.equal(code, 2);
  assert.match(stderr, /flow file not found|cannot write/);

test('CLI-10: missing dependency exits 2 with named error and aggregated details', (t) => {

test('CLI-09: status glyph appears in stdout for successful task', (t) => {
  const { flow } = isolated('test/fixtures/single.json', t);
  const { code, stdout } = runCli(['run', '--file', flow]);
  assert.equal(code, 0);
  // Either Unicode (POSIX TTY) or ASCII (Windows / non-TTY) — both acceptable.
  // Test runner is non-TTY on most CI, so ASCII fallback.
  assert.match(stdout, /\[\s*[v✓]\s*\]/);

test('CLI-08: NO_COLOR env var suppresses ANSI escapes', (t) => {
  const { flow } = isolated('test/fixtures/single.json', t);

  // No ANSI escape sequences (\x1b).
  assert.equal(stdout.includes('\x1b['), false, `NO_COLOR set but ANSI found: ${JSON.stringify(stdout)}`);

test('STATE-03: state file has every task in success after a 3-task linear run', (t) => {
  const { flow, dir } = isolated('test/fixtures/linear.json', t);
  runCli(['run', '--file', flow]);
  const state = JSON.parse(readFileSync(join(dir, 'whooflow.state.json'), 'utf8'));

    assert.equal(state.tasks[id].status, 'success');
    assert.equal(state.tasks[id].exit_code, 0);
    assert.equal(state.tasks[id].attempts, 1);
    assert.ok(state.tasks[id].started_at);
    assert.ok(state.tasks[id].ended_at);

test('CLI-02: --concurrency flag is parsed and accepted (Phase 1 still runs sequentially)', (t) => {
  const { flow } = isolated('test/fixtures/linear.json', t);
  const { code } = runCli(['run', '--file', flow, '--concurrency', '8']);
  assert.equal(code, 0);

test('phase goal: kill resilience — state file is always a valid JSON checkpoint', (t) => {
  // We can't easily kill mid-run synchronously without flakiness, but we can
  // verify the SHAPE invariant: at the end of a successful run, the state file

  const stateContents = readFileSync(join(dir, 'whooflow.state.json'), 'utf8');
  const parsed = JSON.parse(stateContents); // must parse cleanly
  assert.ok(parsed.ended_at);
  assert.ok(parsed.started_at);
  assert.equal(typeof parsed.flow_file, 'string');
});
```
