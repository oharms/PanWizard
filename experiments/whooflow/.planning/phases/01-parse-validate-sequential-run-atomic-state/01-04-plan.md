Create `src/scheduler.js` exporting `runFlow(validatedFlow, state, opts)`. Phase 1 hardcodes `cap = 1` (sequential), but the **structure is the slot-based ready-queue** from architecture.md Pattern 1, NOT a `for (const id of topoOrder) await runTask(...)` shortcut. The reason: Phase 2 lifts the cap by changing one constant; if Phase 1 uses a sequential for-loop, Phase 2 has to rewrite the scheduler entirely.

```js
import { runTask } from './executor.js';
import {
  atomicWriteJson,
  stateFilePath,
  markRunning,
  markSuccess,
  markFailed,
} from './state.js';

/**
 * Phase 1 scheduler: slot-based ready-queue with cap = 1 (sequential).
 *
 * Architecture per architecture.md Pattern 1:
 *   - indegree[id]: count of unresolved dependencies for each task
 *   - ready: FIFO of task ids whose indegree is 0
 *   - running: Set of in-flight task ids (size <= cap)
 *
 * Pump loop: while running.size < cap and ready.length > 0, pop from ready,
 * dispatch via executor, on completion decrement children's indegree, push
 * newly-ready children onto ready, repump.
 *
 * In Phase 1: cap = 1 (sequential, per SCHED-01). Phase 2 lifts this from opts.concurrency.
 *
 * Pitfall mitigations:
 *   - C2 (ready-queue starvation): topoOrder seeds insertion order; depends_on is
 *     guaranteed normalized to [] by loader. indegree decrements ONLY on success.
 *   - SCHED-05 (deterministic): ready array preserves topoOrder ordering â€” siblings
 *     run in topoOrder sequence, not whatever the OS scheduler decides.
 *   - STATE-03 (persist every transition): atomicWriteJson is called after EVERY
 *     state mutation, both pre-spawn (markRunning) and post-spawn (markSuccess/Failed).
 *
 * Phase 1 has NO retry â€” runTask is called exactly once per task. Phase 2 will
 * insert a retry wrapper here. Phase 1 has NO skip-downstream propagation â€” a
 * failed task simply stops dispatching its children (their indegree never reaches
 * 0); they remain `pending` in the final state. The CLI maps `summary.failed > 0`
 * to exit code 1.
 */
export async function runFlow(validatedFlow, state, opts = {}) {
  const cap = opts.concurrency ?? 1;
  const onEvent = opts.onEvent ?? (() => {});
  const stateFile = stateFilePath(state.flow_file);
  const flow = validatedFlow;

  // Build indegree map. flow.children was filled by validator; we just count parents.
  const indegree = Object.create(null);
  for (const id of Object.keys(flow.tasks)) {
    indegree[id] = flow.tasks[id].depends_on.length;
  }

  // Seed ready queue in topoOrder â€” preserves SCHED-05 determinism.
  const ready = [];
  for (const id of flow.topoOrder) {
    if (indegree[id] === 0) ready.push(id);
  }

  const running = new Set();
  const summary = { success: 0, failed: 0, skipped: 0, total: Object.keys(flow.tasks).length };

  // Persist initial state (all tasks pending).
  atomicWriteJson(stateFile, state);

  // Promise that resolves when the run is fully drained.
  let resolveDone;
  const done = new Promise((r) => { resolveDone = r; });

  function maybeFinish() {
    if (running.size === 0 && ready.length === 0) {
      state.ended_at = new Date().toISOString();
      atomicWriteJson(stateFile, state);
      onEvent({ type: 'run', status: 'complete', summary });
      resolveDone();
    }
  }

  function pump() {
    while (running.size < cap && ready.length > 0) {
      const id = ready.shift();
      running.add(id);

      // Mark running + persist + emit event BEFORE spawn (pitfall M4: kill-window safety).
      markRunning(state, id);
      atomicWriteJson(stateFile, state);
      const startedNs = process.hrtime.bigint();
      onEvent({
        type: 'task',
        id,
        status: 'running',
        attempts: state.tasks[id].attempts,
      });

      const task = flow.tasks[id];
      runTask(task)
        .then((result) => {
          running.delete(id);
          const durationNs = process.hrtime.bigint() - startedNs;

          if (result.ok) {
            markSuccess(state, id, result.exit_code);
            atomicWriteJson(stateFile, state);
            summary.success++;
            onEvent({
              type: 'task',
              id,
              status: 'success',
              attempts: state.tasks[id].attempts,
              exit_code: result.exit_code,
              duration_ns: durationNs,
            });
            // Decrement indegree for children, push newly-ready ones.
            for (const c of flow.children[id]) {
              if (--indegree[c] === 0) ready.push(c);
            }
          } else {
            markFailed(state, id, result.exit_code, result.signal);
            atomicWriteJson(stateFile, state);
            summary.failed++;
            onEvent({
              type: 'task',
              id,
              status: 'failed',
              attempts: state.tasks[id].attempts,
              exit_code: result.exit_code,
              duration_ns: durationNs,
            });
            // Phase 1: do NOT decrement children's indegree on failure.
            // Children stay pending. CLI maps summary.failed > 0 to exit code 1.
            // Phase 3 will replace this with a BFS skip-downstream propagation.
          }

          // Continue dispatching.
          maybeFinish();
          pump();
        })
        .catch((err) => {
          // runTask should not reject â€” its Promise always resolves with a result object.
          // Defensive: treat any reject as a failed task.
          running.delete(id);
          markFailed(state, id, -1, null);
          atomicWriteJson(stateFile, state);
          summary.failed++;
          onEvent({
            type: 'task', id, status: 'failed',
            attempts: state.tasks[id].attempts,
            exit_code: -1,
            error: err && err.message,
          });
          maybeFinish();
          pump();
        });
    }
    // If nothing is dispatchable (empty ready, empty running), finalize.
    if (running.size === 0 && ready.length === 0) {
      maybeFinish();
    }
  }

  // Special case: empty flow (loader allows tasks: []).
  if (summary.total === 0) {
    state.ended_at = new Date().toISOString();
    atomicWriteJson(stateFile, state);
    onEvent({ type: 'run', status: 'complete', summary });
    return { state, summary };
  }

  pump();
  await done;
  return { state, summary };
}
```

```js
/**
 * CLI-08: color auto-detects TTY and honors NO_COLOR env var.
 * NO_COLOR convention: https://no-color.org/ â€” any non-empty value disables color.
 */
export function shouldUseColor() {
  if (process.env.NO_COLOR && process.env.NO_COLOR.length > 0) return false;
  return Boolean(process.stdout.isTTY);
}

/**
 * CLI-09 + research/summary.md CC-5 #5: ASCII fallback for Windows or non-TTY.
 * Recommendation: unconditional ASCII on Windows for v1 (cheapest, most reliable).
 * Revisit only if a Windows user with UTF-8 code page complains.
 */
export function shouldUseAscii() {
  if (process.platform === 'win32') return true;
  if (!process.stdout.isTTY) return true;
  return false;
}

const ANSI = {
  reset:  '\x1b[0m',
  green:  '\x1b[32m',
  red:    '\x1b[31m',
  dim:    '\x1b[2m',
};

function paint(color, s) {
  return shouldUseColor() ? `${ANSI[color]}${s}${ANSI.reset}` : s;
}

function glyphSuccess() { return shouldUseAscii() ? '[ v ]' : '[ âś“ ]'; }
function glyphFailed()  { return '[ x ]'; } // ASCII already
function glyphSkipped() { return '[ - ]'; } // ASCII already
function glyphRunning() { return '[ . ]'; } // optional running indicator (low-cost)

function formatDuration(durationNs) {
  // durationNs is a BigInt from process.hrtime.bigint() deltas.
  // Display: <1ms => '<1ms', <1000ms => 'Nms', otherwise => 'N.NNs'.
  if (typeof durationNs !== 'bigint') return '';
  const ns = durationNs;
  const ms = Number(ns / 1000000n);
  if (ms < 1) return '<1ms';
  if (ms < 1000) return `${ms}ms`;
  const s = (Number(ns) / 1e9).toFixed(2);
  return `${s}s`;
}

/**
 * Returns an event subscriber that:
 *   - Prints one glyph line per terminal task transition (success/failed/skipped).
 *   - Skips 'running' transitions (avoid console churn â€” flush on terminal only).
 *   - On 'run complete' prints a one-line summary.
 *
 * Emits to process.stdout (the structured surface). Diagnostic logs (errors,
 * warnings) MUST go to process.stderr â€” NDJSON discipline established now per
 * pitfall N1 even though Phase 1's text format doesn't have a JSON variant yet.
 */
export function createTextFormatter() {
  return function onEvent(event) {
    if (event.type === 'task') {
      switch (event.status) {
        case 'success': {
          const line = `${paint('green', glyphSuccess())} ${event.id}` +
            (event.duration_ns !== undefined ? ` ${paint('dim', `(${formatDuration(event.duration_ns)})`)}` : '');
          process.stdout.write(line + '\n');
          break;
        }
        case 'failed': {
          const detail = event.exit_code !== undefined && event.exit_code !== null
            ? ` (exit ${event.exit_code})`
            : '';
          const line = `${paint('red', glyphFailed())} ${event.id}${paint('dim', detail)}` +
            (event.duration_ns !== undefined ? ` ${paint('dim', `(${formatDuration(event.duration_ns)})`)}` : '');
          process.stdout.write(line + '\n');
          break;
        }
        case 'skipped': {
          const line = `${glyphSkipped()} ${event.id}` +
            (event.skip_reason ? ` ${paint('dim', `(${event.skip_reason})`)}` : '');
          process.stdout.write(line + '\n');
          break;
        }
        // 'running' transitions are not printed â€” only terminal status surfaces in v1.
        default: break;
      }
    } else if (event.type === 'run' && event.status === 'complete') {
      const { success, failed, skipped, total } = event.summary;
      const line = paint('dim', `done: ${success}/${total} success` +
        (failed > 0 ? `, ${failed} failed` : '') +
        (skipped > 0 ? `, ${skipped} skipped` : ''));
      process.stdout.write(line + '\n');
    }
  };
}
```
