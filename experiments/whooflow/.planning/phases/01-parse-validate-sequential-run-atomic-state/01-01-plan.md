```js
// All errors carry a stable `.code` for the exit-code map.
export class ValidationError extends Error {
  constructor(message, details = []) {
    super(message);
    this.name = 'ValidationError';
    this.code = 'EVALIDATION';
    this.details = details; // for aggregated schema errors (M5 mitigation)
  }
}

export class CycleError extends Error {
  constructor(message) {
    super(message);
    this.name = 'CycleError';
    this.code = 'ECYCLE';
  }
}

export class RuntimeError extends Error {
  constructor(message, cause) {
    super(message);
    this.name = 'RuntimeError';
    this.code = 'ERUNTIME';
    this.cause = cause;
  }
}

// Maps any thrown error to the project's exit code policy:
//   ValidationError or CycleError -> 2
//   RuntimeError or anything else -> 1
//   undefined (success path) -> 0
export function exitCodeFor(err) { /* 0|1|2 */ }
```

Create `src/errors.js` exporting THREE error classes and ONE function (use ESM `export class` / `export function`):

1. **`ValidationError`** — extends `Error`. Constructor: `(message, details = [])`. Sets `.name = 'ValidationError'`, `.code = 'EVALIDATION'`, `.details = details`. The `details` array exists so PARSE-01..PARSE-04 (Plan 02) can aggregate multiple schema errors into one throw — mitigates pitfall M5 (first-fail validation).

2. **`CycleError`** — extends `Error`. Constructor: `(message)`. Sets `.name = 'CycleError'`, `.code = 'ECYCLE'`. Used by validator (Plan 02) for cycle detection output.

3. **`RuntimeError`** — extends `Error`. Constructor: `(message, cause)`. Sets `.name = 'RuntimeError'`, `.code = 'ERUNTIME'`, `.cause = cause`. Used for non-validation failures (e.g., state-write EACCES, unexpected throws).

4. **`exitCodeFor(err)`** function — exported. Maps an error (or undefined) to an exit code per CLI-10:
   - `err == null` → return `0` (success path; called with `undefined` after a clean run)
   - `err instanceof ValidationError` OR `err instanceof CycleError` → return `2`
   - any other error (including `RuntimeError`) → return `1`

Code comment at the top of the file: `// Phase 1 — typed errors define the exit-code contract per CLI-10. Adding a new error class? Update exitCodeFor too.`

Create `test/errors.test.js` using `node:test` + `node:assert/strict`:

```js
import { test } from 'node:test';
import assert from 'node:assert/strict';
import { ValidationError, CycleError, RuntimeError, exitCodeFor } from '../src/errors.js';

test('ValidationError carries name, code, and details', () => {
  const e = new ValidationError('bad shape', [{ path: 'tasks[0].id', issue: 'missing' }]);
  assert.equal(e.name, 'ValidationError');
  assert.equal(e.code, 'EVALIDATION');
  assert.equal(e.details.length, 1);
  assert.ok(e instanceof Error);
});

test('CycleError carries name and code', () => {
  const e = new CycleError('cycle detected: a → b → a');
  assert.equal(e.name, 'CycleError');
  assert.equal(e.code, 'ECYCLE');
});

test('RuntimeError carries name, code, and cause', () => {
  const cause = new Error('disk full');
  const e = new RuntimeError('write failed', cause);
  assert.equal(e.code, 'ERUNTIME');
  assert.equal(e.cause, cause);
});

test('exitCodeFor maps ValidationError and CycleError to 2', () => {
  assert.equal(exitCodeFor(new ValidationError('x')), 2);
  assert.equal(exitCodeFor(new CycleError('y')), 2);
});

test('exitCodeFor maps RuntimeError and generic Error to 1', () => {
  assert.equal(exitCodeFor(new RuntimeError('z')), 1);
  assert.equal(exitCodeFor(new Error('generic')), 1);
});

test('exitCodeFor maps null/undefined to 0', () => {
  assert.equal(exitCodeFor(undefined), 0);
  assert.equal(exitCodeFor(null), 0);
});
```

Both files MUST use ESM `import`/`export`. The `import` of `'../src/errors.js'` MUST include the `.js` extension (ESM requires it).
