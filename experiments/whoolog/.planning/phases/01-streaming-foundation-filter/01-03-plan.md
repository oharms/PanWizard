**2. Create `lib/filter.js`:**

```js
'use strict';

const { parseArgs } = require('node:util');
const { sources, lines } = require('./source');
const { decode } = require('./decode');
const { compileWhere } = require('./where');
const { compileTimeFilter } = require('./time-filter');
const { writeJsonl } = require('./format');

const FILTER_OPTIONS = {
  files:             { type: 'string',  multiple: true },
  where:             { type: 'string',  multiple: true },     // CLI-02 + WHR-02: multiple --where AND-ed
  since:             { type: 'string' },
  until:             { type: 'string' },
  'ts-field':        { type: 'string' },
  'keep-missing-ts': { type: 'boolean' },
  'ts-required':     { type: 'boolean' },
  strict:            { type: 'boolean' },
  format:            { type: 'string' },
  help:              { type: 'boolean', short: 'h' },
};

function printFilterHelp() {
  process.stdout.write([
    'Usage: whoolog filter [options]',
    '',
    'Stream-filter a JSONL log to stdout.',
    '',
    'Input:',
    '  --files <pattern>...    File path or glob (e.g., "logs/*.jsonl"). Repeatable.',
    '                          When omitted, reads JSONL from stdin (when piped).',
    '',
    'Predicates:',
    '  --where <expr>...       Field predicate. Repeatable; multiple --where flags AND together.',
    '                          Operators (longer-first lexed): >=, <=, !=, ~, =, >, <',
    '                          Value typing: bare digits → Number, "quoted" → string,',
    '                                        true/false/null → literals, otherwise → string.',
    '                          ~ is a JS regex (case-sensitive, no flags).',
    '                          = and != are TYPE-AWARE: 5 (number) does NOT equal "5" (string).',
    '                          Numeric ops coerce both sides to Number; non-numeric LHS → no match.',
    '',
    'Time filtering:',
    '  --since <date>          Inclusive lower bound. Date-only (YYYY-MM-DD) parses as UTC midnight.',
    '  --until <date>          Exclusive upper bound.',
    '                          ISO datetimes without a Z suffix or offset are REJECTED to avoid',
    '                          local-vs-UTC ambiguity (e.g., use 2026-04-01T00:00:00Z, not …T00:00:00).',
    '  --ts-field <name>       Timestamp field name (default: ts).',
    '  --keep-missing-ts       Keep rows that have no timestamp field (default: drop).',
    '  --ts-required           Exit 1 if any row lacks the timestamp field.',
    '',
    'Malformed lines:',
    '  --strict                Exit 1 on the first malformed JSON line.',
    '                          (Default: warn on stderr "whoolog: skipping malformed line at <file>:<lineno>" and continue.)',
    '',
    'Output:',
    '  --format <json>         Currently only "json" (one JSONL row per line). Default: json.',
    '                          (--format table arrives in v1.1 alongside count and histogram.)',
    '',
    '  -h, --help              Show this help and exit 0.',
    '',
    'Exit codes:',
    '  0  success (including empty result set)',
    '  1  runtime error (file not found, --strict malformed line, --ts-required missing-ts)',
    '  2  usage error (unknown flag, empty --where, naive ISO datetime, etc.)',
    '',
    'Examples:',
    '  whoolog filter --files app.jsonl --where level=error',
    '  whoolog filter --files app.jsonl --where "usage.input_tokens>1000"',
    '  whoolog filter --files "logs/*.jsonl" --where level=error --where msg~timeout',
    '  whoolog filter --files app.jsonl --since 2026-04-01 --until 2026-05-01',
    '  cat app.jsonl | whoolog filter --where level=error',
    '',
  ].join('\n'));
}

async function run(argv) {
  let values;
  try {
    ({ values } = parseArgs({
      args: argv,
      options: FILTER_OPTIONS,
      allowPositionals: false,
      strict: true,    // unknown flags → throw
    }));
  } catch (err) {
    process.stderr.write(`whoolog: ${err.message}\n`);
    process.exit(2);
  }

  if (values.help) {
    printFilterHelp();
    process.exit(0);
  }

  // SRC-05: no --files AND stdin is a TTY → print help, exit 0
  const noFiles = !values.files || values.files.length === 0;
  if (noFiles && process.stdin.isTTY) {
    printFilterHelp();
    process.exit(0);
  }

  // FMT-04: format defaults to 'json' (table is Phase 3)
  const format = values.format || 'json';
  if (format !== 'json') {
    process.stderr.write(`whoolog: --format ${format} is not yet supported (table mode arrives in Phase 3); use --format json\n`);
    process.exit(2);
  }

  // Compile predicates ONCE at startup (WHR-07, KEY-02)
  let predicates, timePred;
  try {
    predicates = (values.where || []).map(compileWhere);
    timePred = compileTimeFilter({
      since: values.since,
      until: values.until,
      tsField: values['ts-field'],
      keepMissing: values['keep-missing-ts'],
      required: values['ts-required'],
    });
  } catch (err) {
    process.stderr.write(`whoolog: ${err.message}\n`);
    process.exit(err.usage ? 2 : 1);
  }

  const passes = (obj) => {
    for (const p of predicates) if (!p(obj)) return false;   // WHR-02: AND
    if (!timePred(obj)) return false;
    return true;
  };

  // Stream
  for await (const source of sources({ files: values.files })) {
    for await (const { obj } of decode(lines(source), { strict: values.strict })) {
      if (passes(obj)) writeJsonl(obj);
    }
  }
  // Empty result is success — exit 0 implicit
}

module.exports = { run };
```

**Locked behaviors:**
- `parseArgs` with `strict: true` rejects unknown flags → caller exits 2 (CLI-03).
- `--where ""` (empty string) flows into `compileWhere` which throws a usage error → exit 2 (Pitfall 26 + WHR-01).
- Multiple `--where` flags collected via `multiple: true`; ANDed via `predicates.every()` (CLI-02 + WHR-02).
- Predicates compiled ONCE (WHR-07).
- Time filter compiled ONCE (TIM-01..04).
- Output via `writeJsonl(obj)` — streaming, O(1) memory (FLT-01).
- Empty result → exit 0 (CLI-05, Pitfall 18).

**Anti-patterns:**
- Do NOT collect rows into an array and emit at the end — that breaks FLT-01 streaming (Pitfall 1 family).
- Do NOT call `console.log` — use `process.stdout.write` (research/stack.md: console.log adds \n + extra processing; for the hot loop, direct write is correct).
- Do NOT swallow errors from compileWhere/compileTimeFilter without checking `err.usage`.
- Do NOT default `--format` to anything other than `'json'` in Phase 1.

'use strict';

const { test } = require('node:test');
const assert = require('node:assert/strict');
const { decode } = require('../lib/decode');

// Helper: drive decode() with a synthetic line iterator and capture stderr
async function runDecode(linesArr, opts = {}) {
  async function* fakeLines() {
    for (const item of linesArr) yield item;
  }
  const out = [];
  const origWrite = process.stderr.write.bind(process.stderr);
  let stderr = '';
  process.stderr.write = (s) => { stderr += s; return true; };
  try {
    for await (const item of decode(fakeLines(), opts)) out.push(item);
  } finally {
    process.stderr.write = origWrite;
  }
  return { out, stderr };
}

test('decode: parses valid JSON lines', async () => {
  const { out, stderr } = await runDecode([
    { line: '{"a":1}', lineno: 1, file: 'x.jsonl' },
    { line: '{"b":2}', lineno: 2, file: 'x.jsonl' },
  ]);
  assert.equal(out.length, 2);
  assert.deepEqual(out[0].obj, { a: 1 });
  assert.deepEqual(out[1].obj, { b: 2 });
  assert.equal(stderr, '');
});

test('decode: skips blank lines silently', async () => {
  const { out, stderr } = await runDecode([
    { line: '{"a":1}', lineno: 1, file: 'x.jsonl' },
    { line: '',        lineno: 2, file: 'x.jsonl' },
    { line: '   ',     lineno: 3, file: 'x.jsonl' },
    { line: '{"b":2}', lineno: 4, file: 'x.jsonl' },
  ]);
  assert.equal(out.length, 2);
  assert.equal(stderr, '');
});

test('decode: malformed line warn-and-skip (DEC-02)', async () => {
  const { out, stderr } = await runDecode([
    { line: '{"a":1}',  lineno: 1, file: 'x.jsonl' },

'use strict';

const { test, before, after } = require('node:test');
const assert = require('node:assert/strict');
const fs = require('node:fs');
const path = require('node:path');
const os = require('node:os');
const { spawn } = require('node:child_process');
const { runCLI, BIN } = require('./runCLI');

let tmpdir;
before(() => { tmpdir = fs.mkdtempSync(path.join(os.tmpdir(), 'whoolog-test-')); });
after(() => { fs.rmSync(tmpdir, { recursive: true, force: true }); });

function writeFixture(name, lines) {
  const file = path.join(tmpdir, name);
  fs.writeFileSync(file, lines.join('\n') + (lines.length > 0 ? '\n' : ''));
  return file;

test('filter: EPIPE handled cleanly when downstream closes (CLI-04, Pitfall 17)', async () => {
  // Generate a large enough fixture that the stream isn't already drained when we close stdout
  const lines = [];
  for (let i = 0; i < 5000; i++) lines.push(JSON.stringify({ x: i }));
  const file = writeFixture('big.jsonl', lines);
  const child = spawn(process.execPath, [BIN, 'filter', '--files', file]);
  // Read first chunk, then destroy the read end → causes EPIPE on next write in child
  let firstChunk = '';
  await new Promise((resolve, reject) => {
    child.stdout.once('data', (chunk) => {
      firstChunk = chunk.toString();
      child.stdout.destroy();
      resolve();
    });
    child.stdout.once('error', () => resolve());  // EPIPE on read side may also trigger here
    setTimeout(() => reject(new Error('child never wrote anything')), 5000);
  });
  const code = await new Promise((resolve) => child.on('close', resolve));
  assert.equal(code, 0, 'EPIPE should result in clean exit 0, not a crash');
  assert.match(firstChunk, /"x":0/);
});
