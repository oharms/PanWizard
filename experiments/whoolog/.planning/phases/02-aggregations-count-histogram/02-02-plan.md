```js
'use strict';

const { parseArgs } = require('node:util');
const { sources, lines } = require('./source');
const { decode } = require('./decode');
const { compileWhere } = require('./where');
const { compileTimeFilter } = require('./time-filter');
const { compileKey } = require('./resolve-key');
const { writeJsonl } = require('./format');

const COUNT_OPTIONS = {
  files:             { type: 'string',  multiple: true },
  by:                { type: 'string' },
  where:             { type: 'string',  multiple: true },
  since:             { type: 'string' },
  until:             { type: 'string' },
  'ts-field':        { type: 'string' },
  'keep-missing-ts': { type: 'boolean' },
  'ts-required':     { type: 'boolean' },
  strict:            { type: 'boolean' },
  format:            { type: 'string' },
  help:              { type: 'boolean', short: 'h' },
};

function printCountHelp() {
  process.stdout.write([
    'Usage: whoolog count --by <field> [options]',
    '',
    'Group rows by a single field and count occurrences. Emits {value, count} JSONL,',
    'sorted by count descending; ties broken by value ascending (stable, deterministic).',
    '',
    'Required:',
    '  --by <field>            Field to group by. Nested keys allowed (a.b.c).',
    '                          Missing/undefined values are emitted as {value: null, count: N}.',
    '',
    'Input:',
    '  --files <pattern>...    File path or glob. Repeatable. Reads stdin when omitted (when piped).',
    '',
    'Predicates (pre-aggregation):',
    '  --where <expr>...       Field predicate, repeatable, AND-ed. Same DSL as `filter`.',
    '  --since <date>          Inclusive lower bound (date-only → UTC midnight).',
    '  --until <date>          Exclusive upper bound.',
    '  --ts-field <name>       Timestamp field for --since/--until (default: ts).',
    '  --keep-missing-ts       Keep rows without --ts-field timestamp (default: drop).',
    '  --ts-required           Exit 1 if any row lacks --ts-field.',
    '',
    'Malformed lines:',
    '  --strict                Exit 1 on first malformed JSON line. Default: warn-and-skip.',
    '',
    'Output:',
    '  --format <json>         Currently only "json". Default: json. (Table mode: Phase 3.)',
    '',
    '  -h, --help              Show this help and exit 0.',
    '',
    'Memory: O(distinct values). Avoid `--by` on high-cardinality fields (e.g., unique timestamps);',
    '        use `histogram --by ts --bucket 1h` for time-series instead.',
    '',
    'Examples:',
    '  whoolog count --files app.jsonl --by level',
    '  whoolog count --files app.jsonl --by usage.model --where level=error',
    '  whoolog count --files "logs/*.jsonl" --by agent --since 2026-04-01',
    '',
  ].join('\n'));
}

async function run(argv) {
  let values;
  try {
    ({ values } = parseArgs({
      args: argv,
      options: COUNT_OPTIONS,
      allowPositionals: false,
      strict: true,
    }));
  } catch (err) {
    process.stderr.write(`whoolog: ${err.message}\n`);
    process.exit(2);
  }

  if (values.help) {
    printCountHelp();
    process.exit(0);
  }

  const noFiles = !values.files || values.files.length === 0;
  if (noFiles && process.stdin.isTTY) {
    printCountHelp();
    process.exit(0);
  }

  if (!values.by) {
    process.stderr.write('whoolog: count requires --by <field>\n');
    process.exit(2);
  }

  const format = values.format || 'json';
  if (format !== 'json') {
    process.stderr.write(`whoolog: --format ${format} is not yet supported (table mode arrives in Phase 3); use --format json\n`);
    process.exit(2);
  }

  let predicates, timePred, getByField;
  try {
    predicates = (values.where || []).map(compileWhere);
    timePred = compileTimeFilter({
      since: values.since,
      until: values.until,
      tsField: values['ts-field'],
      keepMissing: values['keep-missing-ts'],
      required: values['ts-required'],
    });
    getByField = compileKey(values.by);
  } catch (err) {
    process.stderr.write(`whoolog: ${err.message}\n`);
    process.exit(err.usage ? 2 : 1);
  }

  const passes = (obj) => {
    for (const p of predicates) if (!p(obj)) return false;
    if (!timePred(obj)) return false;
    return true;
  };

  // Accumulate. Map<string_key, { originalValue, count }>.
  // The string key uses JSON.stringify so different JS types produce different keys
  // (number 5 → "5", string "5" → "\"5\"", null → "null", undefined → "undefined").
  // The originalValue preserves type fidelity for the output row (Pitfall: numeric-vs-string value).
  const counts = new Map();

  for await (const source of sources({ files: values.files })) {
    for await (const { obj } of decode(lines(source), { strict: values.strict })) {
      if (!passes(obj)) continue;
      const raw = getByField(obj);
      // Normalize undefined → null in OUTPUT (cleaner JSON), but keep them distinct in the KEY
      // (so missing-vs-explicit-null are still bucketed separately if both occur).
      const keyStr = JSON.stringify(raw);            // raw === undefined → keyStr === undefined? No — JSON.stringify(undefined) returns undefined (not the string "undefined")
      // Use a sentinel for undefined to ensure a stable string key:
      const key = (raw === undefined) ? '__undef__' : keyStr;
      const existing = counts.get(key);
      if (existing) {
        existing.count += 1;
      } else {
        counts.set(key, {
          // Output value: undefined → null (cleaner JSON), null → null, others → as-is
          outputValue: (raw === undefined) ? null : raw,
          count: 1,
        });
      }
    }
  }

  // CNT-02: sort by count desc, ties broken by serialized key string ascending.
  const sorted = [...counts.entries()].sort(([ka, a], [kb, b]) => {
    const diff = b.count - a.count;
    if (diff !== 0) return diff;
    if (ka < kb) return -1;
    if (ka > kb) return 1;
    return 0;
  });
  for (const [, { outputValue, count }] of sorted) {
    writeJsonl({ value: outputValue, count });
  }
  // Empty result is success — exit 0 implicit.
}

module.exports = { run };
```

**Locked behaviors (verbatim from research §2 + Open Question §1):**
- `--by` is REQUIRED; missing → exit 2 BEFORE opening any file.
- The Map key is `JSON.stringify(raw)` (research §Implementation §2 + Pitfall guard "Missing vs Null vs Undefined"). For `undefined`, use the sentinel `'__undef__'` because `JSON.stringify(undefined)` returns `undefined` (not a string), which would corrupt the Map key.
- The OUTPUT `value` field is the original JS value when present (number stays number, string stays string, null stays null) — `undefined` becomes `null` in output for clean JSON serialization (Open Question §1: "use null").
- Sort: count descending; ties broken by serialized key string ascending. Locking the tie-break on the SERIALIZED key (not the output value) guarantees deterministic ordering even when output values are mixed types (e.g., a tie between number `5` and string `"5"` resolves stably).
- Output row shape: `{ value, count }` — field order matters.
- Empty result → exit 0.

```js
'use strict';

const { test, before, after } = require('node:test');
const assert = require('node:assert/strict');
const fs = require('node:fs');
const path = require('node:path');
const os = require('node:os');
const { runCLI } = require('./runCLI');

let tmpdir;
before(() => { tmpdir = fs.mkdtempSync(path.join(os.tmpdir(), 'whoolog-count-test-')); });
after(() => { fs.rmSync(tmpdir, { recursive: true, force: true }); });

function writeFixture(name, lines) {
  const file = path.join(tmpdir, name);
  fs.writeFileSync(file, lines.join('\n') + (lines.length > 0 ? '\n' : ''));
  return file;
}

function parseJsonlOutput(stdout) {
  return stdout.trim().split('\n').filter(Boolean).map(l => JSON.parse(l));
}

test('count: basic top-level field, sorted desc by count (CNT-02)', () => {
  const file = writeFixture('app.jsonl', [
    JSON.stringify({ level: 'error' }),
    JSON.stringify({ level: 'info' }),
    JSON.stringify({ level: 'error' }),
    JSON.stringify({ level: 'warn' }),
    JSON.stringify({ level: 'error' }),
    JSON.stringify({ level: 'info' }),
  ]);
  const r = runCLI(['count', '--files', file, '--by', 'level']);
  assert.equal(r.status, 0, `expected exit 0, got ${r.status}: stderr=${r.stderr}`);
  const rows = parseJsonlOutput(r.stdout);
  assert.equal(rows.length, 3);
  // Sorted desc by count: error=3, info=2, warn=1
  assert.deepEqual(rows[0], { value: 'error', count: 3 });
  assert.deepEqual(rows[1], { value: 'info',  count: 2 });
  assert.deepEqual(rows[2], { value: 'warn',  count: 1 });
});

test('count: nested key (CNT-01)', () => {
  const file = writeFixture('models.jsonl', [
    JSON.stringify({ usage: { model: 'gpt-4' } }),
    JSON.stringify({ usage: { model: 'gpt-4' } }),
    JSON.stringify({ usage: { model: 'claude-3' } }),
    JSON.stringify({ usage: null }),                  // null intermediate → undefined → output null
    JSON.stringify({}),                               // missing usage → undefined → output null
  ]);
  const r = runCLI(['count', '--files', file, '--by', 'usage.model']);
  assert.equal(r.status, 0);
  const rows = parseJsonlOutput(r.stdout);
  // gpt-4: 2, claude-3: 1, null: 2 (missing + null intermediate both bucket together via JSON.stringify('__undef__'))
  // Wait — JSON.stringify(undefined) is undefined; the sentinel is '__undef__'.
  // null intermediate: compileKey returns undefined → key '__undef__' → outputValue null
  // missing: same → same key → same outputValue
  // So 2 distinct values: gpt-4 (2), claude-3 (1), null (2)
  assert.equal(rows.length, 3);
  assert.deepEqual(rows[0], { value: 'gpt-4', count: 2 });   // count 2 wins on count desc
  // Tie-break between claude-3 (count 1) and null (count 2): null has higher count, so null comes second
  // Actually null has count 2, gpt-4 has count 2 — tie → break on serialized key asc
  // serialized key for null: '__undef__'
  // serialized key for gpt-4: '"gpt-4"'
```
