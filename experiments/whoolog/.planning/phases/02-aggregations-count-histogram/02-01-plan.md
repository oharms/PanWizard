```js
'use strict';

// Unit → milliseconds. HST-02 lists 1m, 5m, 1h, 1d. The regex accepts any positive integer
// followed by one of m/h/d. Seconds (s) is INTENTIONALLY NOT in this map — research §3.
const UNITS = {
  m: 60_000,         // 1 minute = 60,000 ms
  h: 3_600_000,      // 1 hour   = 3,600,000 ms
  d: 86_400_000,     // 1 day    = 86,400,000 ms
};

function parseBucket(spec) {
  if (typeof spec !== 'string' || spec.length === 0) {
    const e = new Error('--bucket value is required (e.g., 1m, 5m, 1h, 1d)');
    e.usage = true;
    throw e;
  }
  const m = /^(\d+)([mhd])$/.exec(spec);
  if (!m) {
    const e = new Error(`--bucket ${JSON.stringify(spec)}: must be a positive integer followed by m, h, or d (e.g., 1m, 5m, 1h, 1d)`);
    e.usage = true;
    throw e;
  }
  const n = parseInt(m[1], 10);
  if (n <= 0) {
    const e = new Error(`--bucket ${JSON.stringify(spec)}: count must be > 0`);
    e.usage = true;
    throw e;
  }
  return n * UNITS[m[2]];
}

// HST-03 bucket-start formula. Math.floor on integer/integer is exact in JS Number range
// for epoch-ms values (~10^12, well below MAX_SAFE_INTEGER ~9×10^15). Research §Implementation §1.
//
// Boundary case: tsMs = bucketMs * N → Math.floor(N) * bucketMs = bucketMs * N.
// Therefore a row at exactly the bucket boundary lands in THAT bucket (inclusive lower).
// This is the contract checked by TST-04 in Plan 02-02.
function bucketStart(tsMs, bucketMs) {
  return Math.floor(tsMs / bucketMs) * bucketMs;
}

function bucketStartIso(tsMs, bucketMs) {
  return new Date(bucketStart(tsMs, bucketMs)).toISOString();
}

module.exports = { parseBucket, bucketStart, bucketStartIso };
```

**Locked details (do not deviate):**
- Use `Math.floor(tsMs / bucketMs) * bucketMs` exactly as written. Do NOT use `Math.trunc`, `Math.round`, or `Math.ceil` (Pitfall 10 — bucket boundary off-by-one).
- The unit map MUST be `{ m, h, d }` only — do NOT add `s` (seconds) even though it would be a one-line addition; HST-02 explicitly lists only `1m`, `5m`, `1h`, `1d`.
- `parseBucket` MUST throw with `.usage = true` so `histogram.run()` can distinguish usage errors (exit 2) from runtime errors (exit 1) — same pattern as `compileWhere` and `compileTimeFilter` in Phase 1.
- The integer underscore syntax (`60_000`) is supported in Node 18.3+ (engines.node = ">=18.3.0" already locked in package.json). Do not change to `60000`.

**Anti-pattern check:**
- NO floating-point arithmetic that could introduce drift (e.g., `tsMs / bucketMs | 0` would silently overflow on values > 2^31). `Math.floor` is the correct primitive.
- NO `require('node:util')` or other built-ins — this module is pure math + a regex.
- NO calendar-aware logic (DST, month boundaries, leap seconds). Out of scope per project.md.

Module exports: `module.exports = { parseBucket, bucketStart, bucketStartIso };`

Create `lib/histogram.js` following the same shape as `lib/filter.js` (Phase 1 reference). Accumulate rows in a `Map<bucket_start_ms, count>`, then sort + emit:

```js
'use strict';

const { parseArgs } = require('node:util');
const { sources, lines } = require('./source');
const { decode } = require('./decode');
const { compileWhere } = require('./where');
const { compileTimeFilter } = require('./time-filter');
const { compileKey } = require('./resolve-key');
const { parseBucket, bucketStart } = require('./time-bucket');
const { writeJsonl } = require('./format');

const HISTOGRAM_OPTIONS = {
  files:             { type: 'string',  multiple: true },
  by:                { type: 'string' },
  bucket:            { type: 'string' },
  where:             { type: 'string',  multiple: true },
  since:             { type: 'string' },
  until:             { type: 'string' },
  'ts-field':        { type: 'string' },
  'keep-missing-ts': { type: 'boolean' },
  'ts-required':     { type: 'boolean' },
  strict:            { type: 'boolean' },
  format:            { type: 'string' },
  help:              { type: 'boolean', short: 'h' },
};

function printHistogramHelp() {
  process.stdout.write([
    'Usage: whoolog histogram --by <ts-field> --bucket <size> [options]',
    '',
    'Time-bucket matching rows by a timestamp field. Emits {bucket_start, count} JSONL,',
    'sorted by bucket_start ascending. Buckets with zero matching rows are OMITTED from output.',
    '',
    'Required:',
    '  --by <field>            Timestamp field to bucket on (e.g., ts). Nested keys allowed (a.b.c).',
    '  --bucket <size>         Bucket size: 1m, 5m, 1h, 1d. Calendar-aware bucketing is NOT supported.',
    '                          (The bucket formula is Math.floor(ts_ms / bucket_ms) * bucket_ms;',
    '                           a row at exactly the bucket boundary belongs to the LATER bucket.)',
    '',
    'Input:',
    '  --files <pattern>...    File path or glob. Repeatable. Reads stdin when omitted (when piped).',
    '',
    'Predicates (pre-aggregation):',
    '  --where <expr>...       Field predicate, repeatable, AND-ed. Same DSL as `filter`.',
    '  --since <date>          Inclusive lower bound (date-only → UTC midnight).',
    '  --until <date>          Exclusive upper bound.',
    '  --ts-field <name>       Timestamp field for --since/--until (default: ts).',
    '  --keep-missing-ts       Keep rows without --ts-field timestamp (default: drop).',
    '  --ts-required           Exit 1 if any row lacks --ts-field.',
    '',
    'Malformed lines:',
    '  --strict                Exit 1 on first malformed JSON line. Default: warn-and-skip.',
    '',
    'Output:',
    '  --format <json>         Currently only "json". Default: json. (Table mode: Phase 3.)',
    '',
    '  -h, --help              Show this help and exit 0.',
    '',
    'Memory: O(distinct buckets). For minute/hour buckets over typical log volumes this is bounded.',
    '',
    'Examples:',
    '  whoolog histogram --files app.jsonl --by ts --bucket 1h',
    '  whoolog histogram --files app.jsonl --by ts --bucket 5m --where level=error',
    '  whoolog histogram --files "logs/*.jsonl" --by ts --bucket 1d --since 2026-04-01',
    '',
  ].join('\n'));
}

async function run(argv) {
  let values;
  try {
    ({ values } = parseArgs({
      args: argv,
      options: HISTOGRAM_OPTIONS,
      allowPositionals: false,
      strict: true,
    }));
  } catch (err) {
    process.stderr.write(`whoolog: ${err.message}\n`);
    process.exit(2);
  }

  if (values.help) {
    printHistogramHelp();
    process.exit(0);
  }

  // SRC-05: no --files AND stdin TTY → print help, exit 0
  const noFiles = !values.files || values.files.length === 0;
  if (noFiles && process.stdin.isTTY) {
    printHistogramHelp();
    process.exit(0);
  }

  // Required-flag validation BEFORE opening any files (research §3).
  if (!values.by) {
    process.stderr.write('whoolog: histogram requires --by <ts-field>\n');
    process.exit(2);
  }
  if (!values.bucket) {
    process.stderr.write('whoolog: histogram requires --bucket <size>\n');
    process.exit(2);
  }

  const format = values.format || 'json';
  if (format !== 'json') {
    process.stderr.write(`whoolog: --format ${format} is not yet supported (table mode arrives in Phase 3); use --format json\n`);
    process.exit(2);
  }

  // Compile-once (consistent with filter.js pattern).
  let bucketMs, predicates, timePred, getByField;
  try {
    bucketMs = parseBucket(values.bucket);
    predicates = (values.where || []).map(compileWhere);
    timePred = compileTimeFilter({
      since: values.since,
      until: values.until,
      tsField: values['ts-field'],
      keepMissing: values['keep-missing-ts'],
      required: values['ts-required'],
    });
    getByField = compileKey(values.by);
  } catch (err) {
    process.stderr.write(`whoolog: ${err.message}\n`);
    process.exit(err.usage ? 2 : 1);
  }

  const passes = (obj) => {
    for (const p of predicates) if (!p(obj)) return false;
    if (!timePred(obj)) return false;
    return true;
  };

  // Accumulate. Map<number, number> keyed by bucket_start epoch ms.
  // Memory is O(distinct buckets) — for hour buckets over a year that's ≤ 8760 entries.
  const buckets = new Map();

  for await (const source of sources({ files: values.files })) {
    for await (const { obj } of decode(lines(source), { strict: values.strict })) {
      if (!passes(obj)) continue;
      const raw = getByField(obj);
      // Accept either an epoch-ms number or an ISO string. Anything else (or NaN) is silently
      // skipped — same fail-quiet posture as time-filter.js for missing-ts rows.
      let tsMs;
      if (typeof raw === 'number') {
        tsMs = raw;
      } else if (typeof raw === 'string') {
        tsMs = new Date(raw).getTime();
      } else {
        continue;  // missing / null / object / boolean — skip silently
      }
      if (!Number.isFinite(tsMs)) continue;  // invalid date string → NaN
      const key = bucketStart(tsMs, bucketMs);
      buckets.set(key, (buckets.get(key) || 0) + 1);
    }
  }

  // HST-04: sort ascending by bucket_start ms. HST-05: only emit non-empty buckets
  // (the Map already contains only those — there's nothing to skip).
  const sorted = [...buckets.entries()].sort((a, b) => a[0] - b[0]);
  for (const [bucketMsKey, count] of sorted) {
    writeJsonl({ bucket_start: new Date(bucketMsKey).toISOString(), count });
  }
  // Empty result is success — exit 0 implicit.
}

module.exports = { run };
```

**Locked behaviors (verbatim from research):**
- `--by` and `--bucket` are BOTH required; missing either exits 2 BEFORE opening any file (research §3 + Open Question §2).
- `--by` has NO default for histogram (always explicit per Open Question §2). `--ts-field` controls only `--since`/`--until`.
- The accumulator key is `bucketStart(tsMs, bucketMs)` (integer ms). Output uses `new Date(key).toISOString()`.
- Sort comparator: `(a, b) => a[0] - b[0]` — numeric ascending on bucket_start ms. Locale comparators on ISO strings would also work but research locks the numeric form for unambiguous ordering.
- Rows where `getByField(obj)` returns undefined/null/object/boolean → silently skipped. Rows where the value is a string that `new Date()` can't parse → NaN → silently skipped (research §Implementation §3).
- Output row shape MUST be exactly `{ bucket_start, count }` (HST-04). Field order matters for stable JSONL across releases.
- Empty result → exit 0 (CLI-05, consistent with `filter`).
- The `--help` text MUST mention "Buckets with zero matching rows are OMITTED" (HST-05 documentation requirement, Pitfall 11).

**Code Preservation:** Do NOT modify `lib/filter.js`, `lib/source.js`, `lib/decode.js`, `lib/where.js`, `lib/time-filter.js`, `lib/resolve-key.js`, or `lib/format.js`. They are Phase 1 shipped code and must be reused unchanged (research §1).

**Anti-patterns:**
- Do NOT collect rows into an array of full objects before bucketing (Pitfall 1 — defeats streaming). Only the Map is accumulated; rows are processed and discarded.
- Do NOT use `console.log` — `writeJsonl` writes via `process.stdout.write`.
- Do NOT default `--bucket` to anything — research locks it as required.
- Do NOT special-case `seconds`. The `parseBucket` regex rejects `s`; do not work around it here.
- Do NOT serialize the Map to an object before sorting — `[...buckets.entries()]` is the canonical form.
