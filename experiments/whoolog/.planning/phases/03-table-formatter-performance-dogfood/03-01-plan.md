Implement the table formatter (`lib/format.js`) and wire `--format table` into all three subcommands (`filter`, `count`, `histogram`). Locking the formatter contract here also unblocks the dogfood gate (Plan 03-02), which is more useful when its output renders as a table.

```js
'use strict';

// writeJsonl(obj) — re-serialize and emit one JSONL row to stdout.
// Existing Phase 1 export. DO NOT modify the comment block or signature.
function writeJsonl(obj) {
  process.stdout.write(JSON.stringify(obj) + '\n');
}

// formatTable(rows, columns) — pure function returning a rendered fixed-width table.
//
// Contract (FMT-02 + FMT-03):
//   - rows:    array of objects to render. Empty array → return ''.
//   - columns: array of column names (strings); each row is read as row[colName].
//   - Column width = max(colName.length, ...row[colName] string lengths).
//   - Values are coerced via String(v ?? '') so null/undefined render as ''.
//   - Output is: header row, separator row of '-' chars matching widths, then body rows.
//   - Columns are separated by exactly TWO spaces. (Single-space gets visually crowded.)
//   - Each rendered row is right-padded via String#padEnd then joined and newline-terminated.
//   - NEVER emits ANSI escape codes. (V2-OPS-04 defers color; Phase 3 is plain table.)
//
// Pure: no I/O, no globals, deterministic for given input.
function formatTable(rows, columns) {
  if (!Array.isArray(rows) || rows.length === 0) return '';
  if (!Array.isArray(columns) || columns.length === 0) return '';

  const cellStr = (v) => String(v ?? '');
  const widths = columns.map((col) =>
    Math.max(col.length, ...rows.map((r) => cellStr(r[col]).length))
  );
  const pad = (s, w) => cellStr(s).padEnd(w);
  const SEP = '  ';  // two-space column separator

  const header = columns.map((c, i) => pad(c, widths[i])).join(SEP);
  const sepLine = widths.map((w) => '-'.repeat(w)).join(SEP);
  const bodyLines = rows.map((row) =>
    columns.map((c, i) => pad(row[c], widths[i])).join(SEP)
  );
  return [header, sepLine, ...bodyLines].join('\n') + '\n';
}

// writeTable(rows, columns) — convenience writer.
// Empty rows → writes nothing (consistent with empty-result behavior in JSONL mode).
function writeTable(rows, columns) {
  const out = formatTable(rows, columns);
  if (out) process.stdout.write(out);
}

// defaultFormat() — FMT-04. Returns 'table' when stdout is a TTY, else 'json'.
// FORCE_COLOR=1 forces 'table' even when not a TTY (used by tests).
// NO_COLOR is intentionally NOT consulted here: this gate is about layout, not color.
function defaultFormat() {
  if (process.stdout.isTTY === true) return 'table';
  if (process.env.FORCE_COLOR) return 'table';
  return 'json';
}

module.exports = { writeJsonl, formatTable, writeTable, defaultFormat };
```

**Locked details (do not deviate):**
- Two-space column separator (`'  '`). Do NOT change to one space, tab, or pipe characters. Plain `-` separators (no `+` corner chars, no `|` borders). Research §"Don't Hand-Roll" locks the minimal style.
- `cellStr(v) = String(v ?? '')` — `??` (nullish-coalescing) is correct; `||` would coerce `0` and `''` to `''` (the row count `0` would render as empty, hiding zero-counts).
- Empty rows → return `''` (NOT a header-only output). This preserves CLI-05 (empty result → empty stdout, exit 0).
- Empty columns → return `''`. Defensive — keeps `formatTable([], ...)` and `formatTable([{...}], [])` symmetric.
- `defaultFormat` MUST check `process.stdout.isTTY === true` (NOT just truthy) — `isTTY` can be `undefined` in non-TTY contexts and `=== true` is the safe check (Pitfall: TTY detection in tests).
- `FORCE_COLOR` (any truthy value) overrides the TTY check. This is the only way the integration tests can exercise the TTY default branch since `spawnSync` never allocates a PTY (research §Pitfall E).
- Module exports: `module.exports = { writeJsonl, formatTable, writeTable, defaultFormat };` — preserve `writeJsonl` first for diff stability.

**Anti-patterns:**
- Do NOT pull in any third-party library (`cli-table`, `table`, `ansi-colors`, `chalk`). The zero-dep constraint is absolute.
- Do NOT add ANSI escape codes (e.g., `\x1b[1m` for bold headers). Phase 3 scope is plain table; V2-OPS-04 defers color.
- Do NOT use `console.log` — direct `process.stdout.write` only.
- Do NOT compute widths via a multi-pass; the single-pass `Math.max(col.length, ...rows.map(...))` is correct.
- Do NOT convert `rows` into a clone or sorted copy — the caller controls ordering.

```js
const { test } = require('node:test');
const assert = require('node:assert/strict');
const { formatTable, writeJsonl, writeTable, defaultFormat } = require('../lib/format');

test('format: empty rows → empty string', () => {
  assert.equal(formatTable([], ['a', 'b']), '');
});

test('format: empty columns → empty string', () => {
  assert.equal(formatTable([{a: 1}], []), '');
});

test('format: single row, single column — header + sep + body, ends with newline', () => {
  const out = formatTable([{value: 'error'}], ['value']);
  // 'value' is 5 chars, 'error' is 5 chars → width 5
  assert.equal(out, 'value\n-----\nerror\n');
});

test('format: column widths track the LONGER of header vs values', () => {
  // 'count' is 5 chars; values are 1 and 2 chars → header dominates
  const out = formatTable([{count: 3}, {count: 12}], ['count']);
  const lines = out.split('\n');
  assert.equal(lines[0], 'count');
  assert.equal(lines[1], '-----');
  assert.equal(lines[2], '3    ');     // padded to 5
  assert.equal(lines[3], '12   ');     // padded to 5

test('format: column widths track the LONGER of header vs values (values win)', () => {
  // 'k' is 1 char; values are 'short' (5) and 'longer-value' (12) → values dominate
```
