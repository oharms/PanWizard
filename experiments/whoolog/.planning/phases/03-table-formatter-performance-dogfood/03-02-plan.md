```js
#!/usr/bin/env node
'use strict';

// scripts/gen-fixture.js — on-demand JSONL fixture generator for TST-05 perf tests.
//
// Usage: node scripts/gen-fixture.js <outfile> <n_lines>
//
// Writes a deterministic JSONL file: same N produces byte-identical output.
// Used by test/perf.test.js to generate a 1 M-line / ~100 MB fixture before
// the perf assertions run, and to delete it after.
//
// Why a script (not an inline test helper)?
//   1. Makes the fixture re-generable from the command line for manual perf inspection.
//   2. Fixture size (>~100 MB) means we never commit it; this script is the recipe.
//   3. Runs in its own process — keeps the test process memory profile clean.

const fs = require('node:fs');

const [, , outfile, nStr] = process.argv;
if (!outfile || !nStr) {
  process.stderr.write('Usage: node scripts/gen-fixture.js <outfile> <n_lines>\n');
  process.exit(2);
}
const N = parseInt(nStr, 10);
if (!Number.isFinite(N) || N <= 0) {
  process.stderr.write(`gen-fixture: invalid n_lines: ${JSON.stringify(nStr)}\n`);
  process.exit(2);
}

// Locked deterministic dictionaries.
const AGENTS = ['pan-coder', 'pan-optimizer', 'pan-tester', 'general-purpose'];
const MODELS = ['claude-sonnet-4-6', 'claude-opus-4-7', 'claude-haiku-3-5'];
const BASE_TS = Date.parse('2026-01-01T00:00:00.000Z');
const STEP_MS = 60_000;  // 1 minute increments

const stream = fs.createWriteStream(outfile);

// Synchronous write loop — fastest path. Backpressure is handled implicitly:
// Node buffers internally; for ~100 MB total this stays within default heap.
// The drain loop pattern below avoids unbounded buffer growth on slow disks.
function writeLines() {
  for (let i = 0; i < N; i++) {
    const obj = {
      ts: new Date(BASE_TS + i * STEP_MS).toISOString(),
      agent: AGENTS[i % AGENTS.length],
      model: MODELS[i % MODELS.length],
      level: (i % 10 === 0) ? 'error' : 'info',
      input_tokens: 100 + (i % 900),
      output_tokens: 50 + (i % 450),
    };
    const ok = stream.write(JSON.stringify(obj) + '\n');
    if (!ok && i < N - 1) {
      // Backpressure: pause and resume on drain.
      stream.once('drain', () => writeMore(i + 1));
      return;
    }
  }
  stream.end();
}

function writeMore(start) {
  for (let i = start; i < N; i++) {
    const obj = {
      ts: new Date(BASE_TS + i * STEP_MS).toISOString(),
      agent: AGENTS[i % AGENTS.length],
      model: MODELS[i % MODELS.length],
      level: (i % 10 === 0) ? 'error' : 'info',
      input_tokens: 100 + (i % 900),
      output_tokens: 50 + (i % 450),
    };
    const ok = stream.write(JSON.stringify(obj) + '\n');
    if (!ok && i < N - 1) {
      stream.once('drain', () => writeMore(i + 1));
      return;
    }
  }
  stream.end();
}

stream.on('finish', () => process.exit(0));
stream.on('error', (err) => {
  process.stderr.write(`gen-fixture: ${err.message}\n`);
  process.exit(1);
});

writeLines();
```

- `BASE_TS` is `Date.parse('2026-01-01T00:00:00.000Z')` — explicit UTC. Locking the seed ensures determinism.
- `STEP_MS = 60_000` — 1 minute. Over 1 M rows this spans ~694 days, producing distinct hour buckets in the histogram test.
- Field ordering matches the project's existing patterns (ts first, agent/model identity, level + tokens). Do NOT add or rename fields without updating both perf and dogfood test assertions.
- The `level` field is `error` every 10th row (10% error rate), enabling `--where level=error` filter assertions if added later.
- The drain-loop pattern (`writeLines` + `writeMore`) handles backpressure cleanly. A pure synchronous `for` loop would buffer the entire output in memory and risk OOM during fixture generation itself.
- This script is NOT a CLI subcommand — do NOT wire it into `bin/whoolog.js`. It's a project utility script.

**Do NOT add `scripts/` to `.gitignore`** — the script itself IS committed; only the generated `*.jsonl` output is excluded (fixtures live in `os.tmpdir()`, not in the repo).

**Anti-patterns:**
- Do NOT use `Math.random()` or `Date.now()` anywhere. Determinism is the contract.
- Do NOT `require('node:readline')`, `node:util`, or any other module. Pure `node:fs` only.
- Do NOT add a JSON schema validator — the writer controls the schema.
- Do NOT write a header line — JSONL has no header.
- Do NOT pretty-print the JSON — single-line `JSON.stringify(obj)` only.

Create `test/perf.test.js` — the perf integration test suite. Three tests: filter < 10s, histogram < 15s, optional sanity-check filter exits 0 with a non-empty matching set. Uses `spawnSync` with a hard kill 2× the threshold; pass/fail decided by elapsed wall-clock, not by the timeout.

```js
'use strict';

const { test, before, after } = require('node:test');
const assert = require('node:assert/strict');
const fs = require('node:fs');
const path = require('node:path');
const os = require('node:os');
const { spawnSync } = require('node:child_process');
const { BIN } = require('./runCLI');

// Gate: SKIP_PERF=1 in env → skip everything in this file.
// This lets quick local runs avoid the ~30-60s fixture+perf cost.
const SKIP = !!process.env.SKIP_PERF;

const N_LINES = 1_000_000;
const FILTER_LIMIT_MS = 10_000;     // TST-05: filter < 10s
const HISTOGRAM_LIMIT_MS = 15_000;  // TST-05: histogram < 15s
const HARD_KILL_MS = 30_000;        // 2× the largest limit; prevents zombie

let fixturePath;

before(async () => {
  if (SKIP) return;
  fixturePath = path.join(os.tmpdir(), `whoolog-perf-${process.pid}-${Date.now()}.jsonl`);
  const genScript = path.join(__dirname, '..', 'scripts', 'gen-fixture.js');
  const start = Date.now();
  const r = spawnSync(process.execPath, [genScript, fixturePath, String(N_LINES)], {
    timeout: 60_000,  // generation must complete in 60s on commodity hardware
    encoding: 'utf-8',
  });
  if (r.status !== 0) {
    throw new Error(`fixture generation failed: status=${r.status} stderr=${r.stderr}`);
  }
  const elapsed = Date.now() - start;
  // Sanity: file must exist and have ~the right size (>50 MB)
  const stat = fs.statSync(fixturePath);
  if (stat.size < 50 * 1024 * 1024) {
    throw new Error(`fixture too small: ${stat.size} bytes (expected >50 MB)`);
  }
  process.stderr.write(`[perf setup] generated ${N_LINES} lines (${(stat.size / 1024 / 1024).toFixed(1)} MB) in ${elapsed}ms → ${fixturePath}\n`);
});

after(() => {
  if (SKIP || !fixturePath) return;
  try { fs.unlinkSync(fixturePath); } catch (e) { /* ignore */ }
});

test('perf: filter 1 M lines completes in < 10s, exit 0 (TST-05)', { skip: SKIP }, () => {
  const start = Date.now();
  // Use --format json explicitly to keep the streaming O(1) memory path.
  // The filter has no --where, so all 1 M rows match — worst-case throughput.
  const r = spawnSync(process.execPath, [BIN, 'filter', '--files', fixturePath, '--format', 'json'], {
    timeout: HARD_KILL_MS,
    killSignal: 'SIGKILL',
    encoding: 'buffer',  // avoid string decoding overhead in the test process
    maxBuffer: 200 * 1024 * 1024,  // 1 M lines × ~150 bytes ≈ 150 MB → 200 MB headroom
  });
  const elapsed = Date.now() - start;

  // Diagnostic info on failure
  const stderrTail = r.stderr ? r.stderr.toString('utf8').slice(-500) : '';
  assert.equal(r.status, 0, `expected exit 0, got ${r.status} (signal: ${r.signal}); stderr tail: ${stderrTail}`);
  assert.ok(
    elapsed < FILTER_LIMIT_MS,
    `TST-05 filter perf gate: took ${elapsed}ms, limit ${FILTER_LIMIT_MS}ms. Set SKIP_PERF=1 to skip this test on slow machines.`,
  );
  process.stderr.write(`[perf] filter 1M: ${elapsed}ms / ${FILTER_LIMIT_MS}ms limit\n`);
});

test('perf: histogram 1 M lines completes in < 15s, exit 0 (TST-05)', { skip: SKIP }, () => {
  const start = Date.now();
  // 1h buckets over ~694 days → ~16,667 distinct buckets. Map is bounded.
  const r = spawnSync(
    process.execPath,
    [BIN, 'histogram', '--files', fixturePath, '--by', 'ts', '--bucket', '1h', '--format', 'json'],
    {
      timeout: HARD_KILL_MS,
      killSignal: 'SIGKILL',
      encoding: 'buffer',
      maxBuffer: 10 * 1024 * 1024,  // ~16k bucket rows × <100 bytes = ~1.6 MB; 10 MB is huge headroom
    },
  );
  const elapsed = Date.now() - start;

  const stderrTail = r.stderr ? r.stderr.toString('utf8').slice(-500) : '';
  assert.equal(r.status, 0, `expected exit 0, got ${r.status} (signal: ${r.signal}); stderr tail: ${stderrTail}`);
  assert.ok(
    elapsed < HISTOGRAM_LIMIT_MS,
    `TST-05 histogram perf gate: took ${elapsed}ms, limit ${HISTOGRAM_LIMIT_MS}ms. Set SKIP_PERF=1 to skip this test on slow machines.`,
  );
  process.stderr.write(`[perf] histogram 1M: ${elapsed}ms / ${HISTOGRAM_LIMIT_MS}ms limit\n`);
});

test('perf: filter --where level=error on 1 M lines exits 0 with ~10% rows (sanity)', { skip: SKIP }, () => {
  // Sanity check: the fixture has level='error' on every 10th row → ~100,000 matches.
  // This also exercises the predicate hot loop on a real file.
  const r = spawnSync(
    process.execPath,
    [BIN, 'filter', '--files', fixturePath, '--where', 'level=error', '--format', 'json'],
    {
      timeout: HARD_KILL_MS,
      killSignal: 'SIGKILL',
      encoding: 'buffer',
      maxBuffer: 50 * 1024 * 1024,  // 100k matching rows × ~150 bytes = ~15 MB; 50 MB headroom
    },
  );
  assert.equal(r.status, 0, `expected exit 0, got ${r.status}; signal: ${r.signal}`);
  // Count newlines in stdout to verify ~100,000 matches
  const lineCount = r.stdout.toString('utf8').split('\n').filter(Boolean).length;
  // Tolerance: exactly 100,000 expected (every 10th row from 1,000,000 → 100,000)
  assert.equal(lineCount, 100_000, `expected 100000 matching rows, got ${lineCount}`);
});
```

'use strict';

const { test } = require('node:test');
const assert = require('node:assert/strict');
const fs = require('node:fs');
const path = require('node:path');
const { runCLI } = require('./runCLI');

// Project root is one level up from test/ (matches BIN computation in runCLI.js).
const PROJECT_ROOT = path.join(__dirname, '..');
const TOKENS_PATH = path.join(PROJECT_ROOT, '.planning', 'metrics', 'tokens.jsonl');
const TOKENS_EXISTS = fs.existsSync(TOKENS_PATH);

test('dogfood: count --by agent on real tokens.jsonl exits 0 with non-empty output (DOG-01)', { skip: !TOKENS_EXISTS }, () => {
  const r = runCLI(['count', '--files', TOKENS_PATH, '--by', 'agent', '--format', 'json']);
  assert.equal(r.status, 0, `expected exit 0, got ${r.status}: stderr=${r.stderr}`);
  // Non-empty output: at least one JSONL row
  const lines = r.stdout.trim().split('\n').filter(Boolean);
  assert.ok(lines.length >= 1, `expected ≥1 output row, got ${lines.length}; stdout=${r.stdout}`);
  // Each line must parse as JSON with {value, count} shape
